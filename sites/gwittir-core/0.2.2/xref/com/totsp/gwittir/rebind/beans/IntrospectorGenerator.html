104     private int find(MethodWrapper[] search, MethodWrapper match) {
105         for(int i = 0; i < search.length; i++) {
106             if(search[i].equals(match)) {
107                 return i;
108             }
109         }
110         
111         //System.out.println("NO MATCH FOR: " + match.toString());
112         throw new RuntimeException(match.toString());
113     }

115     private MethodWrapper[] findMethods(TreeLogger logger,
116             List introspectables) {
117         HashSet methods = new HashSet();
118         
119         for(Iterator it = introspectables.iterator(); it.hasNext();) {
120             BeanResolver info = (BeanResolver) it.next();
121             logger.branch(logger.DEBUG,
122                     "Method Scanning: " + info.getType().getQualifiedSourceName(), null);
123             
124             try {
125                 
126                 if(info.getProperties().size() ==0) {
127                     continue;
128                 }
129                 
130                 Collection pds = info.getProperties().values();
131                 
132                 for(Iterator pit = pds.iterator(); pit.hasNext(); ) {
133                     Property p = (Property) pit.next();
134                     
135                     if(p.getReadMethod() != null) {
136                         p.getReadMethod().hashWithType = true;
137                         methods.add(p.getReadMethod());
138                     }
139                     
140                     if(p.getWriteMethod() != null) {
141                         p.getWriteMethod().hashWithType =true;
142                         methods.add(p.getWriteMethod());
143                     }
144                 }
145                 
146             } catch(Exception e) {
147                 logger.log(logger.ERROR,
148                         "Unable to introspect class. Is class a bean?", e);
149             }
150         }
151         
152         MethodWrapper[] results = new MethodWrapper[methods.size()];
153         Iterator it = methods.iterator();
154         
155         for(int i = 0; it.hasNext(); i++) {
156             results[i] = (MethodWrapper) it.next();
157         }
158         
159         return results;
160     }

162     public String generate(TreeLogger logger, GeneratorContext context,
163             String typeName) throws UnableToCompleteException {
164         ClassSourceFileComposerFactory cfcf = new ClassSourceFileComposerFactory(packageName,
165                 implementationName);
166         cfcf.addImplementedInterface(typeName);
167         cfcf.addImport("java.util.HashMap");
168         cfcf.addImport(com.totsp.gwittir.client.beans.Method.class
169                 .getCanonicalName());
170         cfcf.addImport(com.totsp.gwittir.client.beans.Property.class
171                 .getCanonicalName());
172         cfcf.addImport(com.totsp.gwittir.client.beans.BeanDescriptor.class
173                 .getCanonicalName());
174         
175         List introspectables = this.getIntrospectableTypes(logger,
176                 context.getTypeOracle());
177         PrintWriter printWriter = context.tryCreate(logger, packageName,
178                 implementationName);
179         
180         if(printWriter == null) {
181             return packageName + "." + implementationName;
182         }
183         
184         SourceWriter writer = cfcf.createSourceWriter(context, printWriter);
185         writer.println("public static final HashMap LOOKUPS = new HashMap();");
186         
187         MethodWrapper[] methods = this.findMethods(logger, introspectables);
188         this.writeMethodsArray(logger, methods, writer);
189         this.writeIntrospectables(logger, introspectables, methods, writer);
190         this.writeResolver(introspectables, writer);
191         writer.println("public BeanDescriptor getDescriptor( Class clazz ){ ");
192         writer.indent();
193         writer.println("return (BeanDescriptor) LOOKUPS.get( clazz ); ");
194         writer.outdent();
195         writer.println("}");
196         writer.println("public BeanDescriptor getDescriptor( Object object ){ ");
197         writer.indent();
198         writer.println(
199                 "return (BeanDescriptor) LOOKUPS.get( this.resolveClass( object) ); ");
200         writer.outdent();
201         writer.println("}");
202         writer.outdent();
203         writer.println("}");
204         
205         context.commit(logger, printWriter);
206         
207         return packageName + "." + implementationName;
208     }

210     private List getIntrospectableTypes(TreeLogger logger, TypeOracle oracle) {
211         ArrayList results = new ArrayList();
212         HashSet resolvers = new HashSet();
213         try {
214             JClassType[] types = oracle.getTypes();
215             JClassType introspectable = oracle.getType(com.totsp.gwittir.client.beans.Introspectable.class
216                     .getCanonicalName());
217             
218             
219             
220             
221             for(int i = 0; i < types.length; i++) {
222                 logger.log(logger.SPAM,
223                         types[i] + " is assignable to " + introspectable + " "
224                         + types[i].isAssignableTo(introspectable)
225                         + " isInterface = " + types[i].isInterface(), null);
226                 
227                 if(types[i].isAssignableTo(introspectable)
228                 && (types[i].isInterface() == null)) {
229                     resolvers.add( new BeanResolver(logger, types[i]));
230                 }
231             }
232             
233             // Do a crazy assed sort to make sure least
234             // assignable types are at the bottom of the list
235             results.addAll( resolvers );
236             boolean swap = true;
237             while(swap){
238                 swap = false;
239                 for( int i = results.size() - 1; i >=0; i --){
240                     BeanResolver type = (BeanResolver) results.get(i);
241                     for( int j = i - 1; j >=0; j-- ){
242                         BeanResolver check = (BeanResolver) results.get(j);
243                         if( type.getType().isAssignableTo( check.getType() ) ){
244                             results.set(i, check);
245                             results.set(j, type );
246                             if( type.toString().equals( check.toString() ) ){
247                                 results.remove(i);
248                             }
249                             swap = true;
250                         }
251                     }
252                 }
253             }
254             
255             
256             
257             
258         } catch(Exception e) {
259             logger.log(logger.ERROR, "Unable to finad Introspectable types.", e);
260         }
261         
262         return results;
263     }

266         if(type.isPrimitive() != null && type.isPrimitive() == JPrimitiveType.INT) {
267             writer.print("((Integer) " + reference + ").intValue()");
268             
269             return true;
270         }
271         
272         if(type.isPrimitive() != null && type.isPrimitive() == JPrimitiveType.LONG) {
273             writer.print("((Long) " + reference + ").longValue()");
274             
275             return true;
276         }
277         
278         if(type.isPrimitive() != null && type.isPrimitive() == JPrimitiveType.FLOAT) {
279             writer.print("((Float) " + reference + ").floatValue()");
280             
281             return true;
282         }
283         
284         if(type.isPrimitive() != null && type.isPrimitive() == JPrimitiveType.DOUBLE) {
285             writer.print("((Double) " + reference + ").doubleValue()");
286             
287             return true;
288         }
289         
290         if(type.isPrimitive() != null && type.isPrimitive() == JPrimitiveType.CHAR) {
291             writer.print("((Character) " + reference + ").charValue()");
292             
293             return true;
294         }
295         
296         if(type.isPrimitive() != null && type.isPrimitive() == JPrimitiveType.BYTE) {
297             writer.print("((Byte) " + reference + ").byteValue()");
298             
299             return true;
300         }
301         
302         if(type.isPrimitive() != null && type.isPrimitive() == JPrimitiveType.BOOLEAN) {
303             writer.print("((Boolean) " + reference + ").booleanValue()");
304             
305             return true;
306         }
307         
308         writer.print("(" + type.getQualifiedSourceName() + ") " + reference);
309         
310         return false;
311     }

313     private void writeBeanDescriptor(TreeLogger logger, BeanResolver info,
314             MethodWrapper[] methods, SourceWriter writer) {
315         writer.println("new BeanDescriptor() { ");
316         writer.indent();
317         writer.println("private HashMap lookup;");
318         writer.println("private Property[] properties;");
319        writer.println("public Property[] getProperties(){");
320         writer.indent();
321         
322         {
323             writer.println("if( this.properties != null ) ");
324             writer.indentln("return this.properties;");
325             writer.println("this.properties = new Property["
326                     + (info.getProperties().size()) + "];");
327             
328             Collection pds = info.getProperties().values();
329             logger.log(logger.SPAM, "" + (pds == null), null);
330             
331             boolean foundClass = false;
332             
333             if(pds != null) {
334                 int i=0;
335                 for(Iterator it = pds.iterator(); it.hasNext(); i++ ) {
336                     Property p = (Property) it.next();
337                     writer.println("{");
338                     writer.indent();
339                     
340                     writer.print("Method readMethod = ");
341                     
342                     if(p.getReadMethod() == null) {
343                         writer.println("null;");
344                     } else {
345                         writer.println("METHODS["
346                                 + this.find(methods, p.getReadMethod() )
347                                 + "];");
348                     }
349                     
350                     writer.print("Method writeMethod = ");
351                     
352                     if(p.getWriteMethod() == null) {
353                         writer.println("null;");
354                     } else {
355                         writer.println("METHODS["
356                                 + this.find(methods, p.getWriteMethod() )
357                                 + "];");
358                     }
359                     
360                     logger.log(logger.SPAM,
361                             p.getName() + " " + p.getType().getQualifiedSourceName(), null);
362                     writer.println("this.properties["
363                             + (i) + "] = new Property( \""
364                             + p.getName() + "\", "
365                             + ((p.getType() != null)
366                             ? p.getType().getQualifiedSourceName() : "Object")
367                             + ".class,  readMethod, writeMethod );");
368                     writer.outdent();
369                     writer.println("}");
370                 }
371             }
372             
373             writer.println("return this.properties;");
374         }
375         
376         writer.outdent();
377         writer.println("} //end getProperties()");
378         writer.println("public Property getProperty( String name ) {");
379         writer.indent();
380         writer.println("Property p = null;");
381         writer.println("if( this.lookup != null ) {");
382         writer.indentln("p = (Property) lookup.get(name); ");
383         writer.println("} else {");
384         writer.indent();
385         writer.println("this.lookup = new HashMap();");
386         writer.println("Property[] props = this.getProperties(); ");
387         writer.println("for( int i=0; i < props.length; i++ ) {");
388         writer.indent();
389         writer.println("this.lookup.put( props[i].getName(), props[i] );");
390         writer.outdent();
391         writer.println("}");
392         writer.println("p = (Property) this.lookup.get(name);");
393         writer.outdent();
394         writer.println("}");
395         writer.println(
396                 "if( p == null ) throw new RuntimeException(\"Couldn't find property \"+name+\" for "+info.getType().getQualifiedSourceName()+ "\");");
397         writer.println("else return p;");
398         writer.outdent();
399         writer.println("}");
400         
401         writer.outdent();
402         writer.print("}");
403     }

405     private void writeIntrospectables(TreeLogger logger, List introspectables,
406             MethodWrapper[] methods, SourceWriter writer) {
407         writer.println("static { ");
408         writer.indent();
409         
410         for(Iterator it = introspectables.iterator(); it.hasNext();) {
411             BeanResolver bean  = (BeanResolver) it.next();
412             
413             logger.branch(logger.DEBUG,
414                     "Introspecting: " + bean.getType().getQualifiedSourceName(), null);
415             
416             try {
417                 if(bean.getProperties().size() == 0) {
418                     continue;
419                 }
420                 
421                 writer.print("LOOKUPS.put( " + bean.getType().getQualifiedSourceName()
422                 + ".class, ");
423                 this.writeBeanDescriptor(logger, bean, methods, writer);
424                 writer.println(" );");
425                 
426             } catch(Exception  e) {
427                 logger.log(logger.ERROR,
428                         "Unable to introspect class. Is class a bean?", e);
429             }
430         }
431         
432         writer.outdent();
433         writer.println("}");
434     }

436     private void writeMethod(TreeLogger logger, MethodWrapper method,
437             SourceWriter writer) {
438         writer.println("new Method(){ ");
439         writer.indent();
440         writer.println("public String getName() {");
441         writer.indentln("return \"" + method.getBaseMethod().getName() + "\";");
442         writer.println(" }");
443         writer.println(
444                 "public Object invoke( Object target, Object[] args ) throws Exception {");
445         writer.indent();
446         writer.println(method.getDeclaringType().getQualifiedSourceName()
447         + " casted =");
448         writer.println("(" + method.getDeclaringType().getQualifiedSourceName()
449         + ") target;");
450         logger.log(logger.SPAM,
451                 "Method: " + method.getBaseMethod().getName() + " "
452                 + method.getBaseMethod().getReturnType().getQualifiedSourceName(), null);
453         
454         if(!(method.getBaseMethod().getReturnType().isPrimitive() == JPrimitiveType.VOID)) {
455             writer.print("return ");
456         }
457         
458         boolean boxed = this.box(method.getBaseMethod().getReturnType(), writer);
459         writer.print("casted." + method.getBaseMethod().getName() + "(");
460         if(method.getBaseMethod().getParameters() != null) {
461             for(int j = 0; j < method.getBaseMethod().getParameters().length; j++) {
462                 this.unbox(method.getBaseMethod().getParameters()[j].getType(), "args[" + j + "]",
463                         writer);
464                 
465                 if(j != (method.getBaseMethod().getParameters().length - 1)) {
466                     writer.print(", ");
467                 }
468             }
469         }
470         
471         writer.print(")");
472         
473         if(boxed) {
474             writer.print(")");
475         }
476         
477         writer.println(";");
478         
479         if(method.getBaseMethod().getReturnType().getQualifiedSourceName().equals("void")) {
480             writer.println("return null;");
481         }
482         
483         writer.outdent();
484         writer.println("}");
485         writer.outdent();
486         writer.println("};");
487     }

489     private void writeMethodsArray(TreeLogger logger,
490             MethodWrapper[] methods, SourceWriter writer) {
491         writer.println("static final Method[] METHODS = new Method["
492                 + methods.length + "];");
493         writer.println("static{");
494         writer.indent();
495         
496         for(int i = 0; i < methods.length; i++) {
497             writer.print("METHODS[" + i + "] =");
498             writeMethod(logger, methods[i], writer);
499         }
500         
501         writer.outdent();
502         writer.println("}//end methods");

505     private void writeResolver(List introspectables,
506             SourceWriter writer) {
507         writer.println("public Class resolveClass(Object object){");
508         writer.indent();
509         
510         for(Iterator it = introspectables.iterator(); it.hasNext();) {
511             BeanResolver type = (BeanResolver) it.next();
512             writer.println("if( object instanceof "
513                     + type.getType().getQualifiedSourceName() + " ) return "
514                     + type.getType().getQualifiedSourceName() + ".class;");
515         }
516         
517         writer.println(
518                 "throw new RuntimeException( \"Object \"+object+\"could not be resolved.\" );");
519         writer.outdent();
520         writer.println("}");
521     }
