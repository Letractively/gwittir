30  public class LogServiceServlet extends RemoteServiceServlet implements LogService{
31      private List<ServerLogService> services;
32      
33      /** Creates a new instance of LogServiceServlet */
34      public LogServiceServlet() {
35      }
36      
37      public void init(ServletConfig config) throws ServletException {
38          super.init(config);
39          this.log( "LogServiceServlet init.");
40          Enumeration<URL> enu;
41          try {
42              enu = LogServiceServlet.class.getClassLoader().getResources("META-INF/services/com.totsp.gwittir.service.ServerLogService");
43              ArrayList<String> providers = new ArrayList<String>();
44              while( enu != null && enu.hasMoreElements() ){
45                  providers.addAll( this.readProviders( enu.nextElement().openStream() ) );
46              }
47              ArrayList<ServerLogService> services = new ArrayList();
48              for( String className : providers ){
49                  try{
50                      services.add( (ServerLogService) Class.forName(className).newInstance() );
51                  } catch(Exception e){
52                      this.log( "Exception creating "+ className, e );
53                  }
54              }
55              this.services = services;
56          } catch (IOException ex) {
57              this.log("Unable to read ServerLogService providers", ex);
58          }
59          
60      }
61      
62      public void log(int level, String logger, String message, String exceptionMessage) {
63          if( services != null && services.size() > 0 ){
64              for( ServerLogService service : services ){
65                  service.log( level, logger, message, exceptionMessage );
66              }
67          } else {
68              this.log( message, new RemoteException( exceptionMessage ) );
69          }
70      }
71      
72      private List<String> readProviders(InputStream is) throws IOException {
73          InputStreamReader reader = new InputStreamReader(is);
74          BufferedReader br = new BufferedReader( reader );
75          String line = br.readLine();
76          ArrayList<String> providers = new ArrayList<String>();
77          while( line != null ){
78              if( line.indexOf("#") != -1 ){
79                  line = line.substring(0, line.indexOf("#") );
80              }
81              line = line.trim();
82              if( line.length() > 0 ){
83                  providers.add( line );
84                  this.log( "Found Log SPI "+ line);
85              }
86              line = br.readLine();
87          }
88          br.close();
89          return providers;
90      }
