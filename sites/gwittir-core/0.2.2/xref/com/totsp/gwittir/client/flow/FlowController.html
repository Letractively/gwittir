44      public static boolean call(Widget sender, String name, Object model) {
45          Widget contextRoot = sender;
46          FlowContext context = (FlowContext) contexts.get(contextRoot);
47  
48          while((context == null) && (contextRoot != null)) {
49              contextRoot = contextRoot.getParent();
50              context = (FlowContext) contexts.get(contextRoot);
51          }
52  
53          if(context == null || !(contextRoot instanceof HasWidgets || contextRoot instanceof HasWidget )) {
54              throw new RuntimeException(
55                  "Unknown destination name for this elements context.");
56          }
57  
58          Object panel = contextRoot;
59          BoundWidget widget = context.get(name);
60  
61          if(widget == null) {
62              return call(contextRoot.getParent(), name, model);
63          }
64          if( model != null ){
65              widget.setModel(model);
66          }
67          BoundWidget old = null;
68          
69  
70          if( panel instanceof SimplePanel ){
71              SimplePanel sp = (SimplePanel) panel;
72              if( sp.getWidget() instanceof BoundWidget ){
73                  old = (BoundWidget) sp.getWidget();
74              }
75              sp.setWidget( (Widget) widget );
76          } else if( panel instanceof HasWidget ){
77              HasWidget hw = (HasWidget) panel;
78              if( hw.getWidget() instanceof BoundWidget ){
79                  old = (BoundWidget) hw.getWidget();
80              }
81              hw.setWidget( (Widget) widget );
82          } else if( panel instanceof HasWidgets ){
83              HasWidgets hw = (HasWidgets) panel;
84              Iterator it = hw.iterator();
85              while(it.hasNext()) {
86                  Object next = it.next();
87  
88                  if(next instanceof BoundWidget) {
89                      old = (BoundWidget) next;
90  
91                      break;
92                  }
93              }
94              hw.clear();
95              hw.add((Widget) widget );
96          }
97  
98          if(FlowController.manager != null) {
99              manager.transition(name, old, widget);
100         }
101         context.fireEvents( name, widget, old );
102 
103         return true;
104     }
