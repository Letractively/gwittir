33  public class ReflectionIE6 extends Reflection {
34      SimplePanel p = new SimplePanel();
35      Image image = new Image();
36      
37      /** Creates a new instance of ReflectionIE6 */
38      public ReflectionIE6() {
39          super();
40          p.add( image );
41          super.setElement( p.getElement() );
42      }
43      
44      public void paint(Image base, int baseWidth, int baseHeight, double height, double opacity){
45          int reflectHeight = (int) Math.round(baseHeight * height);
46          DOM.setStyleAttribute( p.getElement(), "overflow", "hidden");
47          DOM.setStyleAttribute( p.getElement(), "margin", "0px");
48          p.setPixelSize( baseWidth, reflectHeight );
49          
50          image.setUrl( base.getUrl() );
51          image.setWidth( baseWidth+"px");
52          //DOM.setStyleAttribute( image.getElement(), "marginBottom", "-"+(baseHeight-reflectHeight)+"px");
53          DOM.setStyleAttribute( image.getElement(), "filter", "flipv progid:DXImageTransform.Microsoft.Alpha(opacity="+(opacity*100)+", style=1, finishOpacity=0, startx=0, starty=0, finishx=0, finishy="+(height*100)+")" );
54          //DOM.setStyleAttribute( base.getElement(), "cssText", "vertical-align: bottom");
55          Logger.getAnonymousLogger().log( Level.SPAM, image.toString(), null );
56          
57      }
