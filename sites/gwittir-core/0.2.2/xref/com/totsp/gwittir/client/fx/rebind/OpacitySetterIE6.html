20  package com.totsp.gwittir.client.fx.rebind;
21  
22  import com.google.gwt.user.client.DOM;
23  import com.google.gwt.user.client.ui.UIObject;
24  
25  
26  /**
27   * This is an opactiy setter than works by parsing the alpha() block
28   * on the MSIE "filter" CSS property.
29   * @author <a href="mailto:cooper@screaming-penguin.com">Robert "kebernet" Cooper</a>
30   */
31  public class OpacitySetterIE6 extends OpacitySetter {
32      /**
33       * Text of the alpha block.
34       */
35      protected static final String ALPHA = "alpha(";
36  
37      /**
38       * Text of the opacity block.
39       */
40      protected static final String OPACITY = "opacity=";
41  
42      /**
43       * A string containing number characters for comparison.
44       */
45      protected static final String NUMBERS = "1234567890.";
46  
47      /** Creates a new instance of OpacitySetter */
48      public OpacitySetterIE6() {
49      }
50  
51      /**
52       * Gets the current opacity value.
53       * @param o UIObject to inspect.
54       * @return The current opacity between 0.0 and 1.0
55       */
56      public Double getOpacity(UIObject o) {
57          String str = DOM.getStyleAttribute(o.getElement(), "filter");
58          str = parseOrReplace(str, null);
59  
60          if((str == null) || (str.length() == 0)) {
61              return new Double(1.0);
62          } else {
63              return new Double(Integer.parseInt(str) / 100);
64          }
65      }
66  
67      /**
68       * Gets the current opacity value.
69       * @param opacity the new opacity value between 0.0 and 1.0
70       * @param o UIObject to inspect.
71       */
72      public void setOpacity(UIObject o, Double opacity) {
73          String filter = DOM.getStyleAttribute(o.getElement(), "filter");
74  
75          if(filter == null) {
76              filter = "";
77          }
78  
79          if(opacity != null) {
80              filter = parseOrReplace(filter, "" + Math.round(opacity.doubleValue() * 100d));
81              DOM.setStyleAttribute(o.getElement(), "filter", filter);
82          } else {
83              filter = parseOrReplace(filter, "100");
84              DOM.setStyleAttribute(o.getElement(), "filter", filter);
85          }
86      }
87  
88      static String parseOrReplace(String filter, String replace) {
89          int start = filter.indexOf(ALPHA);
90  
91          if(start == -1) {
92              if(replace != null) {
93                  if(filter == null) {
94                      return ALPHA + OPACITY + replace + ")";
95                  } else {
96                      return filter + ALPHA + OPACITY + replace + ")";
97                  }
98              }
99  
100             return null;
101         }
102 
103         int alphaEnd = filter.indexOf(")", start + ALPHA.length());
104         String alphaBlock = filter.substring(start + ALPHA.length(), alphaEnd);
105         int opacityStart = alphaBlock.indexOf(OPACITY);
106 
107         if(opacityStart == -1) {
108             if(replace != null) {
109                 return filter.substring(0, start + ALPHA.length()) + OPACITY + replace + "," + alphaBlock
110                 + filter.substring(alphaEnd, filter.length());
111             } else {
112                 return null;
113             }
114         }
115 
116         opacityStart += (start + ALPHA.length());
117 
118         int opacityEnd = opacityStart + OPACITY.length() + 1;
119 
120         for(; NUMBERS.indexOf(filter.charAt(opacityEnd)) != -1; opacityEnd++) {
121             ;
122         }
123 
124         if(replace == null) {
125             return filter.substring(opacityStart + OPACITY.length(), opacityEnd);
126         } else {
127             return filter.substring(0, opacityStart + OPACITY.length()) + replace
128             + filter.substring(opacityEnd, filter.length());
129         }
130     }
