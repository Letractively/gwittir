32  /**
33   * Renders a Date value to a Label with a Image src of
34   * "[module root]/calendar-icon.gif", which when clicked, will pop up a DatePicker
35   * for selection.
36   * @author rcooper
37   */
38  public class PopupDatePicker extends AbstractBoundWidget
39          implements SourcesCalendarDrawEvents, SourcesCalendarEvents, Renderers, HasFocus, SourcesClickEvents {
40      DatePicker base = new DatePicker();
41      Label label = new Label();
42      Image icon = new Image(GWT.getModuleBaseURL()+"/calendar-icon.gif");
43      FocusPanel fp = new FocusPanel();
44      HorizontalPanel hp = new HorizontalPanel();
45      PopupPanel pp = new PopupPanel(true);
46      /** Creates a new instance of PopupDatePicker */
47      public PopupDatePicker() {
48          this.setRenderer( PopupDatePicker.SHORT_DATE_RENDERER );
49          Binding b = new Binding( label, "value", base, "value");
50          b.setLeft();
51          b.bind();
52          pp.setWidget(base);
53          this.hp.add( this.label );
54          this.hp.add( this.icon );
55          fp.setWidget( hp );
56          this.initWidget( fp );
57          this.setStyleName("gwittir-PopupDatePicker");
58          icon.addClickListener( new ClickListener(){
59              public void onClick(Widget sender) {
60                  fp.setFocus(true);
61                  if( pp.isAttached() ){
62                      pp.hide();
63                  } else {
64                      int width = Window.getClientWidth() + Window.getScrollLeft();
65                      pp.setPopupPosition( getAbsoluteLeft(),
66                              getAbsoluteTop() + getOffsetHeight() );
67                      base.addCalendarListener(new CalendarListener(){
68                          public boolean onDateClicked(Calendar calendar, Date date) {
69                              if(date.getMonth() != base.getRenderDate().getMonth() ||
70                                      date.getYear() != base.getRenderDate().getYear() ){
71                                  return true;
72                              }
73                              pp.hide();
74                              calendar.removeCalendarListener(this);
75                              return true;
76                          }
77                          
78                      });
79                      pp.show();
80                      if( pp.getPopupLeft() + pp.getOffsetWidth()  > width  ){
81                          pp.setPopupPosition( pp.getPopupLeft() + 
82                                  (width - pp.getPopupLeft()- pp.getOffsetWidth() ), pp.getPopupTop() );
83                      }
84                  }
85              }
86          });
87          this.base.addPropertyChangeListener("value", new PropertyChangeListener(){
88              public void propertyChange(PropertyChangeEvent evt) {
89                  changes.firePropertyChange("value", evt.getOldValue(), evt.getNewValue() );
90              }
91              
92          });
93      }
