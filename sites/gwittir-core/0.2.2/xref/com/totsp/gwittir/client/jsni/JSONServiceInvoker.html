12  import com.google.gwt.core.client.JavaScriptObject;
13  import com.google.gwt.user.client.DOM;
14  import com.google.gwt.user.client.Element;
15  import com.google.gwt.user.client.ui.RootPanel;

17  /**
18   * The JSON service invoker is a utility class for invoking JSON-P style web 
19   * services. 
20   * 
21   * It has a single method to invoke a service with a specified callback method.
22   * @author rcooper
23   */
24  public class JSONServiceInvoker {
25      
26      /**
27       * Invokes a JSON-P service. 
28       * 
29       * <b>Note</b>: If you make multiple calls to a service with a single
30       * callback method name, calls made AFTER the first one will effectively be lost,
31       * or you will get multiple resonses sent to the later callback listener.
32       * 
33       * For JSON-P services like the Facebook API, where you can specify different
34       * different callback method names for each call, you should avail yourself of this.
35       * For services like the Flickr Feed service, you should simply be aware of this.
36       * @param url URL to invoke. This will become the "src" attribute of a &lt;script&gt; tag on 
37       * the page.
38       * 
39       * @param callbackMethodName The name of the callback method the service will invoke. This will be added to 
40       * the window/$wnd reference.
41       * @param callback A callback that will be invoked by the service.
42       */
43      public static void invoke( String url, String callbackMethodName, JSONCallback callback) {
44          createCallbackMethod( callbackMethodName, callback );
45          createScriptTag( url );
46      }
47      
48      native static void createCallbackMethod( String callbackMethodName, JSONCallback callback  )
49      /*-{
50          $wnd[callbackMethodName] = function( argument ){
51              var wrapped = @com.totsp.gwittir.client.jsni.JSONServiceInvoker::wrapObject(Lcom/google/gwt/core/client/JavaScriptObject;)(argument);
52              callback.@com.totsp.gwittir.client.jsni.JSONCallback::onJSONResult(Lcom/totsp/gwittir/client/jsni/JavaScriptObjectDecorator;)(wrapped);
53          }
54      }-*/;
55      
56      static JavaScriptObjectDecorator wrapObject( JavaScriptObject object ){
57          return new JavaScriptObjectDecorator(object);
58      }
59  
60      static void createScriptTag( String url ){
61          Element script = DOM.createElement("script");
62          DOM.setElementAttribute( script, "src", url );
63          DOM.setElementAttribute(script, "type", "text/javascript");
64          DOM.appendChild( RootPanel.get().getElement(), script );
65      }
66      
67  }
