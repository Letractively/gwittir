32   *
33   * @author <a href="mailto:cooper@screaming-penguin.com">Robert "kebernet" Cooper</a>
34   */
35  public abstract class AbstractTableWidget extends AbstractBoundWidget {
36      protected BoundWidgetTypeFactory factory;
37  
38      /** Creates a new instance of AbstractTableWidget */
39      public AbstractTableWidget() {
40      }
41  
42      protected BoundWidget createWidget(Binding parent, Field field,
43          Bindable target) {
44          final BoundWidget widget;
45          Binding binding;
46  
47          if(field.getCellProvider() != null) {
48              widget = field.getCellProvider().get();
49          } else {
50              final Property p = Introspector.INSTANCE.getDescriptor(target)
51                                                      .getProperty(field
52                      .getPropertyName());
53              widget = this.factory.getWidgetProvider(field.getPropertyName(),
54                      p.getType()).get();
55  
56              // TODO Figure out some way to make this read only.
57          }
58  
59          binding = new Binding(widget, "value", field.getValidator(),
60                  field.getFeedback(), target, field.getPropertyName(), null, null);
61          widget.setModel(this.getValue());
62  
63          if(field.getRenderer() != null) {
64              widget.setRenderer(field.getRenderer());
65          }
66  
67          if(field.getComparator() != null) {
68              widget.setComparator(field.getComparator());
69          }
70  
71          parent.getChildren().add(binding);
72  
73          return widget;
74      }
75  }
