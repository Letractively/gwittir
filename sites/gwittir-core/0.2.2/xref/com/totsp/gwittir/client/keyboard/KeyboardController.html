74      private void register(final KeyBinding binding, final Object target)
75          throws KeyBindingException {
76          if(binding instanceof SuggestedKeyBinding) {
77              KeyBinding actual = new KeyBinding(
78                      binding.getKey(), binding.isControl(), binding.isAlt(),
79                      binding.isShift());
80  
81              if(bindings.containsKey(actual) || bindings.containsKey(binding)) {
82                  KeyboardController.LOG.log( Level.SPAM, "Contains suggested. Failing silently", null);
83                  return;
84              }
85          } else {
86              SuggestedKeyBinding sb = new SuggestedKeyBinding(
87                      binding.getKey(), binding.isControl(), binding.isAlt(),
88                      binding.isShift());
89              this.unregister(sb);
90  
91              if(this.bindings.containsKey(binding)) {
92                  throw new KeyBindingAlreadyRegisteredException();
93              }
94          }
95  
96          if(binding.getKey() == KeyBinding.F1) {
97              this.clobberHelp();
98          }
99  
100         bindings.put(binding, target);
101 
102         for(int i = 0; i < binding.getKeyBindingEventListeners().length; i++) {
103             binding.getKeyBindingEventListeners()[i].onBind(binding);
104         }
105 
106         for(int i = 0; i < this.getKeyBindingEventListeners().length; i++) {
107             this.getKeyBindingEventListeners()[i].onBind(binding);
108         }
109     }

111     public boolean unregister(final KeyBinding binding) {
112         if(this.bindings.containsKey(binding)) {
113             KeyboardController.LOG.log( Level.SPAM, "Unegistering from controller"+binding, null);
114             for(Iterator it = this.bindings.entrySet().iterator(); it.hasNext();) {
115                 Entry e = (Entry) it.next();
116 
117                 if(e.getValue().equals(binding)) {
118                     KeyBinding b = (KeyBinding) e.getValue();
119                     KeyboardController.LOG.log( Level.SPAM, "Found identity instance"+binding, null);
120                     for(int i = 0; i < b.getKeyBindingEventListeners().length;
121                             i++) {
122                         b.getKeyBindingEventListeners()[i].onUnbind(b);
123                     }
124 
125                     for(
126                         int i = 0;
127                             i < this.getKeyBindingEventListeners().length;
128                             i++) {
129                         this.getKeyBindingEventListeners()[i].onUnbind(b);
130                     }
131                 }
132             }
133         }
134 
135         return this.bindings.remove(binding) != null;
136     }

138     boolean handleEvent(char keyCode, boolean ctrl, boolean alt, boolean shift) {
139         KeyboardController.LOG.log(
140             Level.SPAM, "key event:" + keyCode + " ctrl:" + ctrl + " alt:"
141             + alt, null);
142 
143         //TODO: Do this without so much object creation.
144         KeyBinding check = new KeyBinding(keyCode, ctrl, alt, shift);
145         Object execute = this.bindings.get(check);
146 
147         if(execute == null) {
148             check = new SuggestedKeyBinding(keyCode, ctrl, alt, shift);
149             KeyboardController.LOG.log( Level.SPAM, "Bindings registered: "+ this.bindings.size(), null );
150             execute = this.bindings.get(check);
151             KeyboardController.LOG.log( Level.SPAM, "Found Suggested binding "+ check+ " "+(execute != null), null);
152         }
153 
154         if(execute != null) {
155             if(execute instanceof Task) {
156                 ((Task) execute).run();
157             } else if(execute instanceof Action) {
158                 ((Action) execute).execute(null);
159             } else if(execute instanceof BoundWidget) {
160                 ((BoundWidget) execute).getAction()
161                  .execute((BoundWidget) execute);
162             }
163 
164             return false;
165         }
166 
167         return true;
168     }

175     public AutoMappedBinding findSuggestedMapping(final String match) {
176         if((match == null) || (match.trim().length() == 0)) {
177             return null;
178         }
179 
180         for(int i = 0; i < match.length(); i++) {
181             KeyBinding b = new KeyBinding(
182                     match.toUpperCase().charAt(i), false, true, false);
183             SuggestedKeyBinding sb = new SuggestedKeyBinding(
184                     match.toUpperCase().charAt(i), false, true, false);
185 
186             if(!this.bindings.containsKey(b) && !this.bindings.containsKey(sb)) {
187                 String newHtml = match.substring(0, i);
188                 newHtml += "<u>";
189                 newHtml += match.substring(i, i + 1);
190                 newHtml += "</u>";
191 
192                 if(i+1 < (match.length() - 1)) {
193                     newHtml += match.substring(i+1, match.length());
194                 }
195 
196                 AutoMappedBinding ab = new AutoMappedBinding();
197                 ab.newHtml = newHtml;
198                 ab.binding = sb;
199 
200                 return ab;
201             }
202         }
203 
204         return null;
205     }
