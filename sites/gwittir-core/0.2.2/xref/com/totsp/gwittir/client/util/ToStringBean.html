35  /**
36   * Bean to introspect properties and provide toString() in GWT environ.
37   * Inspired by the Rome bean utils -
38   * https://rome.dev.java.net/source/browse/rome/src/java/com/sun/syndication/feed/impl/.
39   * Powered by the gwittir Introspection API.
40   * 
41   * Example:
42   * 
43   *  public String toString() {
44   *       ToStringBean b = new ToStringBean(this);
45   *
46   *       return b.toString();
47   *   }
48   *
49   *   public boolean equals(Object obj) {
50   *       EqualsHashCodeBean b = new EqualsHashCodeBean(this);
51   *
52   *       return b.equals(obj);
53   *   }
54   * 
55   * @see <code>com.totsp.gwittir.client.beans.AbstractModelBean</code>
56   *
57   * @author ccollins
58   */
59  public class ToStringBean implements Serializable {
60      private static final long serialVersionUID = 1L;
61      private Introspectable bean;
62      private Stack stack;
63  
64      /**
65       * Constructs a new instance of ToStringBean
66       * @param bean Introspectable object to create a toString for.
67       */
68      public ToStringBean(Introspectable bean) {
69          this.bean = bean;
70          this.stack = new Stack();
71      }
72  
73      /**
74       * returns a string value for the bean.
75       * @return returns a string value for the bean.
76       */
77      public String toString() {
78          String[] tsInfo = (String[]) ((stack.isEmpty()) ? null : stack.peek());
79          String prefix;
80  
81          if (tsInfo == null) {
82              String className = GWT.getTypeName(bean);
83              prefix = className.substring(className.lastIndexOf(".") + 1);
84          } else {
85              prefix = tsInfo[0];
86              tsInfo[1] = prefix;
87          }
88  
89          return toString(prefix);
90      }
91  
92      private String toString(String prefix) {
93          StringBuffer sb = new StringBuffer(128);
94  
95          try {
96              BeanDescriptor bd = Introspector.INSTANCE.getDescriptor(bean);
97              Property[] properties = bd.getProperties();
98  
99              if (properties != null) {
100                 for (int i = 0; i < properties.length; i++) {
101                     String pName = properties[i].getName();
102                     Property p = bd.getProperty(pName);
103                     Method m = p.getAccessorMethod();
104 
105                     if (m != null) { // ensure it has a getter method
106 
107                         Object value = m.invoke(bean, null);
108                         printProperty(sb, prefix + "." + pName, value);
109                     }
110                 }
111             }
112         } catch (Exception ex) {
113             sb.append("\n\nEXCEPTION: unable to process " + prefix +
114                 ".toString(): " + ex.getMessage() + "\n");
115         }
116 
117         return sb.toString();
118     }
119 
120     private void printProperty(StringBuffer sb, String prefix, Object value) {
121         if (value == null) {
122             sb.append(prefix).append("=null\n");
123         } else if (ArrayUtils.isArray(value)) {
124             printArrayProperty(sb, prefix, value);
125         } else if (value instanceof Map) {
126             Map map = (Map) value;
127             Iterator i = map.entrySet().iterator();
128 
129             if (i.hasNext()) {
130                 while (i.hasNext()) {
131                     Map.Entry me = (Map.Entry) i.next();
132                     String ePrefix = prefix + "[" + me.getKey() + "]";
133                     Object eValue = me.getValue();
134 
135                     //NEW
136                     String[] tsInfo = new String[2];
137                     tsInfo[0] = ePrefix;
138                     stack.push(tsInfo);
139 
140                     String s = (eValue != null) ? eValue.toString() : "null";
141                     stack.pop();
142 
143                     if (tsInfo[1] == null) {
144                         sb.append(ePrefix).append("=").append(s).append("\n");
145                     } else {
146                         sb.append(s);
147                     }
148                 }
149             } else {
150                 sb.append(prefix).append("=[]\n");
151             }
152         } else if (value instanceof Collection) {
153             Collection collection = (Collection) value;
154             Iterator i = collection.iterator();
155 
156             if (i.hasNext()) {
157                 int c = 0;
158 
159                 while (i.hasNext()) {
160                     String cPrefix = prefix + "[" + (c++) + "]";
161                     Object cValue = i.next();
162 
163                     //NEW
164                     String[] tsInfo = new String[2];
165                     tsInfo[0] = cPrefix;
166                     stack.push(tsInfo);
167 
168                     String s = (cValue != null) ? cValue.toString() : "null";
169                     stack.pop();
170 
171                     if (tsInfo[1] == null) {
172                         sb.append(cPrefix).append("=").append(s).append("\n");
173                     } else {
174                         sb.append(s);
175                     }
176                 }
177             } else {
178                 sb.append(prefix).append("=[]\n");
179             }
180         } else {
181             String[] tsInfo = new String[2];
182             tsInfo[0] = prefix;
183             stack.push(tsInfo);
184 
185             String s = value.toString();
186             stack.pop();
187 
188             if (tsInfo[1] == null) {
189                 sb.append(prefix).append("=").append(s).append("\n");
190             } else {
191                 sb.append(s);
192             }
193         }
194     }
195 
196     private void printArrayProperty(StringBuffer sb, String prefix, Object array) {
197         int length = ArrayUtils.getArrayLength(array);
198 
199         for (int i = 0; i < length; i++) {
200             Object obj = ArrayUtils.getArrayElement(array, i);
201             printProperty(sb, prefix + "[" + i + "]", obj);
202         }
203     }
