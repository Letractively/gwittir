157     /**
158     * Returns the {@link Class} instance for the named class or primitive type.
159     *
160     * @param serializedName the serialized name of a class or primitive type
161     * @param classLoader the classLoader used to load {@link Class}es
162     * @return Class instance for the given type name
163     * @throws ClassNotFoundException if the named type was not found
164     */
165     private static Class<?> getClassFromSerializedName(String serializedName,
166         ClassLoader classLoader) throws ClassNotFoundException {
167         Class<?> value = TYPE_NAMES.get(serializedName);
168 
169         if (value != null) {
170             return value;
171         }
172 
173         return Class.forName(serializedName, false, classLoader);
174     }

176     /**
177      * Straight copy from
178      * {@link com.google.gwt.dev.util.TypeInfo#getSourceRepresentation(Class)} to
179      * avoid runtime dependency on gwt-dev.
180      */
181     private static String printTypeName(Class<?> type) {
182         // Primitives
183         //
184         if (type.equals(Integer.TYPE)) {
185             return "int";
186         } else if (type.equals(Long.TYPE)) {
187             return "long";
188         } else if (type.equals(Short.TYPE)) {
189             return "short";
190         } else if (type.equals(Byte.TYPE)) {
191             return "byte";
192         } else if (type.equals(Character.TYPE)) {
193             return "char";
194         } else if (type.equals(Boolean.TYPE)) {
195             return "boolean";
196         } else if (type.equals(Float.TYPE)) {
197             return "float";
198         } else if (type.equals(Double.TYPE)) {
199             return "double";
200         }
201 
202         // Arrays
203         //
204         if (type.isArray()) {
205             Class<?> componentType = type.getComponentType();
206 
207             return printTypeName(componentType) + "[]";
208         }
209 
210         // Everything else
211         //
212         return type.getName().replace('$', '.');
213     }

215     /**
216     * Used to determine whether the specified interface name is implemented by
217     * the service class. This is done without loading the class (for security).
218     */
219     private static boolean implementsInterface(Class<?> service, String intfName) {
220         synchronized (serviceToImplementedInterfacesMap) {
221             // See if it's cached.
222             //
223             Set<String> interfaceSet = serviceToImplementedInterfacesMap.get(service);
224 
225             if (interfaceSet != null) {
226                 if (interfaceSet.contains(intfName)) {
227                     return true;
228                 }
229             } else {
230                 interfaceSet = new HashSet<String>();
231                 serviceToImplementedInterfacesMap.put(service, interfaceSet);
232             }
233 
234             if (!service.isInterface()) {
235                 while ((service != null) &&
236                         !RemoteServiceServlet.class.equals(service)) {
237                     Class<?>[] intfs = service.getInterfaces();
238 
239                     for (Class<?> intf : intfs) {
240                         if (implementsInterfaceRecursive(intf, intfName)) {
241                             interfaceSet.add(intfName);
242 
243                             return true;
244                         }
245                     }
246 
247                     // did not find the interface in this class so we look in the
248                     // superclass
249                     //
250                     service = service.getSuperclass();
251                 }
252             } else {
253                 if (implementsInterfaceRecursive(service, intfName)) {
254                     interfaceSet.add(intfName);
255 
256                     return true;
257                 }
258             }
259 
260             return false;
261         }
262     }

264     /**
265      * Only called from implementsInterface().
266      */
267     private static boolean implementsInterfaceRecursive(Class<?> clazz,
268         String intfName) {
269         assert (clazz.isInterface());
270 
271         if (clazz.getName().equals(intfName)) {
272             return true;
273         }
274 
275         // search implemented interfaces
276         Class<?>[] intfs = clazz.getInterfaces();
277 
278         for (Class<?> intf : intfs) {
279             if (implementsInterfaceRecursive(intf, intfName)) {
280                 return true;
281             }
282         }
283 
284         return false;
285     }

287     private static String formatMethodNotFoundErrorMessage(
288         Class<?> serviceIntf, String serviceMethodName,
289         Class<?>[] parameterTypes) {
290         StringBuffer sb = new StringBuffer();
291 
292         sb.append("Could not locate requested method '");
293         sb.append(serviceMethodName);
294         sb.append("(");
295 
296         for (int i = 0; i < parameterTypes.length; ++i) {
297             if (i > 0) {
298                 sb.append(", ");
299             }
300 
301             sb.append(printTypeName(parameterTypes[i]));
302         }
303 
304         sb.append(")'");
305 
306         sb.append(" in interface '");
307         sb.append(printTypeName(serviceIntf));
308         sb.append("'");
309 
310         return sb.toString();
311     }

314     /**
315    * Returns a string that encodes the results of an RPC call. Private overload
316    * that takes a flag signaling the preamble of the response payload.
317    *
318    * @param object the object that we wish to send back to the client
319    * @param wasThrown if true, the object being returned was an exception thrown
320    *          by the service method; if false, it was the result of the service
321    *          method's invocation
322    * @return a string that encodes the response from a service method
323    * @throws SerializationException if the object cannot be serialized
324    */
325   public static String encodeResponse(Class<?> responseClass, Object object,
326       boolean wasThrown, SerializationPolicy serializationPolicy)
327       throws SerializationException {
328 
329     ServerSerializationStreamWriter stream = new ServerSerializationStreamWriter(
330         serializationPolicy);
331 
332     stream.prepareToWrite();
333     if (responseClass != void.class) {
334       stream.serializeValue(object, responseClass);
335     }
336 
337     String bufferStr = (wasThrown ? "//EX" : "//OK") + stream.toString();
338     return bufferStr;
339   }
