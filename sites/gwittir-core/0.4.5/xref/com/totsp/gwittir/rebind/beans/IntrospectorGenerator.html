116     private int find(MethodWrapper[] search, MethodWrapper match) {
117         for (int i = 0; i < search.length; i++) {
118             if (search[i].equals(match)) {
119                 return i;
120             }
121         }
122 
123         //System.out.println("NO MATCH FOR: " + match.toString());
124         throw new RuntimeException(match.toString());
125     }

127     private MethodWrapper[] findMethods(TreeLogger logger, List introspectables) {
128         HashSet methods = new HashSet();
129 
130         for (Iterator it = introspectables.iterator(); it.hasNext();) {
131             BeanResolver info = (BeanResolver) it.next();
132             logger.branch(TreeLogger.DEBUG, "Method Scanning: " + info.getType().getQualifiedSourceName(), null);
133 
134             try {
135                 if (info.getProperties().size() == 0) {
136                     continue;
137                 }
138 
139                 Collection<Property> pds = info.getProperties().values();
140 
141                 for ( Property p : pds) {
142                    
143                     if (p.getReadMethod() != null) {
144                         p.getReadMethod().hashWithType = true;
145                         methods.add(p.getReadMethod());
146                     }
147 
148                     if (p.getWriteMethod() != null) {
149                         p.getWriteMethod().hashWithType = true;
150                         methods.add(p.getWriteMethod());
151                     }
152                 }
153             } catch (Exception e) {
154                 logger.log(TreeLogger.ERROR, "Unable to introspect class. Is class a bean?", e);
155             }
156         }
157 
158         MethodWrapper[] results = new MethodWrapper[methods.size()];
159         Iterator it = methods.iterator();
160 
161         for (int i = 0; it.hasNext(); i++) {
162             results[i] = (MethodWrapper) it.next();
163         }
164 
165         return results;
166     }

168     public String generate(TreeLogger logger, GeneratorContext context, String typeName)
169         throws UnableToCompleteException {
170         try {
171             this.objectType = context.getTypeOracle().getType("java.lang.Object");
172         } catch (NotFoundException ex) {
173             logger.log(TreeLogger.ERROR, typeName, ex);
174 
175             return null;
176         }
177 
178         List<BeanResolver> introspectables = this.getIntrospectableTypes(logger, context.getTypeOracle());
179 
180         MethodWrapper[] methods = this.findMethods(logger, introspectables);
181 
182         ClassSourceFileComposerFactory mcf = new ClassSourceFileComposerFactory(
183                 this.packageName,
184                 this.methodsImplementationName
185             );
186         mcf.addImport(com.totsp.gwittir.client.beans.Method.class.getCanonicalName());
187 
188         PrintWriter methodsPrintWriter = context.tryCreate(logger, this.packageName, this.methodsImplementationName);
189 
190         if (methodsPrintWriter != null) {
191             SourceWriter methodsWriter = mcf.createSourceWriter(context, methodsPrintWriter);
192             this.writeMethods(logger, methods, methodsWriter);
193             methodsWriter.println("}");
194             context.commit(logger, methodsPrintWriter);
195         }
196 
197         ClassSourceFileComposerFactory cfcf = new ClassSourceFileComposerFactory(
198                 this.packageName, this.implementationName
199             );
200         cfcf.addImplementedInterface(typeName);
201         cfcf.addImport("java.util.HashMap");
202         cfcf.addImport(com.totsp.gwittir.client.beans.Method.class.getCanonicalName());
203         cfcf.addImport(com.totsp.gwittir.client.beans.Property.class.getCanonicalName());
204         cfcf.addImport(com.totsp.gwittir.client.beans.BeanDescriptor.class.getCanonicalName());
205 
206         PrintWriter printWriter = context.tryCreate(logger, packageName, implementationName);
207 
208         if (printWriter == null) {
209             return packageName + "." + implementationName;
210         }
211 
212         SourceWriter writer = cfcf.createSourceWriter(context, printWriter);
213         this.writeIntrospectables(logger, introspectables, methods, writer);
214         this.writeResolver(introspectables, writer);
215 
216         writer.println("public BeanDescriptor getDescriptor( Object object ){ ");
217         writer.indent();
218         writer.println("if( object == null ) throw new NullPointerException(\"Attempt to introspect null object\");");
219         writer.println("if( object instanceof "+SelfDescribed.class.getCanonicalName()+" ) return ((SelfDescribed)object).__descriptor();");
220         for (BeanResolver resolver : introspectables) {
221             writer.println("if( object instanceof " + resolver.getType().getQualifiedSourceName() + " ) {");
222             writer.indent();
223 
224             String name = resolver.getType().getQualifiedSourceName().replaceAll("\\.", "_");
225             logger.log(TreeLogger.DEBUG, "Writing : " + name, null);
226             writer.print("return " + name + " == null ? " + name + " = ");
227             this.writeBeanDescriptor(logger, resolver, methods, writer);
228             writer.print(": " + name + ";");
229             writer.outdent();
230             writer.println("}");
231         }
232 
233         writer.println(" throw new IllegalArgumentException(\"Unknown type\" + object.getClass() ); ");
234         writer.outdent();
235         writer.println("}");
236         writer.outdent();
237         writer.println("}");
238 
239         context.commit(logger, printWriter);
240 
241         return packageName + "." + implementationName;
242     }

310     private List<BeanResolver> getFileDeclaredTypes(TreeLogger logger, TypeOracle oralce ) throws UnableToCompleteException {
311         ArrayList<BeanResolver> results = new ArrayList<BeanResolver>();
312         ClassLoader ctxLoader = Thread.currentThread().getContextClassLoader();
313         try{
314             Enumeration<URL> introspections = ctxLoader.getResources( "gwittir-introspection.properties");
315             while(introspections.hasMoreElements() ){
316                 URL propsUrl = introspections.nextElement();
317                 logger.log(TreeLogger.Type.INFO, "Loading: "+propsUrl.toString() );
318                 Properties props = new Properties();
319                 props.load( propsUrl.openStream() );
320                 for(Entry entry : props.entrySet()){
321                     String className = entry.getKey().toString();
322                     String[] includedProps = entry.getValue().toString().split(",");
323                     JClassType type = oralce.findType(className);
324                     if(type == null ){
325                         logger.log(TreeLogger.Type.ERROR, "Unable to find type "+className+" declared in "+propsUrl);
326                         throw new UnableToCompleteException();
327                     }
328                     results.add( new BeanResolver(logger, type, includedProps));
329                 }
330 
331             }
332 
333         } catch(IOException ioe){
334             logger.log(TreeLogger.Type.WARN, "Exception looking for properties files", ioe);
335         }
336         return results;
337     }

339     private boolean isIntrospectable(TreeLogger logger, JType type){
340         if( type == null ) return false;
341         JClassType ct = type.isClassOrInterface();
342         if(ct != null){
343             if( ct.getAnnotation(Introspectable.class) != null ){
344                 return true;
345             }
346             for( JClassType iface : ct.getImplementedInterfaces() ){
347                 if( isIntrospectable(logger, iface) ){
348 
349                     return true;
350                 }
351             }
352             if( isIntrospectable(logger, ct.getSuperclass() ) ){
353                 return true;
354             }
355         }
356         return false;
357     }

530     private void writeMethod(TreeLogger logger, MethodWrapper method, SourceWriter writer) {
531         JType ptype =  this.resolveType(method.getDeclaringType());
532         
533         writer.println("new Method(){ ");
534         writer.indent();
535         writer.println("public String getName() {");
536         writer.indentln("return \"" + method.getBaseMethod().getName() + "\";");
537         writer.println(" }");
538         writer.println("public Object invoke( Object target, Object[] args ) throws Exception {");
539         writer.indent();
540         writer.println(ptype.getQualifiedSourceName() + " casted =");
541         writer.println("(" + ptype.getQualifiedSourceName() + ") target;");
542         logger.log(
543                 TreeLogger.SPAM,
544                 "Method: " + method.getBaseMethod().getName() + " " +
545                 method.getBaseMethod().getReturnType().getQualifiedSourceName(),
546                 null
547             );
548 
549         if (!(method.getBaseMethod().getReturnType().isPrimitive() == JPrimitiveType.VOID)) {
550             writer.print("return ");
551         }
552 
553         JType type = this.resolveType(method.getBaseMethod().getReturnType());
554 
555         boolean boxed = this.box(type, writer);
556         writer.print("casted." + method.getBaseMethod().getName() + "(");
557 
558         if (method.getBaseMethod().getParameters() != null) {
559             for (int j = 0; j < method.getBaseMethod().getParameters().length; j++) {
560                 JType arg = this.resolveType(method.getBaseMethod().getParameters()[j].getType());
561                 
562                 this.unbox(arg, "args[" + j + "]", writer);
563 
564                 if (j != (method.getBaseMethod().getParameters().length - 1)) {
565                     writer.print(", ");
566                 }
567             }
568         }
569 
570         writer.print(")");
571 
572         if (boxed) {
573             writer.print(")");
574         }
575 
576         writer.println(";");
577 
578         if (method.getBaseMethod().getReturnType().getQualifiedSourceName().equals("void")) {
579             writer.println("return null;");
580         }
581 
582         writer.outdent();
583         writer.println("}");
584         writer.outdent();
585         writer.println("};");
586     }

612     private JType resolveType(final JType type){
613         JType ret = type;
614         JParameterizedType pt = type.isParameterized();
615         if( pt != null ){
616             ret = pt.getRawType();
617         }
618         JTypeParameter tp = ret.isTypeParameter();
619         if( tp != null ){
620             ret =tp.getBaseType();
621         }
622         return ret;
