194     protected JMethod getAsyncMethod( JClassType asyncType, JMethod syncMethod ) throws UnableToCompleteException{
195         for(JMethod sourceMethod : asyncType.getMethods() ){
196 
197             if( sourceMethod.getName(). equals(syncMethod.getName() )
198                     && sourceMethod.getParameters().length == syncMethod.getParameters().length + 1){
199                     boolean notIt = false;
200                     for( int i=0; i < syncMethod.getParameters().length; i++ ){
201                         if( !syncMethod.getParameters()[i].getType().equals( sourceMethod.getParameters()[i].getType() ) ){
202                             notIt = true;
203                             break;
204                         }
205                     }
206                     if( notIt ){
207                         continue;
208                     } else {
209                         return sourceMethod;
210                     }
211             }
212         }
213         throw new UnableToCompleteException();
214 
215     }

217     private String getJSNIType(JType type){
218        if (type.isArray() != null ){
219             return type.getJNISignature().replaceAll("/", ".");
220        } else if( type.getJNISignature().length() > 1 ){
221            return type.getQualifiedSourceName();
222        } else {
223            return type.getJNISignature();

227     private String getWriteCallName(JType type){
228         if( getJSNIType(type).length() == 1){
229             String typeName = type.toString();
230             return "write"+typeName.substring(0,1).toUpperCase() + typeName.substring(1, typeName.length() );
231         } else if( getJSNIType(type).equals("java.lang.String")  ){
232             return "writeString";
233         } else {
234             return "writeObject";
235         }
