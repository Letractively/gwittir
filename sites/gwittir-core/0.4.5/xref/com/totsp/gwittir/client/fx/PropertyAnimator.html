44      /** Creates a new instance of PropertyAnimator */
45      public PropertyAnimator(Object target, String propertyName,
46          Object initialValue, Object finalValue, MutationStrategy strategy,
47          int duration, int stepTime, AnimationFinishedCallback callback) {
48          this.target = target;
49          this.property = Introspector.INSTANCE.getDescriptor(target)
50                                               .getProperty(propertyName);
51  
52          if((property == null)
53                  || (
54                      (property.getAccessorMethod() == null)
55                      && (initialValue == null)
56                  ) || (property.getMutatorMethod() == null)) {
57              throw new RuntimeException("Invalid property.");
58          }
59  
60          try {
61              this.initialValue = (initialValue == null)
62                  ? property.getAccessorMethod().invoke(target, null) : initialValue;
63          } catch(Exception e) {
64              throw new RuntimeException(e);
65          }
66  
67          this.finalValue = finalValue;
68          this.strategy = strategy;
69          this.callback = callback;
70          this.duration = duration;
71          this.stepTime = stepTime;
72      }

106     void invoke(Method method, Object value) throws Exception {
107         if(this.property.getType() == null) {
108             method.invoke(this.target, (Object[]) value);
109         } else {
110             Object[] args = { value };
111             method.invoke(this.target, args);
112         }
113     }

115     public void start() {
116         final long startTime = System.currentTimeMillis();
117         final PropertyAnimator instance = this;
118 
119         try {
120             invoke(this.property.getMutatorMethod(), this.initialValue);
121         } catch(Exception e) {
122             if(this.callback != null) {
123                 this.callback.onFailure(this, e);
124 
125                 return;
126             }
127         }
128 
129          timer= new Timer() {
130                 private long lastStep = System.currentTimeMillis();
131                 public void run() {
132                     try {
133                         double percentComplete = (double) (
134                                 System.currentTimeMillis() - startTime
135                             ) / (double) duration;
136                         Object newValue = strategy.mutateValue(initialValue,
137                                 finalValue, percentComplete);
138 
139                         if(percentComplete < 1) {
140                             invoke(property.getMutatorMethod(), newValue);
141                         } else {
142                             invoke(property.getMutatorMethod(), finalValue);
143 
144                             if(callback != null) {
145                                 callback.onFinish(instance);
146                             }
147                             
148                             timer.cancel();
149                             return;
150                         }
151                     } catch(Exception e) {
152                         if(callback != null) {
153                             callback.onFailure(instance, e);
154                             timer.cancel();
155 
156                             return;
157                         }
158                     }
159                     if( lastStep - System.currentTimeMillis() > stepTime * 2 ){
160                         stepTime = (int) lastStep - (int) System.currentTimeMillis();
161                     }
162                     timer.schedule(stepTime);
163                 }
164                
165             };
166             this.timer.schedule( this.stepTime );
167         
168     }
169     
170     public void cancel(){
171         timer.cancel();
172         if( this.callback != null ){
173             callback.onFailure(this, null);
174         }
175     }
