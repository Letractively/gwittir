41  /**
42   * A wrapper around a Calendar that provides Month/Year navigation for selection.
43   * @author cooper
44   */
45  public class DatePicker extends AbstractBoundWidget<Date> implements
46          SourcesCalendarDrawEvents, SourcesCalendarEvents {
47      private static final String VALUE_PROPERTY_NAME = "value";
48      
49      private Calendar calendar = new Calendar();
50      private HorizontalPanel hp = new HorizontalPanel();
51      private List years = new ArrayList();
52      private ListBox year = new ListBox();
53      private ListBox month;
54      /** Creates a new instance of DatePicker */
55      public DatePicker() {
56          VerticalPanel vp = new VerticalPanel();
57          Label back = new Label("<<");
58          back.addClickListener( new ClickListener(){
59              public void onClick(Widget sender) {
60                  Date current = calendar.getRenderDate();
61                  if( current.getMonth() -1 >= 0 ){
62                      current = new Date( current.getYear(), current.getMonth() - 1 , 1);
63                  } else {
64                      LOG.log( Level.SPAM, "current Year"+ current.getYear(), null );
65                      current = new Date(current.getYear() - 1, 12 , 0 );
66                      LOG.log( Level.SPAM, "new date"+ current.getYear(), null );
67                  }
68                  LOG.log(Level.SPAM, current.toString(), null);
69                  calendar.setRenderDate( current );
70              }
71          });
72          hp.add( back );
73          month = new ListBox();
74          final ArrayList months = new ArrayList();
75          for( String monthString : Calendar.MONTHS_OF_YEAR_SHORT ){
76              months.add(monthString);
77          }
78          month.setOptions(months);
79          month.addPropertyChangeListener(DatePicker.VALUE_PROPERTY_NAME, new PropertyChangeListener(){
80              public void propertyChange(PropertyChangeEvent propertyChangeEvent) {
81                  Date current = calendar.getRenderDate();
82                  current = new Date( current.getYear(),
83                          indexOf( Calendar.MONTHS_OF_YEAR_SHORT, ""+ Converter.FROM_COLLECTION_CONVERTER.convert((Collection)propertyChangeEvent.getNewValue()) )
84                          ,1);
85                  calendar.setRenderDate( current );
86              }
87              
88          });
89          
90          this.updateMonth();
91          hp.add( this.month );
92          this.updateYears();
93          GWT.log("INit to:"+Integer.toString( calendar.getRenderDate().getYear() + 1900 ), null);
94          this.year.setValue( year.single(Integer.toString( calendar.getRenderDate().getYear() + 1900 )) );
95          hp.add( this.year );
96          this.year.addPropertyChangeListener(VALUE_PROPERTY_NAME, new PropertyChangeListener(){
97              public void propertyChange(PropertyChangeEvent propertyChangeEvent) {
98                  if( propertyChangeEvent.getNewValue() == null || ((Collection)propertyChangeEvent.getNewValue()).size() == 0){
99                      return;
100                 }
101                 Date current = calendar.getRenderDate();
102                 current = new Date( Integer.parseInt( ((Collection)propertyChangeEvent.getNewValue()).iterator().next().toString() ) -1900, current.getMonth(), 1 );
103                 LOG.log(Level.SPAM, current.toString(), null );
104                 calendar.setRenderDate( current );
105             }
106             
107         });
108         final Label next = new Label(">>");
109         next.addClickListener( new ClickListener(){
110             public void onClick(Widget sender) {
111                 Date current = calendar.getRenderDate();
112                 if( current.getMonth() + 1 < 12 ){
113                     current = new Date( current.getYear(), current.getMonth() + 1 , 1);
114                 } else {
115                     LOG.log( Level.SPAM, "current Year"+ current.getYear(), null );
116                     current = new Date(current.getYear() + 1, 1, 0 );
117                     LOG.log( Level.SPAM, "new date"+ current.getYear(), null );
118                 }
119                 LOG.log(Level.SPAM, current.toString(), null);
120                 calendar.setRenderDate( current );
121             }
122         });
123         hp.add( next );
124         vp.add( hp );
125         vp.add( this.calendar );
126         this.calendar.addPropertyChangeListener("renderDate", new PropertyChangeListener(){
127             public void propertyChange(PropertyChangeEvent propertyChangeEvent) {
128                 updateYears();
129                 updateMonth();
130             }
131             
132         });
133         this.calendar.addPropertyChangeListener(VALUE_PROPERTY_NAME, new PropertyChangeListener(){
134             public void propertyChange(PropertyChangeEvent evt) {
135                 changes.firePropertyChange(VALUE_PROPERTY_NAME, evt.getOldValue(), evt.getNewValue()  );
136             }
137             
138         });
139         super.initWidget( vp );
140         this.setStyleName("gwittir-DatePicker");
141         
142     }

208     private void updateYears(){
209         this.years.clear();
210         int year = this.calendar.getRenderDate().getYear() + 1900;
211         for( int i = year - 20; i < year + 20; i++){
212             this.years.add( Integer.toString( i ) );
213         }
214         this.year.setOptions(this.years);
215         this.year.setValue( this.year.single(Integer.toString( this.calendar.getRenderDate().getYear() + 1900 )) );
216         LOG.log( Level.SPAM, Integer.toString( this.calendar.getRenderDate().getYear() + 1900 ), null );
217         
218     }
219     private void updateMonth(){
220         this.month.setValue( this.month.single(Calendar.MONTHS_OF_YEAR_SHORT[ this.calendar.getRenderDate().getMonth() ]));
221     }
222     private int indexOf( final String[] values, final String check ){
223         int index = -1;
224         for( int i=0; i < values.length; i++ ){
225             if( values[i].equals( check ) ){
226                 index = i;
227                 break;
228             }
229             
230         }
231         return index;
232     }
