34  public class HistoryTokenizer {
35      private Map<String, String> tokens = new HashMap<String, String>();
36      private boolean isInTransaction = false;
37  
38      public HistoryTokenizer() {
39          this(History.getToken());
40      }
41      
42      public HistoryTokenizer(Map<String, String> tokens){
43      	this.tokens = tokens;
44      }
45  
46      public HistoryTokenizer(String anchor) {
47          if (anchor != null && anchor.trim().length() > 0) {
48              String[] values = anchor.split("&");
49              
50              for (String value : values) {
51                  String[] nameVal = value.split("=");
52  
53                  if ((nameVal.length == 1) && (nameVal[0].length() > 0)) {
54                      tokens.put(nameVal[0], "");
55                  } else {
56                      tokens.put(nameVal[0].replaceAll("u!038;", "&")
57                                           .replaceAll("u!061;", "="),
58                          nameVal[1].replaceAll("u!038;", "&")
59                                    .replaceAll("u!061;", "="));
60                  }
61              }
62          }
63      }
64  
65      public Map<String, String> getTokensMap(){
66      	return tokens;
67      }
68      
69      public Set<Entry<String, String>> getEntries() {
70          return tokens.entrySet();
71      }
72  
73      public void setToken(String token, String value) {
74          this.tokens.put(token, value);
75  
76          if (!this.isInTransaction) {
77              this.store();
78          }
79      }
80  
81      public String getToken(String token) {
82          return tokens.get(token);
83      }
84  
85      public void begin() {
86          this.isInTransaction = true;
87      }
88  
89      public void commit() {
90          this.isInTransaction = false;
91          this.store();
92      }
93  
94      @Override
95      public boolean equals(Object obj) {
96          if (obj == null) {
97              return false;
98          }
99  
100         if (getClass() != obj.getClass()) {
101             return false;
102         }
103 
104         final HistoryTokenizer other = (HistoryTokenizer) obj;
105 
106         if ((this.tokens != other.tokens) &&
107                 ((this.tokens == null) || !this.tokens.equals(other.tokens))) {
108             return false;
109         }
110 
111         return true;
112     }
113 
114     @Override
115     public int hashCode() {
116         int hash = 3;
117         hash = (97 * hash) +
118             ((this.tokens != null) ? this.tokens.hashCode() : 0);
119 
120         return hash;
121     }
122 
123     public String tokenize() {
124         StringBuilder sb = new StringBuilder();
125 
126         for (Entry<String, String> entry : this.tokens.entrySet()) {
127             sb = sb.append(entry.getKey().replaceAll("&", "u!038;")
128                                 .replaceAll("=", "u!061;"));
129 
130             if (entry.getValue().length() > 0) {
131                 sb = sb.append("=")
132                        .append(entry.getValue().replaceAll("&", "u!038;")
133                                     .replaceAll("=", "u!061;"));
134             }
135 
136             sb = sb.append("&");
137         }
138 
139         return sb.toString();
140     }
141 
142     private void store() {
143         History.newItem(this.tokenize());
144     }
