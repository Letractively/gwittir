12  abstract class HuffTree implements Comparable{
13  
14    int frequency;
15  
16    public int getFrequency(){
17      return frequency;
18    }//end getFrequency
19  
20    //This method compares this object to an object whose
21    // reference is received as an incoming parameter.
22    // The method guarantees that sorting processes that
23    // depend on this method, such as TreeSet objects, will
24    // sort the objects into a definitive order.
25  
26    // If the frequency values of the two objects are
27    // different, the sort is based on the frequency values.
28    // If the frequency values are equal, the objects are
29    // sorted based on their relative hashCode values.
30    // Thus, if the same two objects with the same frequency
31    // value are compared two or more times during the
32    // execution of the program, those two objects will
33    // always be sorted into the same order.  There is no
34    // chance of an ambiguous tie as to which object
35    // should be first except for the case where an object
36    // is compared to itself using two references to the
37    // same object.
38    public int compareTo(Object obj){
39      HuffTree theTree = (HuffTree)obj;
40      if (frequency == theTree.frequency){
41        //The objects are in a tie based on the frequency
42        // value.  Return a tiebreaker value based on the
43        // relative hashCode values of the two objects.
44        return (hashCode() - theTree.hashCode());
45      }else{
46        //Return negative or positive as this frequency is
47        // less than or greater than the frequency value of
48        // the object referred to by the parameter.
49        return frequency - theTree.frequency;
50      }//end else
51    }//end compareTo
