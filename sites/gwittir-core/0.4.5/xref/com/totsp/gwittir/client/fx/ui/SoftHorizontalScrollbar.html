43  public class SoftHorizontalScrollbar extends Composite {
44      FocusPanel barTarget = new FocusPanel();
45      FocusPanel higherTarget = new FocusPanel();
46      FocusPanel lowerTarget = new FocusPanel();
47      ScrollListener scrollListener = new ScrollListener() {
48              public void onScroll(Widget widget, int scrollLeft, int scrollTop) {
49                  refresh();
50              }
51          };
52  
53      SimplePanel bar = new SimplePanel();
54      SimplePanel base;
55      SimplePanel higher = new SimplePanel();
56      SimplePanel lower = new SimplePanel();
57      SoftScrollArea target;
58      private MouseListener barListener = new MouseListenerAdapter() {
59              boolean inDrag = false;
60              int start;
61  
62              public void onMouseLeave(Widget sender) {
63                  inDrag = false;
64              }
65  
66              public void onMouseUp(Widget sender, int x, int y) {
67                  inDrag = false;
68              }
69  
70              public void onMouseDown(Widget sender, int x, int y) {
71                  inDrag = true;
72                  start = x;
73              }
74  
75              public void onMouseMove(Widget sender, int x, int y) {
76                  if(inDrag) {
77                      float newPercent = (float) (
78                              (x - start) + lower.getOffsetWidth()
79                          ) / (float) base.getOffsetWidth();
80                      Logger.getAnonymousLogger().log( Level.SPAM, x + " " + newPercent, null);
81  
82                      int newPosition = Math.round((
83                                  target.getOffsetWidth()
84                                  + target.getMaxHorizontalScrollPosition()
85                              ) * newPercent) + x;
86                      target.setHorizontalScrollPosition(newPosition);
87                  }
88              }
89          };
90  
91      private MouseListener higherListener = new MouseListenerAdapter() {
92              public void onMouseDown(Widget sender, int x, int y) {
93                  float newPercent = (float) (
94                          x + lower.getOffsetWidth() + bar.getOffsetWidth()
95                      ) / (float) base.getOffsetWidth();
96                  int newPosition = Math.round((
97                              target.getOffsetWidth()
98                              + target.getMaxHorizontalScrollPosition()
99                          ) * newPercent) - (target.getOffsetWidth() / 2);
100 
101                 target.setHorizontalScrollPosition(newPosition);
102             }
103         };
104 
105     private MouseListener lowerListener = new MouseListenerAdapter() {
106             public void onMouseDown(Widget sender, int x, int y) {
107                 float newPercent = (float) x / (float) base.getOffsetWidth();
108                 int newPosition = Math.round((
109                             target.getOffsetWidth()
110                             + target.getMaxHorizontalScrollPosition()
111                         ) * newPercent) - (target.getOffsetWidth() / 2);
112                 Logger.getAnonymousLogger().log( Level.SPAM, "New Position: " + newPosition, null);
113                 target.setHorizontalScrollPosition(newPosition);
114             }
115         };
116 
117     private WindowResizeListener windowListener = new WindowResizeListener() {
118             public void onWindowResized(int width, int height) {
119                 refresh();
120             }
121         };

123     /** Creates a new instance of SoftVerticalScroll */
124     public SoftHorizontalScrollbar(SoftScrollArea target) {
125         this.target = target;
126         this.base = new SimplePanel();
127 
128         Panel horizontal = new HorizontalPanel();
129         this.base.setWidget(horizontal);
130         super.initWidget(this.base);
131         this.higherTarget.setWidget(this.higher);
132         this.barTarget.setWidget(this.bar);
133         this.lowerTarget.setWidget(this.lower);
134 
135         horizontal.add(lowerTarget);
136         horizontal.add(barTarget);
137         horizontal.add(higherTarget);
138         this.setStyleName("gwittir-SoftHorizontalScrollbar");
139         this.bar.setStyleName("bar");
140         this.lower.setStyleName("lower");
141         this.higher.setStyleName("higher");
142         DOM.setStyleAttribute(this.lower.getElement(), "overflow", "hidden");
143         DOM.setStyleAttribute(this.bar.getElement(), "overflow", "hidden");
144         DOM.setStyleAttribute(this.higher.getElement(), "overflow", "hidden");
145         DOM.setStyleAttribute(this.getElement(), "overflow", "hidden");
146     }

150     }
151

153         return this.higherListener;
154     }

160     protected void onAttach() {
161         super.onAttach();
162         target.addScrollListener(this.scrollListener);
163         Window.addWindowResizeListener(this.windowListener);
164         this.lowerTarget.addMouseListener(this.getLowerListener());
165         this.higherTarget.addMouseListener(this.getHigherListener());
166         this.barTarget.addMouseListener(this.getBarListener());
167 
168         Timer t = new Timer() {
169                 public void run() {
170                     target.setScrollPosition(target.getScrollPosition());
171                     refresh();
172                 }
173             };
174 
175         t.schedule(10);
176     }
177 
178     protected void onDetach() {
179         super.onDetach();
180         Window.removeWindowResizeListener(this.windowListener);
181         this.lowerTarget.removeMouseListener(this.getLowerListener());
182         this.higherTarget.removeMouseListener(this.getHigherListener());
183         this.barTarget.removeMouseListener(this.getBarListener());
184         target.removeScrollListener(this.scrollListener);
185     }

187     public void refresh() {
188         int currentWidth = this.getOffsetWidth();
189 
190         float pageSize = (float) target.getOffsetWidth() / (float) (
191                 target.getMaxHorizontalScrollPosition()
192                 + target.getOffsetWidth()
193             );
194 
195         //GWT.log("Page sisze " + pageSize, null);
196         float scrollPercentage = (float) target.getHorizontalScrollPosition() / (
197                 target.getMaxHorizontalScrollPosition() + 1
198             );
199 
200         //GWT.log("Scroll Percentage " + scrollPercentage, null);
201         int lowerWidth = Math.round((
202                     currentWidth - Math.round((float) currentWidth * pageSize)
203                 ) * scrollPercentage);
204 
205         //GWT.log("Lower width " + lowerWidth, null);
206         if(lowerWidth == 0) {
207             lowerWidth = 1;
208         }
209 
210         this.lower.setWidth(lowerWidth + "px");
211 
212         int barWidth = Math.round(currentWidth * pageSize);
213         //GWT.log("Bar width " + barWidth, null);
214         if(barWidth >=0 ){
215             this.bar.setWidth(barWidth + "px");
216         }
217         int higherWidth = currentWidth - lowerWidth - barWidth;
218         if( higherWidth >= 0 ){
219             this.higher.setWidth(higherWidth + "px");
220         }
221         this.lower.setHeight(this.getOffsetHeight() + "px");
222         this.bar.setHeight(this.getOffsetHeight() + "px");
223         this.higher.setHeight(this.getOffsetHeight() + "px");
224     }
