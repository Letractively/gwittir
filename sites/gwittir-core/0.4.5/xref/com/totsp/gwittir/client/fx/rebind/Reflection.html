36  public class Reflection extends Widget {
37      private Canvas canvas = new Canvas();
38      private Element element = canvas.getElement();
39      private Image base;
40      private double height;
41      private double opacity;
42      private int baseWidth;
43      private int baseHeight;
44      private SimplePanel panel = new SimplePanel();
45  
46      /** Creates a new instance of Reflection */
47      public Reflection() {
48          super();
49          init();
50      }
51  
52      protected void init(){
53          Logger.getAnonymousLogger().log( Level.DEBUG, "Init default Reflection", null );
54          //panel.setWidget( canvas );
55          setElement(canvas.getElement());
56          
57      }
58  
59      public void paint(Image base, int baseWidth, int baseHeight, double height, double opacity) {
60          this.height = height;
61          this.opacity = opacity;
62          if( this.base != null ){
63              paint();
64          }
65          this.base = base;
66          this.baseWidth = baseWidth;
67          this.baseHeight = baseHeight;
68  
69          int reflectHeight = (int) Math.round(baseHeight * height);
70          DOM.setStyleAttribute(element, "width", baseWidth + "px");
71          DOM.setStyleAttribute(element, "height", reflectHeight + "px");
72          /*DOM.setStyleAttribute(element, "marginBottom",
73              "-" + reflectHeight + "px");*/
74          DOM.setElementAttribute(element, "width", "" + baseWidth);
75          DOM.setElementAttribute(element, "height", "" + reflectHeight);
76          this.panel.setPixelSize(baseWidth, reflectHeight);
77          paint();
78      }
79  
80      private void paint() {
81      	GWT.log("Paint!"+this.baseWidth+"x"+this.baseHeight, null);
82          int reflectHeight = (int) Math.round(baseHeight * height);
83          Canvas.Context ctx = canvas.getContext();
84          ctx.save();
85          ctx.clearRect(0, 0, this.baseWidth, reflectHeight );
86          ctx.save();
87          ctx.translate(0, this.baseHeight -1 );
88          ctx.scale(1, -1);
89          ctx.drawImage(this.base, 0, 0, this.baseWidth, this.baseHeight );
90          ctx.restore();
91          ctx.setGlobalCompositeOperation("destination-out");
92          Canvas.LinearGradient gradient = ctx.createLinearGradient(0,0, 0, reflectHeight );
93          gradient.addColorStop(1, "rgba(255, 255, 255, 1.0)");
94  	gradient.addColorStop(0, "rgba(255, 255, 255, "+(1-this.opacity)+")");
95          ctx.setFillStyle( gradient );
96          ctx.fillRect(0,0, this.baseWidth, reflectHeight );
97          ctx.restore();
98      }
