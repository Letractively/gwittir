55      /**
56       *  Calls a new flow context name, from the sender context with the given object model.
57       * @param sender The lowest level widget to begin looking for the name
58       * @param name The activity name to look for
59       * @param model The model object to pass to the registered BoundWidget
60       * @return boolean indicating whether the call resulted in a state change.
61       */
62      public static boolean call(Widget sender, String name, Object model, boolean fireEvents) {
63          Widget contextRoot = sender;
64          FlowContext context = (FlowContext) contexts.get(contextRoot);
65  
66          while((context == null) && (contextRoot != null)) {
67              contextRoot = contextRoot.getParent();
68              context = (FlowContext) contexts.get(contextRoot);
69          }
70  
71          if(context == null || !(contextRoot instanceof HasWidgets || contextRoot instanceof HasWidget )) {
72              throw new RuntimeException(
73                  "Unknown destination name for this elements context.");
74          }
75  
76          Object panel = contextRoot;
77          BoundWidget<?> widget = context.get(name);
78  
79          if(widget == null) {
80              return call(contextRoot.getParent(), name, model);
81          }
82          if( model != null ){
83              widget.setModel(model);
84          }
85          BoundWidget<?> old = null;
86          
87  
88          if( panel instanceof SimplePanel ){
89              SimplePanel sp = (SimplePanel) panel;
90              if( sp.getWidget() instanceof BoundWidget ){
91                  old = (BoundWidget<?>) sp.getWidget();
92              }
93              sp.setWidget( (Widget) widget );
94          } else if( panel instanceof HasWidget ){
95              HasWidget hw = (HasWidget) panel;
96              if( hw.getWidget() instanceof BoundWidget ){
97                  old = (BoundWidget<?>) hw.getWidget();
98              }
99              hw.setWidget( (Widget) widget );
100         } else if( panel instanceof HasWidgets ){
101             HasWidgets hw = (HasWidgets) panel;
102             Iterator<Widget> it = hw.iterator();
103             while(it.hasNext()) {
104                 Object next = it.next();
105 
106                 if(next instanceof BoundWidget) {
107                     old = (BoundWidget<?>) next;
108 
109                     break;
110                 }
111             }
112             hw.clear();
113             hw.add((Widget) widget );
114         }
115         FlowEvent event = context.createEvent(contextRoot, name, widget, old );
116         if(fireEvents){
117 	        context.fireEvent(event);
118         }
119 	        
120 	    manager.transition(event, fireEvents);
121 	      
122         
123         return true;
124     }

126     /**
127      * Finds the first managed widget/panel from a source widget that has a name
128      * mapped in its flow context
129      * @param widget The low-level widget to search from
130      * @param name The name to search a flow context for
131      * @return the Managed widget that matches first.
132      */
133     public static Widget findContextWidget(Widget widget, String name) {
134         Widget contextRoot = widget;
135         FlowContext context = (FlowContext) contexts.get(contextRoot);
136 
137         while((context == null || context.get(name) == null) && (contextRoot != null) ) {
138             contextRoot = contextRoot.getParent();
139             context = (FlowContext) contexts.get(contextRoot);
140         }
141 
142         return contextRoot;
143     }
