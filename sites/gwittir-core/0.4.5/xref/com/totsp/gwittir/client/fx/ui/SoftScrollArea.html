80      public void animateToHorizontalScrollPosition(int newPosition,
81          MutationStrategy integerStrategy, int duration) {
82          int max = widget.getOffsetWidth() - base.getOffsetWidth();
83  
84          if(limit & (newPosition > max)) {
85              newPosition = max;
86          }
87  
88          if(limit & (newPosition < 0)) {
89              newPosition = 0;
90          }
91  
92          if(this.horizontalAnimator != null) {
93              this.horizontalAnimator.cancel();
94          }
95  
96          horizontalAnimator = new PropertyAnimator(this,
97                  "horizontalScrollPosition", new Integer(newPosition),
98                  integerStrategy, duration,
99                  new AnimationFinishedCallback() {
100                     public void onFinish(PropertyAnimator animator) {
101                         horizOverride = Integer.MIN_VALUE;
102                         horizontalAnimator = null;
103                     }
104 
105                     public void onFailure(PropertyAnimator animator, Exception e) {
106                         horizOverride = Integer.MIN_VALUE;
107                         horizontalAnimator = null;
108                     }
109                 });
110         horizOverride = newPosition;
111         this.horizontalAnimator.start();
112     }

358     public final MouseListener MOUSE_MOVE_SCROLL_LISTENER  = new MouseListenerAdapter(){
359            
360             public void onMouseMove(Widget sender, int x, int y) {
361                 Logger.getAnonymousLogger().log( Level.SPAM,  x+" "+y, null);
362                 int baseX = getOffsetWidth();
363                 int baseY = getOffsetHeight();
364                 double percentX = (double) x / (double) baseX;
365                 double percentY = (double) y / (double) baseY;
366                 Integer currentX = (Integer) MutationStrategy.INTEGER_SINOIDAL
367                         .mutateValue( new Integer(0), 
368                         new Integer(getMaxHorizontalScrollPosition()),
369                         percentX);
370                 Integer currentY = (Integer) MutationStrategy.INTEGER_SINOIDAL
371                         .mutateValue( new Integer(0), 
372                         new Integer(getMaxScrollPosition()),
373                         percentY);
374                 setHorizontalScrollPosition( currentX.intValue() );
375                 setScrollPosition( currentY.intValue() );
376             }
377         };
