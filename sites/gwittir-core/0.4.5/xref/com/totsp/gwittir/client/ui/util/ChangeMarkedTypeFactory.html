21  package com.totsp.gwittir.client.ui.util;
22  
23  import com.google.gwt.user.client.ui.Widget;
24  import com.totsp.gwittir.client.ui.BoundWidget;
25  import java.util.ArrayList;
26  import java.util.HashMap;
27  import java.util.Iterator;
28  import java.util.List;
29  
30  /**
31   *
32   * @author cooper
33   */
34  public class ChangeMarkedTypeFactory extends BoundWidgetTypeFactory {
35      
36      HashMap widgetLookup = new HashMap(); /*<Bindable, ArrayList<ChangeMarkListener>>*/
37      private boolean marking = false;
38      
39      public ChangeMarkedTypeFactory(){
40          super(true);
41      }
42  
43      public BoundWidgetProvider getWidgetProvider(Class type) {
44          BoundWidgetProvider retValue;
45          
46          retValue = new BoundWidgetProviderWrapper(super.getWidgetProvider(type), this);
47          return retValue;
48      }
49  
50      public BoundWidgetProvider getWidgetProvider(String propertyName, Class type) {
51          BoundWidgetProvider retValue;
52          
53          retValue = new BoundWidgetProviderWrapper(super.getWidgetProvider(propertyName, type), this);
54          return retValue;
55      }
56      
57      public void reset(Object o){
58          List widgets = (List) this.widgetLookup.get( o );
59          if( widgets == null || widgets.size() == 0){
60              return;
61          }
62          for( Iterator it = new ArrayList(widgets).iterator(); it.hasNext(); ){
63              Widget widget = (Widget) it.next();
64              ChangeMarkerListener.hideMarker( widget );
65              widgets.remove( widget );
66          }
67      }
68      
69      public boolean hasBeenEdited(Object o){
70          List widgets = (List) this.widgetLookup.get( o );
71          return !( widgets == null || widgets.size() == 0);
72      }
73      
74      private static class BoundWidgetProviderWrapper implements BoundWidgetProvider{
75          BoundWidgetProvider p;
76          ChangeMarkedTypeFactory factory;
77          BoundWidgetProviderWrapper(BoundWidgetProvider p, ChangeMarkedTypeFactory factory){
78              this.p = p;
79              this.factory = factory;
80          }
81          
82          public BoundWidget get() {
83              BoundWidget w = p.get();
84              w.addPropertyChangeListener( new ChangeMarkerListener(this.factory) );
85              return w;
86          }
87          
88      }
89  
90      public boolean isMarking() {
91          return marking;
92      }
93  
94      public void setMarking(boolean marking) {
95          this.marking = marking;
96      }
97      
98  }
