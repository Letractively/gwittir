31  /** A FlowContext is a registry of BoundWidgets or BoundWidgetProviders and Actions to activity names.
32   * Each .add() method returns this, so you can use a builder pattern:
33   * FlowContext context = new FlowContext()
34   * 								.add()
35   * 								.add()
36   * 								.add();
37   *
38   * @author <a href="mailto:cooper@screaming-penguin.com">Robert "kebernet" Cooper</a>
39   */
40  public class FlowContext {
41      private String fromName = null;
42      private final HashMap<String,Action<?>> actions = new HashMap<String,Action<?>>();
43      private final HashMap<String,BoundWidgetProvider<? extends BoundWidget<?>>> destinations = new HashMap<String,BoundWidgetProvider<? extends BoundWidget<?>>>();
44      private ArrayList<FlowEventListener> listeners = new ArrayList<FlowEventListener>();
45      /** Creates a new instance of FlowContext */
46      public FlowContext() {
47          super();
48      }
49  
50      /** Adds a BoundWidget to a given activity name
51       * 
52       * @param name The activity name to register
53       * @param widget The BoundWidget to use
54       * @return this (builder pattern)
55       */
56      public FlowContext add(String name, BoundWidget<?> widget) {
57          destinations.put( name, new SimpleProvider<BoundWidget<?>>(widget));
58  
59          return this;
60      }
61  
62      /** Adds a BoundWidgetProvider for a given activity name.
63       * This method should be used where the BoundWidget needs to be recreated
64       * with each activity call (not a singleton). Optionally you can use a 
65       * Singleton provider to delay initialization of the BoundWidget until it
66       * is needed.
67       * 
68       * @param name The activity name to register
69       * @param provider The BoundWidgetProvider to use
70       * @return this (builder pattern)
71       */
72      public FlowContext add(String name, BoundWidgetProvider<?> provider) {
73          destinations.put(name, provider);
74  
75          return this;
76      }
77  
78      /** Adds a BoundWidget with an Action mapping
79       * 
80       * @param name The activity name to register
81       * @param widget The BoundWidget to use
82       * @param action The Action to register with the BoundWidget
83       * @return this (builder pattern)
84       */
85      public FlowContext add(String name, BoundWidget<?> widget, Action action) {
86          add(name, widget);
87          actions.put(name, action);
88  
89          return this;
90      }
91  
92      /** Adds a BoundWidgetProvider with an Action
93       * 
94       * @param name The activity name to register
95       * @param provider The BoundWidgetProvider to use
96       * @param action The Action to register with the BoundWidget
97       * @return this (builder pattern)
98       */
99      public FlowContext add(String name, BoundWidgetProvider<?> provider,
100         Action action) {
101         add(name, provider);
102         actions.put(name, action);
103 
104         return this;
105     }
106 
107     /** Returns a BoundWidget instance for a given activity name
108      * 
109      * @param name The activity name to lookup
110      * @return BoundWidget or null
111      */
112     public <T> BoundWidget<T> get(String name) {
113         BoundWidgetProvider<BoundWidget<T>> value =  (BoundWidgetProvider<BoundWidget<T>>) destinations.get(name);
114         BoundWidget<T> ret;
115 
116         ret = value.get();
117         
118 
119         if((ret != null) && (actions.get(name) != null)) {
120             ret.setAction(  (Action<BoundWidget<T>>) actions.get(name) );
121         }
122 
123         return ret;
124     }

126     FlowEvent createEvent(Widget managedWidget, String toName, BoundWidget<?> toWidget, BoundWidget<?> fromWidget ){
127     	FlowEvent e = new FlowEvent( this, managedWidget, fromWidget, fromWidget == null ? null : fromWidget.getModel(),
128                 this.fromName,
129                 toWidget, toWidget == null ? null : toWidget.getModel(), toName);
130     	this.fromName = e.getToName();
131     	return e;
132     }
133     
134     void fireEvent( FlowEvent e ){
135         
136         for(FlowEventListener listener : this.listeners ){
137                 listener.onFlowEvent( e );
138         }
139     }
140     
141     /** Adds a FlowEventListener
142      * 
143      * @param l Listener
144      */
145     public void addFlowEventListener( FlowEventListener l ){
146         this.listeners.add( l );
147     }
148     
149     /** Removes a FlowEventListener
150      * 
151      * @param l listener
152      */
153     public void removeFlowEventListener( FlowEventListener l ){
154         this.listeners.remove( l );
155     }
156     
157     /** Returns the currently registered FlowEventListeners
158      * 
159      * @return the currently registered FlowEventListeners
160      */
161     public FlowEventListener[] getFlowEventListeners(){
162         FlowEventListener[] listeners = new FlowEventListener[this.listeners.size() ];
163         this.listeners.toArray((Object[]) listeners);
164         return listeners;
165     }

167     private static class SimpleProvider<T extends BoundWidget<?>> implements BoundWidgetProvider<T> {
168 
169     	private T widget;
170     	public SimpleProvider(T widget){
171     		this.widget = widget;
172     	}
173     	
174 		public T get() {
175 			return this.widget;
176 		}
177     	
178     }
