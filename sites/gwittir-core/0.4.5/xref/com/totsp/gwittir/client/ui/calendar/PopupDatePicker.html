32  /**
33   * Renders a Date value to a Label with a Image src of
34   * "[module root]/calendar-icon.gif", which when clicked, will pop up a DatePicker
35   * for selection.
36   * @author rcooper
37   */
38  public class PopupDatePicker extends AbstractBoundWidget<Date>
39          implements SourcesCalendarDrawEvents, SourcesCalendarEvents, DateRenderers, HasFocus, SourcesClickEvents {
40      DatePicker base = new DatePicker();
41      Label label = new Label();
42      Image icon = new Image(GWT.getModuleBaseURL()+"calendar-icon.gif");
43      FocusPanel fp = new FocusPanel();
44      HorizontalPanel hp = new HorizontalPanel();
45      PopupPanel pp = new PopupPanel(true);
46      private ConverterWrapper converter = new ConverterWrapper();
47      private boolean hasFirstSet = false;
48      /** Creates a new instance of PopupDatePicker */
49      public PopupDatePicker() {
50          
51          Binding b = new Binding( label, "value", null, base, "value", converter);
52          b.setLeft();
53          b.bind();
54          pp.setWidget(base);
55          this.hp.add( this.label );
56          this.hp.add( this.icon );
57          fp.setWidget( hp );
58          this.initWidget( fp );
59          this.setStyleName("gwittir-PopupDatePicker");
60          icon.addClickListener( new ClickListener(){
61              public void onClick(Widget sender) {
62                  fp.setFocus(true);
63                  if( pp.isAttached() ){
64                      pp.hide();
65                  } else {
66                      int width = Window.getClientWidth() + Window.getScrollLeft();
67                      pp.setPopupPosition( getAbsoluteLeft(),
68                              getAbsoluteTop() + getOffsetHeight() );
69                      base.addCalendarListener(new CalendarListener(){
70                          public boolean onDateClicked(Calendar calendar, Date date) {
71                              if(date.getMonth() != base.getRenderDate().getMonth() ||
72                                      date.getYear() != base.getRenderDate().getYear() ){
73                                  return true;
74                              }
75                              pp.hide();
76                              calendar.removeCalendarListener(this);
77                              return true;
78                          }
79                          
80                      });
81                      pp.show();
82                      if( pp.getPopupLeft() + base.getOffsetWidth()  > width  ){
83                          pp.setPopupPosition( pp.getPopupLeft() + 
84                                  (width - pp.getPopupLeft()- base.getOffsetWidth() ), pp.getPopupTop() );
85                      }
86                  }
87              }
88          });
89          this.base.addPropertyChangeListener("value", new PropertyChangeListener(){
90              public void propertyChange(PropertyChangeEvent evt) {
91              	if(!hasFirstSet){
92              		hasFirstSet=true;
93              	} else {
94              		changes.firePropertyChange("value", evt.getOldValue(), evt.getNewValue() );
95              	}
96              }
97              
98          });
99      }
