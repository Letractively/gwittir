32      //This method populates a lookup table that relates eight
33      // bits represented as a String to eight actual bits for
34      // all possible combinations of eight bits.  This is
35      // essentially a reverse lookup table relative to the
36      // encodingBitMap table that is used to encode the
37      // message.  The only difference between the two is a
38      // reversal of the key and the value in the HashMap

40      void buildDecodingBitMap() {
41          for (int cnt = 0; cnt <= 255; cnt++) {
42              StringBuffer workingBuf = new StringBuffer();
43  
44              if ((cnt & 128) > 0) {
45                  workingBuf.append("1");
46              } else {
47                  workingBuf.append("0");
48              }
49  
50              if ((cnt & 64) > 0) {
51                  workingBuf.append("1");
52              } else {
53                  workingBuf.append("0");
54              }

56              if ((cnt & 32) > 0) {
57                  workingBuf.append("1");
58              } else {
59                  workingBuf.append("0");
60              }
61  
62              if ((cnt & 16) > 0) {
63                  workingBuf.append("1");
64              } else {

66              }
67  
68              if ((cnt & 8) > 0) {
69                  workingBuf.append("1");

75                  workingBuf.append("1");

87                  workingBuf.append("1");

97      //This method creates a Huffman decoding table by
98      // swapping the keys and the values from the Huffman
99      // encoding table received as an incoming parameter by
100     // the decode method.
101     void buildHuffDecodingTable() {
102         Iterator<Character> enumerator = huffEncodeTable.keySet().iterator();
103 
104         while (enumerator.hasNext()) {
105             Character nextKey = enumerator.next();
106             String nextString = huffEncodeTable.get(nextKey);
107             huffDecodeTable.put(nextString, nextKey);
108         } //end while
109     } //end buildHuffDecodingTable()

114     //This method receives a Huffman-encoded message along
115     // with a data structure containing particulars as to how
116     // the original message was encoded and the length of the
117     // original message.  It decodes the original message and
118     // returns the decoded version as a String object.
119     public String decode(byte[] encodedData,
120         HashMap<Character, String> huffEncodeTable, int rawDataLen) {
121         //Save the incoming parameters.
122         ArrayList<Byte> encodedDataList = new ArrayList<Byte>(encodedData.length);
123         for(byte b: encodedData){
124             encodedDataList.add(b);
125         }
126         this.binaryEncodedData = encodedDataList;
127         this.huffEncodeTable = huffEncodeTable;
128         this.rawDataLen = rawDataLen;
129 
130         //Create a decoding bit map, which is essentially the
131         // reverse of the encoding bit map that was used to
132         // encode the original message.
133         buildDecodingBitMap();
134 
135         //Decode the encoded message from a binary
136         // representation to a String of 1 and 0 characters
137         // that represent the actual bits in the encoded
138         // message.  Also, for illustration purposes only,
139         // this method may optionally display the String.
140         decodeToBitsAsString();
141 
142         //Create a Huffman decoding table by swapping the keys
143         // and values from the Huffman encoding table received
144         // as an incoming parameter by the decode method.
145         buildHuffDecodingTable();
146 
147         //Decode the String containing only 1 and 0 characters
148         // that represent the bits in the encoded message. This
149         // produces a replica of the original message that was
150         // subjected to Huffman encoding.  Write the resulting
151         // decoded message into a String object referred to by
152         // decoded data.
153         decodeStringBitsToCharacters();
154 
155         //Return the decoded message.  Eliminate the extraneous
156         // characters from the end of the message on the basis
157         // of the known length of the original message.
158         return decodedData.substring(0, rawDataLen);
159     } //end decode method

169     //The method reads the String containing only 1 and 0
170     // characters that represent the bits in the encoded
171     // message (stringDecodedData).  The characters are read
172     // from this string one character at a time.  As each new
173     // character is read, it is appended to the StringBuffer
174     // object referred to by workBuf.
175     //As each new character is appended to the StringBuffer
176     // object, a test is performed to determine if the
177     // current contents of the StringBuffer object match one
178     // of the keys in a lookup table that relates strings
179     // representing Huffman bitcodes to characters in the
180     // original message.
181     //When a match is found, the value  associated with that
182     // key is extracted and appended to the StringBuffer
183     // object referred to by output.  Thus, the output text
184     // is built up one character at a time.
185     //Having processed the matching key, A new empty
186     // StringBuffer object is instantiated, referred to by
187     // workBuf, and the process of reading, appending, and
188     // testing for a match is repeated until all of the
189     // characters in the string that represents the bits in
190     // the encoded message have been processed.  At that
191     // point, the StringBuffer object referred to by output
192     // contains the entire decoded message.  It is converted
193     // to type String and written into the object referred to
194     // by decodedData.  Then the method returns with the task
195     // of decoding the encoded message having been completed.
196     void decodeStringBitsToCharacters() {
197         StringBuffer output = new StringBuffer();
198         StringBuffer workBuf = new StringBuffer();
199 
200         for (int cnt = 0; cnt < stringDecodedData.length(); cnt++) {
201             workBuf.append(stringDecodedData.charAt(cnt));
202 
203             if (huffDecodeTable.containsKey(workBuf.toString())) {
204                 output.append(huffDecodeTable.get(workBuf.toString()));
205                 workBuf = new StringBuffer();
206             } //end if
207         } //end for loop
208 
209         decodedData = output.toString();
210     }
