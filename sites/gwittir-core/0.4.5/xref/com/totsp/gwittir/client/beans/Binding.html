178             Binding child = children.get(i);

208     /**
209      * Establishes a two-way binding between the objects.
210      */
211     public void bind() {
212         if ((left != null) && (right != null)) {
213             left.object.addPropertyChangeListener(left.property.getName(),
214                 left.listener);
215 
216             if (left.nestedListener != null) {
217                 left.nestedListener.setup();
218             }
219 
220             right.object.addPropertyChangeListener(right.property.getName(),
221                 right.listener);
222 
223             if (right.nestedListener != null) {
224                 right.nestedListener.setup();
225             }
226         }
227 
228         for (int i = 0; (children != null) && (i < children.size()); i++) {
229             Binding child = children.get(i);
230             child.bind();
231         }
232 
233         this.bound = true;
234     }

244     public boolean validate() {
245         boolean valid = true;
246 
247         if ((left != null) && (right != null)) {
248             if (left.validator != null) {
249                 try {
250                     left.validator.validate(left.property.getAccessorMethod()
251                                                          .invoke(left.object,
252                             null));
253                 } catch (ValidationException ve) {
254                     valid = false;
255 
256                     if (left.feedback != null) {
257                         left.feedback.handleException(left.object, ve);
258                     }
259 
260                     if (left.listener instanceof DefaultPropertyChangeListener) {
261                         ((DefaultPropertyChangeListener) left.listener).lastException = ve;
262                     }
263                 } catch (Exception e) {
264                     valid = false;
265                     Binding.LOGGER.log( Level.WARN, null, e);
266                 }
267             }
268 
269             if (right.validator != null) {
270                 try {
271                     right.validator.validate(right.property.getAccessorMethod()
272                                                            .invoke(right.object,
273                             null));
274                 } catch (ValidationException ve) {
275                     valid = false;
276 
277                     if (right.feedback != null) {
278                         right.feedback.handleException(right.object, ve);
279                     }
280 
281                     if (right.listener instanceof DefaultPropertyChangeListener) {
282                         ((DefaultPropertyChangeListener) right.listener).lastException = ve;
283                     }
284                 } catch (Exception e) {
285                     valid = false;
286                     Binding.LOGGER.log(Level.WARN, null, e);
287                 }
288             }
289         }
290 
291         for (int i = 0; (children != null) && (i < children.size()); i++) {
292             Binding child = children.get(i);
293             valid = valid & child.isValid();
294         }
295 
296         return valid;
297     }

299     /**
300      * Performs a quick validation on the Binding to determine if it is valid.
301      * @return boolean indicating all values are valid.
302      */
303     public boolean isValid() {
304         try {
305             if ((left != null) && (right != null)) {
306                 if (left.validator != null) {
307                     left.validator.validate(left.property.getAccessorMethod()
308                                                          .invoke(left.object,
309                             null));
310                 }
311 
312                 if (right.validator != null) {
313                     right.validator.validate(right.property.getAccessorMethod()
314                                                            .invoke(right.object,
315                             null));
316                 }
317             }
318 
319             boolean valid = true;
320 
321             for (int i = 0; (children != null) && (i < children.size()); i++) {
322                 Binding child = children.get(i);
323                 valid = valid & child.isValid();
324             }
325 
326             return valid;
327         } catch (ValidationException ve) {
328             Binding.LOGGER.log(Level.INFO, "Invalid", ve);
329 
330             return false;
331         } catch (Exception e) {
332             throw new RuntimeException(e);
333         }
334     }

336     /**
337      * Breaks the two way binding and removes all listeners.
338      */
339     public void unbind() {
340         if ((left != null) && (right != null)) {
341             left.object.removePropertyChangeListener(left.property.getName(),
342                 left.listener);
343 
344             if (left.nestedListener != null) {
345                 left.nestedListener.cleanup();
346             }
347 
348             right.object.removePropertyChangeListener(right.property.getName(),
349                 right.listener);
350 
351             if (right.nestedListener != null) {
352                 right.nestedListener.cleanup();
353             }
354         }
355 
356         for (int i = 0; (children != null) && (i < children.size()); i++) {
357             Binding child = children.get(i);
358             child.unbind();
359         }
360 
361         this.bound = false;
362     }

381     public boolean equals(final Object obj) {
382         if (obj == null) {
383             return false;
384         }
385         
386         final Binding other = (Binding) obj;
387         if (this.left != other.left && (this.left == null || !this.left.equals(other.left))) {
388             return false;
389         }
390         if (this.right != other.right && (this.right == null || !this.right.equals(other.right))) {
391             return false;
392         }
393         return true;
394     }

413     private Bindable getDiscriminatedObject(Object collectionOrArray,
414         String descriminator) {
415         int equalsIndex = descriminator.indexOf("=");
416 
417         if (collectionOrArray instanceof Collection && (equalsIndex == -1)) {
418             return this.getBindableAtCollectionIndex((Collection) collectionOrArray,
419                 Integer.parseInt(descriminator));
420         } else if (collectionOrArray instanceof Collection) {
421             return this.getBindableFromCollectionWithProperty((Collection) collectionOrArray,
422                 descriminator.substring(0, equalsIndex),
423                 descriminator.substring(equalsIndex + 1));
424         } else if( collectionOrArray instanceof Map ){
425             return this.getBindableAtMapKey((Map) collectionOrArray, descriminator);
426         }else if (equalsIndex == -1) {
427             return ((Bindable[]) collectionOrArray)[Integer.parseInt(descriminator)];
428         } else {
429             return getBindableFromArrayWithProperty((Bindable[]) collectionOrArray,
430                 descriminator.substring(0, equalsIndex),
431                 descriminator.substring(equalsIndex + 1));
432         }
433     }

435     private Bindable getBindableFromArrayWithProperty(Bindable[] array,
436         String propertyName, String stringValue) {
437         Method access = null;
438 
439         for (int i = 0; i < array.length; i++) {
440             if (access == null) {
441                 access = Introspector.INSTANCE.getDescriptor(array[i])
442                                               .getProperty(propertyName)
443                                               .getAccessorMethod();
444             }
445 
446             try {
447                 if ((access.invoke(array[i], null) + "").equals(stringValue)) {
448                     return array[i];
449                 }
450             } catch (Exception e) {
451                 throw new RuntimeException(e);
452             }
453         }
454 
455         throw new RuntimeException("No Object matching " + propertyName + "=" +
456             stringValue);
457     }

459     private Bindable getBindableFromCollectionWithProperty(
460         Collection collection, String propertyName, String stringValue) {
461         Method access = null;
462 
463         for (Iterator it = collection.iterator(); it.hasNext();) {
464             Bindable object = (Bindable) it.next();
465 
466             if (access == null) {
467                 access = Introspector.INSTANCE.getDescriptor(object)
468                                               .getProperty(propertyName)
469                                               .getAccessorMethod();
470             }
471 
472             try {
473                 if ((access.invoke(object, null) + "").equals(stringValue)) {
474                     return object;
475                 }
476             } catch (Exception e) {
477                 throw new RuntimeException(e);
478             }
479         }
480 
481         throw new RuntimeException("No Object matching " + propertyName + "=" +
482             stringValue);
483     }

516     private BindingInstance createBindingInstance(Bindable object,
517         String propertyName) {
518         int dotIndex = propertyName.indexOf(".");
519         BindingInstance instance = new BindingInstance();
520         NestedPropertyChangeListener rtpcl = (dotIndex == -1) ? null
521                                                               : new NestedPropertyChangeListener(instance,
522                 object, propertyName);
523         ArrayList parents = new ArrayList();
524         ArrayList propertyNames = new ArrayList();
525 
526         while (dotIndex != -1) {
527             String pname = propertyName.substring(0, dotIndex);
528             propertyName = propertyName.substring(dotIndex + 1);
529             parents.add(object);
530 
531             try {
532                 String descriminator = null;
533                 int descIndex = pname.indexOf("[");
534 
535                 if (descIndex != -1) {
536                     descriminator = pname.substring(descIndex + 1,
537                             pname.indexOf("]", descIndex));
538                     pname = pname.substring(0, descIndex);
539                 }
540 
541                 propertyNames.add(pname);
542 
543                 if (descriminator != null) {
544                     //TODO Need a loop here to handle multi-dimensional/collections of collections
545                     Object collectionOrArray = INTROSPECTOR.getDescriptor(object)
546                                                            .getProperty(pname)
547                                                            .getAccessorMethod()
548                                                            .invoke(object, null);
549                     object = this.getDiscriminatedObject(collectionOrArray,
550                             descriminator);
551                 } else {
552                     object = (Bindable) INTROSPECTOR.getDescriptor(object)
553                                                     .getProperty(pname)
554                                                     .getAccessorMethod()
555                                                     .invoke(object, null);
556                 }
557             } catch (ClassCastException cce) {
558                 throw new RuntimeException("Nonbindable sub property: " +
559                     object + " . " + pname, cce);
560             } catch (Exception e) {
561                 throw new RuntimeException(e);
562             }
563 
564             dotIndex = propertyName.indexOf(".");
565         }
566 
567         if (rtpcl != null) {
568             rtpcl.parents = new Bindable[parents.size()];
569             parents.toArray(rtpcl.parents);
570             rtpcl.propertyNames = new String[propertyNames.size()];
571             propertyNames.toArray(rtpcl.propertyNames);
572         }
573 
574         instance.object = object;
575         try{
576         	instance.property = INTROSPECTOR.getDescriptor(object)
577                                         	.getProperty(propertyName);
578         } catch(NullPointerException e){
579         	throw new RuntimeException("Exception getting property "+propertyName, e );
580         }
581         instance.nestedListener = rtpcl;
582 
583         return instance;
584     }

622     private static class DefaultPropertyChangeListener
623         implements PropertyChangeListener {
624         private BindingInstance instance;
625         private BindingInstance target;
626         private ValidationException lastException = null;
627 
628         DefaultPropertyChangeListener(BindingInstance instance,
629             BindingInstance target) {
630             this.instance = instance;
631             this.target = target;
632         }
633 
634         public void propertyChange(PropertyChangeEvent propertyChangeEvent) {
635             Object value = propertyChangeEvent.getNewValue();
636 
637             if (instance.validator != null) {
638                 try {
639                     value = instance.validator.validate(value);
640                 } catch (ValidationException ve) {
641                     if (instance.feedback != null) {
642                         if (this.lastException != null) {
643                             instance.feedback.resolve(propertyChangeEvent.getSource());
644                         }
645 
646                         instance.feedback.handleException(propertyChangeEvent.getSource(),
647                             ve);
648                         this.lastException = ve;
649 
650                         return;
651                     } else {
652                         this.lastException = ve;
653                         throw new RuntimeException(ve);
654                     }
655                 }
656             }
657 
658             if (this.instance.feedback != null) {
659                 this.instance.feedback.resolve(propertyChangeEvent.getSource());
660             }
661 
662             this.lastException = null;
663 
664             // Adding this to simplify simple toString conversions.
665             // TODO add a nice way to register global converter defaults
666             if( instance.converter == null && target.property.getType() == String.class && instance.property.getType() != String.class ){
667                 instance.converter = Converter.TO_STRING_CONVERTER;
668             }
669             if (instance.converter != null) {
670                 value = instance.converter.convert(value);
671             }
672 
673             Object[] args = new Object[1];
674             args[0] = value;
675 
676             try {
677                 target.property.getMutatorMethod().invoke(target.object, args);
678             } catch (Exception e) {
679                 //throw new RuntimeException( "Exception setting property: "+target.property.getName(), e);
680                 LOGGER.log( Level.ERROR, "Exception setting property: "+target.property.getName(), e);
681             }
682         }
683 
684         public String toString() {
685             return "[Listener on : " + this.instance.object + " ] ";
686         }
687     }

689     private class NestedPropertyChangeListener implements PropertyChangeListener {
690         BindingInstance target;
691         Bindable sourceObject;
692         String propertyName;
693         Bindable[] parents;
694         String[] propertyNames;
695 
696         NestedPropertyChangeListener(BindingInstance target,
697             Bindable sourceObject, String propertyName) {
698             this.target = target;
699             this.sourceObject = sourceObject;
700             this.propertyName = propertyName;
701         }
702 
703         public void propertyChange(PropertyChangeEvent evt) {
704             if (bound) {
705                 unbind();
706                 bound = true;
707             }
708 
709             BindingInstance newInstance = createBindingInstance(sourceObject,
710                     propertyName);
711             target.object = newInstance.object;
712             target.nestedListener = newInstance.nestedListener;
713             target.nestedListener.target = target;
714             target.property = newInstance.property;
715 
716             if (lastSet == Boolean.TRUE) {
717                 setLeft();
718             } else if (lastSet == Boolean.FALSE) {
719                 setRight();
720             }
721 
722             if (bound) {
723                 bind();
724             }
725         }
726 
727         public void setup() {
728             for (int i = 0; i < parents.length; i++) {
729                 parents[i].addPropertyChangeListener(this.propertyNames[i], this);
730             }
731         }
732 
733         public void cleanup() {
734             for (int i = 0; i < parents.length; i++) {
735                 parents[i].removePropertyChangeListener(this.propertyNames[i],
736                     this);
737             }
738         }
