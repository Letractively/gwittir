42  public class ReflectedImageGroup<T> extends AbstractBoundCollectionWidget<T,String> {
43      
44      private Collection value;
45      final private double maxScalar = 1.25D;
46      private transient ReflectedImage[] images;
47      final private int baseWidth;
48      final private int baseHeight;
49      final private double reflectHeight;
50      final private double opacity;
51      final private Grid grid = new Grid(1,3);
52      final private FlexTable imagesPanel = new FlexTable();
53      private Object selected;
54      private int lastSelectedIndex = -1;
55      
56      /** Creates a new instance of ReflectedImageGroup */
57      public ReflectedImageGroup(final int baseWidth,
58              final int baseHeight, final double reflectHeight, final double opacity) {
59          super();
60          this.baseWidth = baseWidth;
61          this.baseHeight = baseHeight;
62          this.reflectHeight = reflectHeight;
63          this.opacity = opacity;
64          this.imagesPanel.setHeight("100%");
65          final int  padding = ((int)(baseHeight * maxScalar) - baseHeight )/4;
66          this.grid.setCellSpacing( padding );
67          this.grid.setHeight("100%");
68          this.grid.setStyleName("gwittir-ReflectedFisheyeImageGroup");
69          this.grid.getRowFormatter().setVerticalAlign(0, HasVerticalAlignment.ALIGN_MIDDLE);
70          this.imagesPanel.getRowFormatter().setVerticalAlign(0, HasVerticalAlignment.ALIGN_MIDDLE);
71          this.grid.setWidget(0,1, imagesPanel );
72          this.imagesPanel.setCellPadding(0);
73          this.imagesPanel.setCellSpacing(2);
74          Renderer<T, String> renderer = (Renderer<T, String>) ToStringRenderer.INSTANCE;
75          this.setRenderer( renderer );
76          super.initWidget( grid );
77      }
78      
79      
80      public Collection<T> getValue() {
81          return this.value;
82      }
83      
84      public void setValue(Collection<T> value) {
85          this.value = value;
86          this.changes.firePropertyChange( "value", null, value );
87          this.render( this.value == null ? new ArrayList() : this.value );
88      }
89      
90      protected void render(Collection value){
91          this.images = new ReflectedImage[ value.size() ];
92          this.lastSelectedIndex = -1;
93          this.imagesPanel.clear();
94          if( this.imagesPanel.getRowCount() > 0 ){
95              this.imagesPanel.removeRow(0);
96          }
97          if( value == null ){
98              return;
99          }
100         Iterator<T> it = value.iterator();
101         final int maxSize = (int)(baseWidth*maxScalar) +4;
102         this.imagesPanel.setWidth( maxSize * value.size() +"px");
103         for( int i=0; it.hasNext(); i++ ){
104             final int index = i;
105             final T selectObject = it.next();
106             this.images[i] = new ReflectedImage(
107                     this.getRenderer().render( selectObject ).toString(),
108                     this.baseWidth, this.baseHeight,
109                     this.reflectHeight, this.opacity );
110             
111             this.imagesPanel.getCellFormatter().setWidth(0, index, maxSize+"px");
112             this.imagesPanel.getCellFormatter().setHorizontalAlignment(0, index, HasHorizontalAlignment.ALIGN_CENTER);
113             this.imagesPanel.setWidget( 0, index, this.images[i] );
114             
115             
116             this.images[i].addMouseListener( new MouseListener(){
117                 final Timer t = new Timer(){
118                     public void run() {
119                         if( selectObject != selected ){
120                             images[index].setHeight( (int) (baseHeight * maxScalar) );
121                             images[index].setWidth( (int) (baseWidth * maxScalar) );
122                         }
123                     }
124                     
125                 };
126                 public void onMouseUp(Widget sender, int x, int y) {
127                 }
128                 
129                 public void onMouseMove(Widget sender, int x, int y) {
130                 }
131                 
132                 public void onMouseDown(Widget sender, int x, int y) {
133                     setSelected( selectObject );
134                 }
135                 
136                 public void onMouseLeave(Widget sender) {
137                     t.cancel();
138                     if( selectObject != selected ){
139                         images[index].setHeight( baseHeight );
140                         images[index].setWidth( baseWidth );
141                     }
142                 }
143                 
144                 public void onMouseEnter(Widget sender) {
145                     t.schedule(50);
146                 }
147                 
148             });
149         }
150     }

152     public Object getSelected() {
153         return selected;
154     }
155     
156     public void setSelected(Object selected) {
157         LOG.log( Level.SPAM, selected + " :: " + selected.getClass(), null);
158         if( selected == this.selected ){
159             return;
160         }
161         Object old = this.selected;
162         
163         Object[] arr = value.toArray();
164         if( lastSelectedIndex != -1 ){
165             this.imagesPanel.getCellFormatter()
166             .removeStyleName(0, lastSelectedIndex, "selected");
167             this.images[lastSelectedIndex].setHeight( baseHeight );
168             this.images[lastSelectedIndex].setWidth( baseWidth );   
169         }
170         this.selected = selected;
171         
172         boolean found = false;
173         int index;
174         for( index = 0 ; selected != null &&  index < arr.length; index++ ){
175             if( arr[index] == selected ){
176                 found = true;
177                 break;
178             }
179         }
180         if( !found && selected !=null ){
181             throw new RuntimeException("Not an option");
182         }
183         images[index].setHeight( (int) (baseHeight * maxScalar) );
184         images[index].setWidth( (int) (baseWidth * maxScalar) );
185         this.imagesPanel.getCellFormatter().setStyleName(0, index, "selected");
186         this.changes.firePropertyChange( "selected", old, selected );
187         this.lastSelectedIndex = index;
188     }
