119 	@Override
120 	public void onBrowserEvent(Event event) {
121 		switch (DOM.eventGetType(event)) {
122 		case Event.ONCONTEXTMENU:
123 			GWT.log("ONCONTEXTMENU", null);
124 			DOM.eventPreventDefault(event);
125 			this.popup.show();
126 			this.popup.setPopupPosition(x, y);
127 			break;
128 
129 		case Event.ONMOUSEDOWN:
130 			GWT.log("ONMOUSEDOWN", null);
131 			if ((DOM.eventGetButton(event) == Event.BUTTON_RIGHT)
132 					|| DOM.eventGetCtrlKey(event)) {
133 				rightDown = true;
134 				DOM.eventPreventDefault(event);
135 				x = DOM.eventGetClientX(event);
136 				y = DOM.eventGetClientY(event);
137 				if ((DOM.eventGetButton(event) == Event.BUTTON_RIGHT)) {
138 					t.schedule(5);
139 				}
140 			} else {
141 				rightDown = false;
142 			}
143 
144 			break;
145 
146 		case Event.ONMOUSEUP:
147 			if (rightDown) {
148 				this.popup.show();
149 				this.popup.setPopupPosition(x, y);
150 				DOM.eventCancelBubble(event, true);
151 			} else {
152 				rightDown = false;
153 			}
154 
155 			break;
156 		case Event.ONCLICK:
157 			if (DOM.eventGetButton(event) == Event.BUTTON_RIGHT) {
158 				GWT.log("RIGHT CLICK", null);
159 				DOM.eventPreventDefault(event);
160 			}
161 		default:
162 			break;
163 		}
164 	}
