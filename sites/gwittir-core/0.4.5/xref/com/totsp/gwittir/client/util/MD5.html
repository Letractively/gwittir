103   /** Create this State as a copy of another state */
104   public MD5State (MD5State from) {
105     this();
106 
107     int i;
108 
109     for (i = 0; i < buffer.length; i++)
110       this.buffer[i] = from.buffer[i];
111 
112     for (i = 0; i < state.length; i++)
113       this.state[i] = from.state[i];

116       this.count[i] = from.count[i];

175   private int rotate_left (int x, int n) {
176     return (x << n) | (x >>> (32 - n));

182   private int uadd (int a, int b) {
183     long aa, bb;
184     aa = ((long) a) & 0xffffffffL;
185     bb = ((long) b) & 0xffffffffL;
186 
187     aa += bb;
188 
189     return (int) (aa & 0xffffffffL);
190   }

200   private int FF (int a, int b, int c, int d, int x, int s, int ac) {
201     a = uadd(a, ((b & c) | (~b & d)), x, ac);
202     return uadd(rotate_left(a, s), b);
203   }

205   private int GG (int a, int b, int c, int d, int x, int s, int ac) {
206     a = uadd(a, ((b & d) | (c & ~d)), x, ac);
207     return uadd(rotate_left(a, s), b);
208   }

210   private int HH (int a, int b, int c, int d, int x, int s, int ac) {
211     a = uadd(a, (b ^ c ^ d), x, ac);
212     return uadd(rotate_left(a, s) , b);
213   }

215   private int II (int a, int b, int c, int d, int x, int s, int ac) {
216     a = uadd(a, (c ^ (b | ~d)), x, ac);
217     return uadd(rotate_left(a, s), b);
218   }

330   /**
331    * Updates hash with the bytebuffer given (using at maximum length bytes from
332    * that buffer)
333    *
334    * @param state	Which state is updated
335    * @param buffer	Array of bytes to be hashed
336    * @param offset	Offset to buffer array
337    * @param length	Use at maximum `length' bytes (absolute
338    *			maximum is buffer.length)
339    */
340   public void Update (MD5State stat, byte buffer[], int offset, int length) {
341     int	index, partlen, i, start;
342 
343 /*    System.out.print("Offset = " + offset + "\tLength = " + length + "\t");
344     System.out.print("Buffer = ");
345     for (i = 0; i < buffer.length; i++)
346 	System.out.print((int) (buffer[i] & 0xff) + " ");
347     System.out.print("\n");*/
348 
349     finals = null;
350 
351     /* Length can be told to be shorter, but not inter */
352     if ((length - offset)> buffer.length)
353       length = buffer.length - offset;
354 
355     /* compute number of bytes mod 64 */
356     index = (int) (stat.count[0] >>> 3) & 0x3f;
357 
358     if ((stat.count[0] += (length << 3)) <
359 	(length << 3))
360       stat.count[1]++;
361 
362     stat.count[1] += length >>> 29;
363 
364     partlen = 64 - index;
365 
366     if (length >= partlen) {
367       for (i = 0; i < partlen; i++)
368 	stat.buffer[i + index] = buffer[i + offset];
369 
370       Transform(stat, stat.buffer, 0);
371 
372       for (i = partlen; (i + 63) < length; i+= 64)
373 	Transform(stat, buffer, i);
374 
375       index = 0;
376     } else
377       i = 0;
378 
379     /* buffer remaining input */
380     if (i < length) {
381       start = i;
382       for (; i < length; i++)
383 	stat.buffer[index + i - start] = buffer[i + offset];
384     }
385   }

455   private byte[] Encode (int input[], int len) {
456     int		i, j;
457     byte	out[];
458 
459     out = new byte[len];
460 
461     for (i = j = 0; j  < len; i++, j += 4) {
462       out[j] = (byte) (input[i] & 0xff);
463       out[j + 1] = (byte) ((input[i] >>> 8) & 0xff);
464       out[j + 2] = (byte) ((input[i] >>> 16) & 0xff);
465       out[j + 3] = (byte) ((input[i] >>> 24) & 0xff);
466     }
467 
468     return out;
469   }

471   /**
472    * Returns array of bytes (16 bytes) representing hash as of the
473    * current state of this object. Note: getting a hash does not
474    * invalidate the hash object, it only creates a copy of the real
475    * state which is finalized.
476    *
477    * @return	Array of 16 bytes, the hash of all updated bytes
478    */
479   public synchronized byte[] Final () {
480     byte	bits[];
481     int		index, padlen;
482     MD5State	fin;
483 
484     if (finals == null) {
485       fin = new MD5State(state);
486 
487       bits = Encode(fin.count, 8);
488 
489       index = (int) ((fin.count[0] >>> 3) & 0x3f);
490       padlen = (index < 56) ? (56 - index) : (120 - index);
491 
492       Update(fin, padding, 0, padlen);
493       /**/
494       Update(fin, bits, 0, 8);
495 
496       /* Update() sets finalds to null */
497       finals = fin;
498     }
499 
500     return Encode(finals.state, 16);
501   }
