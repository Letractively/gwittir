18  package java.util;
19  
20  import com.google.gwt.core.client.GWT;
21  
22  /**
23   * STUB: ResourceBundle is an abstract class which is the superclass of classes which
24   * provide locale specific resources. A bundle contains a number of named
25   * resources, where the names are Strings. A bundle may have a parent bundle,
26   * when a resource is not found in a bundle, the parent bundle is searched for
27   * the resource.
28   * 
29   * @see Properties
30   * @see PropertyResourceBundle
31   * @since 1.1
32   */
33  public abstract class ResourceBundle {
34  	
35  	/**
36  	 * The parent of this ResourceBundle.
37  	 */
38  	protected ResourceBundle parent;
39  
40  	private Locale locale;
41  
42  	static class MissingBundle extends ResourceBundle {
43          public Enumeration getKeys() {
44  			return null;
45  		}
46  
47          public Object handleGetObject(String name) {
48  			return null;
49  		}
50  	}
51  
52  	/**
53  	 * Constructs a new instance of this class.
54  	 * 
55  	 */
56  	public ResourceBundle() {
57  		/*empty*/
58  	}
59  
60  	/**
61  	 * Finds the named resource bundle for the default locale.
62  	 * 
63  	 * @param bundleName
64  	 *            the name of the resource bundle
65  	 * @return ResourceBundle
66  	 * 
67  	 * @exception MissingResourceException
68  	 *                when the resource bundle cannot be found
69  	 */
70  	public static final ResourceBundle getBundle(String bundleName)
71  			throws MissingResourceException {
72  		return getBundleImpl(bundleName, Locale.getDefault());
73  	}
74  
75  	/**
76  	 * Finds the named resource bundle for the specified locale.
77  	 * 
78  	 * @param bundleName
79  	 *            the name of the resource bundle
80  	 * @param locale
81  	 *            the locale
82  	 * @return ResourceBundle
83  	 * 
84  	 * @exception MissingResourceException
85  	 *                when the resource bundle cannot be found
86  	 */
87  	public static final ResourceBundle getBundle(String bundleName,
88  			Locale locale) {
89  		return getBundleImpl(bundleName, locale);
90  	}
91  
92  	private static ResourceBundle getBundleImpl(String bundleName, Locale locale) throws MissingResourceException {
93  		if (bundleName != null) {
94  			if (!locale.equals(Locale.getDefault())) {
95  				String localeName = locale.toString();
96  				if (localeName.length() > 0) {
97                      localeName = "_" + localeName; //$NON-NLS-1$
98                  }
99  			}
100 			String localeName = Locale.getDefault().toString();
101 			if (localeName.length() > 0) {
102                 localeName = "_" + localeName; //$NON-NLS-1$
103             }
104 			throw new MissingResourceException(null, bundleName + '_' + locale , ""); //$NON-NLS-1$
105 		}
106         throw new NullPointerException();
107 	}
108 
109 	/**
110 	 * Answers the names of the resources contained in this ResourceBundle.
111 	 * 
112 	 * @return an Enumeration of the resource names
113 	 */
114 	public abstract Enumeration getKeys();
115 
116 	/**
117 	 * Gets the Locale of this ResourceBundle.
118 	 * 
119 	 * @return the Locale of this ResourceBundle
120 	 */
121 	public Locale getLocale() {
122 		return locale;
123 	}
124 
125 	/**
126 	 * Answers the named resource from this ResourceBundle.
127 	 * 
128 	 * @param key
129 	 *            the name of the resource
130 	 * @return the resource object
131 	 * 
132 	 * @exception MissingResourceException
133 	 *                when the resource is not found
134 	 */
135 	public final Object getObject(String key) {
136 		ResourceBundle last, theParent = this;
137 		do {
138 			Object result = theParent.handleGetObject(key);
139 			if (result != null) {
140                 return result;
141             }
142 			last = theParent;
143 			theParent = theParent.parent;
144 		} while (theParent != null);
145 		throw new MissingResourceException(null, GWT.getTypeName(last), key);
146 	}
147 
148 	/**
149 	 * Answers the named resource from this ResourceBundle.
150 	 * 
151 	 * @param key
152 	 *            the name of the resource
153 	 * @return the resource string
154 	 * 
155 	 * @exception MissingResourceException
156 	 *                when the resource is not found
157 	 */
158 	public final String getString(String key) {
159 		return (String) getObject(key);
160 	}
161 
162 	/**
163 	 * Answers the named resource from this ResourceBundle.
164 	 * 
165 	 * @param key
166 	 *            the name of the resource
167 	 * @return the resource string array
168 	 * 
169 	 * @exception MissingResourceException
170 	 *                when the resource is not found
171 	 */
172 	public final String[] getStringArray(String key) {
173 		return (String[]) getObject(key);
174 	}
175 
176 	/**
177 	 * Answers the named resource from this ResourceBundle, or null if the
178 	 * resource is not found.
179 	 * 
180 	 * @param key
181 	 *            the name of the resource
182 	 * @return the resource object
183 	 */
184 	protected abstract Object handleGetObject(String key);
185 
186 	/**
187 	 * Sets the parent resource bundle of this ResourceBundle. The parent is
188 	 * searched for resources which are not found in this resource bundle.
189 	 * 
190 	 * @param bundle
191 	 *            the parent resource bundle
192 	 */
193 	protected void setParent(ResourceBundle bundle) {
194 		parent = bundle;
195 	}
