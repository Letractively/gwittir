33  package java.beans;
34  
35  /**
36   * A type of {@link PropertyChangeEvent} that indicates that an indexed property
37   * has changed.
38   * 
39   * @since 1.5
40   */
41  public class IndexedPropertyChangeEvent extends PropertyChangeEvent {
42      // TODO: remove the @skip when Java 1.5 is officially supported
43      /*
44       * This file is based on code from the Apache Harmony Project.
45       * http://svn.apache.org/repos/asf/harmony/enhanced/classlib/trunk/modules/beans/src/main/java/java/beans/IndexedPropertyChangeEvent.java
46       *
47       * This file has been siginificantly modified for the target enviroment of a
48       * browser but the Harmony layout and structure has been kept to try to help
49       * make future re-syncs easier.
50       */
51  
52      private final int index;
53  
54      /**
55       * Creates a new property changed event with an indication of the property
56       * index.
57       * 
58       * @param source
59       *            the changed bean.
60       * @param propertyName
61       *            the changed property, or <code>null</code> to indicate an
62       *            unspecified set of the properties have changed.
63       * @param oldValue
64       *            the previous value of the property, or <code>null</code> if
65       *            the <code>propertyName</code> is <code>null</code> or the
66       *            previous value is unknown.
67       * @param newValue
68       *            the new value of the property, or <code>null</code> if the
69       *            <code>propertyName</code> is <code>null</code> or the new
70       *            value is unknown..
71       * @param index
72       *            the index of the property.
73       */
74      public IndexedPropertyChangeEvent(Object source, String propertyName,
75              Object oldValue, Object newValue, int index) {
76          super(source, propertyName, oldValue, newValue);
77          this.index = index;
78      }
79  
80      /**
81       * Answer the index of the property that was changed in this event.
82       * 
83       * @return The property element index.
84       */
85      public int getIndex() {
86          return index;
87      }
88  }
