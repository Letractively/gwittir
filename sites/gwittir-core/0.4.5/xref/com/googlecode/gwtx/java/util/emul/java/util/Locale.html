185 	/**
186      * Constructs a new Locale using the specified language, country, and
187      * variant codes.
188      * 
189      * @param language
190      * @param country
191      * @param variant
192      * @throws NullPointerException if <code>language</code>,
193      *         <code>country</code> or <code>variant</code> is
194      *         <code>null</code>.
195      */
196     public Locale(String language, String country, String variant) {
197         if (language == null || country == null || variant == null) {
198             throw new NullPointerException();
199         }
200         languageCode = language.toLowerCase();
201         // Map new language codes to the obsolete language
202         // codes so the correct resource bundles will be used.
203         if (languageCode.equals("he")) {//$NON-NLS-1$
204             languageCode = "iw"; //$NON-NLS-1$
205         } else if (languageCode.equals("id")) {//$NON-NLS-1$
206             languageCode = "in"; //$NON-NLS-1$
207         } else if (languageCode.equals("yi")) {//$NON-NLS-1$
208             languageCode = "ji"; //$NON-NLS-1$
209         }
210 
211         countryCode = country.toUpperCase();
212 
213         variantCode = variant;
214     }

307 	 * Gets the full country name in the specified Locale for the country code
308 	 * of this Locale. If there is no matching country name, the country code is
309 	 * returned.
310 	 * 
311 	 * @param locale
312 	 *            the Locale
313 	 * @return a country name
314 	 */
315 	public String getDisplayCountry(Locale locale) {
316 		if (countryCode.length() == 0) {
317             return countryCode;
318         }
319 		try {
320 			// First try the specified locale
321 			ResourceBundle bundle = getBundle("Country", locale); //$NON-NLS-1$
322 			String result = (String) bundle.handleGetObject(countryCode);
323 			if (result != null) {
324                 return result;
325             }
326 			// Now use the default locale
327 			if (locale != Locale.getDefault()) {
328                 bundle = getBundle("Country", Locale.getDefault()); //$NON-NLS-1$
329             }
330 			return bundle.getString(countryCode);
331 		} catch (MissingResourceException e) {
332 			return countryCode;
333 		}
334 	}

388 	 * Gets the full language, country, and variant names in the specified
389 	 * Locale for the codes of this Locale.
390 	 * 
391 	 * @param locale
392 	 *            the Locale
393 	 * @return a Locale name
394 	 */
395 	public String getDisplayName(Locale locale) {
396 		int count = 0;
397 		StringBuffer buffer = new StringBuffer();
398 		if (languageCode.length() > 0) {
399 			buffer.append(getDisplayLanguage(locale));
400 			count++;
401 		}
402 		if (countryCode.length() > 0) {
403 			if (count == 1) {
404                 buffer.append(" ("); //$NON-NLS-1$
405             }
406 			buffer.append(getDisplayCountry(locale));
407 			count++;
408 		}
409 		if (variantCode.length() > 0) {
410 			if (count == 1) {
411                 buffer.append(" ("); //$NON-NLS-1$
412             } else if (count == 2) {
413                 buffer.append(","); //$NON-NLS-1$
414             }
415 			buffer.append(getDisplayVariant(locale));
416 			count++;
417 		}
418 		if (count > 1) {
419             buffer.append(")"); //$NON-NLS-1$
420         }
421 		return buffer.toString();
422 	}

436 	 * Gets the full variant name in the specified Locale for the variant code
437 	 * of this Locale. If there is no matching variant name, the variant code is
438 	 * returned.
439 	 * 
440 	 * @param locale
441 	 *            the Locale
442 	 * @return a variant name
443 	 */
444 	public String getDisplayVariant(Locale locale) {
445 		if (variantCode.length() == 0) {
446             return variantCode;
447         }
448 		ResourceBundle bundle;
449 		try {
450 			bundle = getBundle("Variant", locale); //$NON-NLS-1$
451 		} catch (MissingResourceException e) {
452 			return variantCode.replace('_', ',');
453 		}
454 
455 		StringBuffer result = new StringBuffer();
456 		StringTokenizer tokens = new StringTokenizer(variantCode, "_"); //$NON-NLS-1$
457 		while (tokens.hasMoreTokens()) {
458 			String code, variant = tokens.nextToken();
459 			try {
460 				code = bundle.getString(variant);
461 			} catch (MissingResourceException e) {
462 				code = variant;
463 			}
464 			result.append(code);
465 			if (tokens.hasMoreTokens()) {
466                 result.append(',');
467             }
468 		}
469 		return result.toString();
470 	}

577 	/**
578 	 * Answers the string representation of this Locale.
579 	 * 
580 	 * @return the string representation of this Locale
581 	 */
582     public final String toString() {
583 		StringBuffer result = new StringBuffer();
584 		result.append(languageCode);
585 		if (countryCode.length() > 0) {
586 			result.append('_');
587 			result.append(countryCode);
588 		}
589 		if (variantCode.length() > 0 && result.length() > 0 ) {
590 			if (0 == countryCode.length()) {
591                 result.append("__"); //$NON-NLS-1$
592             } else {
593                 result.append('_');
594             }
595 			result.append(variantCode);
596 		}
597 		return result.toString();
598 	}
