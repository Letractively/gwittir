66  	/**
67  	 * Parse the <code>string</code>str into <code>URL</code> using u's
68  	 * context. URL strings generally have the following format:
69  	 * <code><center>//www.company.com/java/file1.java#reference </center></code>
70  	 * The string is parsed in HTTP format. If the protocol has a different URL
71  	 * format this method must be overridden.
72  	 * 
73  	 * @param u
74  	 *            java.net.URL The URL to receive parsed values.
75  	 * @param str
76  	 *            java.lang.String The string URL spec from which u is derived
77  	 * @param start
78  	 *            int The index in the string from which to begin parsing
79  	 * @param end
80  	 *            int The index to stop parsing
81  	 * 
82  	 * @see #toExternalForm
83  	 * @see URL
84  	 */
85  	protected void parseURL(URL u, String str, int start, int end) {
86  		if (end < start) {
87              return;
88          }
89  		String parseString = "";
90  		if (start < end) {
91              parseString = str.substring(start, end);
92          }
93  		end -= start;
94  		int fileIdx = 0;
95  
96  		// Default is to use info from context
97  		String host = u.getHost();
98  		int port = u.getPort();
99  		String ref = u.getRef();
100 		String file = u.getPath();
101 		String query = u.getQuery();
102 		String authority = u.getAuthority();
103 		String userInfo = u.getUserInfo();
104 
105 		int refIdx = parseString.indexOf('#', 0);
106 		if (parseString.startsWith("//")) {
107 			int hostIdx = 2, portIdx = -1;
108 			port = -1;
109 			fileIdx = parseString.indexOf('/', hostIdx);
110             int questionMarkIndex = parseString.indexOf('?', hostIdx);
111             if ((questionMarkIndex != -1)
112                     && ((fileIdx == -1) || (fileIdx > questionMarkIndex))) {
113                 fileIdx = questionMarkIndex;
114             }
115 			if (fileIdx == -1) {
116 				fileIdx = end;
117 				// Use default
118 				file = "";
119 			}
120 			int hostEnd = fileIdx;
121 			if (refIdx != -1 && refIdx < fileIdx) {
122                 hostEnd = refIdx;
123             }
124 			int userIdx = parseString.lastIndexOf('@', hostEnd);
125 			authority = parseString.substring(hostIdx, hostEnd);
126 			if (userIdx > -1) {
127 				userInfo = parseString.substring(hostIdx, userIdx);
128 				hostIdx = userIdx + 1;
129 			}
130 
131 			portIdx = parseString.indexOf(':', userIdx == -1 ? hostIdx
132 					: userIdx);
133 			int endOfIPv6Addr = parseString.indexOf(']');
134 			// if there are square braces, ie. IPv6 address, use last ':'
135 			if (endOfIPv6Addr != -1) {
136 				try {
137 					if (parseString.length() > endOfIPv6Addr + 1) {
138 						char c = parseString.charAt(endOfIPv6Addr + 1);
139 						if (c == ':') {
140 							portIdx = endOfIPv6Addr + 1;
141 						} else {
142 							portIdx = -1;
143 						}
144 					} else {
145 						portIdx = -1;
146 					}
147 				} catch (Exception e) {
148 					// Ignored
149 				}
150 			}
151 
152 			if (portIdx == -1 || portIdx > fileIdx) {
153                 host = parseString.substring(hostIdx, hostEnd);
154             } else {
155 				host = parseString.substring(hostIdx, portIdx);
156 				String portString = parseString.substring(portIdx + 1, hostEnd);
157 				if (portString.length() == 0) {
158                     port = -1;
159                 } else {
160                     port = Integer.parseInt(portString);
161                 }
162 			}
163 		}
164 
165 		if (refIdx > -1) {
166             ref = parseString.substring(refIdx + 1, end);
167         }
168 		int fileEnd = (refIdx == -1 ? end : refIdx);
169 
170 		int queryIdx = parseString.lastIndexOf('?', fileEnd);
171         boolean canonicalize = false;
172         if (queryIdx > -1) {
173             query = parseString.substring(queryIdx + 1, fileEnd);
174             if (queryIdx == 0 && file != null) {
175                 if (file.equals("")) {
176                     file = "/";
177                 } else if (file.startsWith("/")) {
178                     canonicalize = true;
179                 }
180                 int last = file.lastIndexOf('/') + 1;
181                 file = file.substring(0, last);
182             }
183             fileEnd = queryIdx;
184         } else
185         // Don't inherit query unless only the ref is changed
186         if (refIdx != 0) {
187             query = null;
188         }
189 
190 		if (fileIdx > -1) {
191 			if (fileIdx < end && parseString.charAt(fileIdx) == '/') {
192                 file = parseString.substring(fileIdx, fileEnd);
193             } else if (fileEnd > fileIdx) {
194 				if (file == null) {
195                     file = "";
196                 } else if (file.equals("")) {
197                     file = "/";
198                 } else if (file.startsWith("/")) {
199                     canonicalize = true;
200                 }
201 				int last = file.lastIndexOf('/') + 1;
202 				if (last == 0) {
203                     file = parseString.substring(fileIdx, fileEnd);
204                 } else {
205                     file = file.substring(0, last)
206 							+ parseString.substring(fileIdx, fileEnd);
207                 }
208 			}
209 		}
210 		if (file == null) {
211             file = "";
212         }
213 
214 		if (host == null) {
215             host = "";
216         }
217 
218 		if (canonicalize) {
219 			// modify file if there's any relative referencing
220 			int dirIndex;
221 			while ((dirIndex = file.indexOf("/./")) >= 0) {
222                 file = file.substring(0, dirIndex + 1)
223 						+ file.substring(dirIndex + 3);
224             }
225 			if (file.endsWith("/.")) {
226                 file = file.substring(0, file.length() - 1);
227             }
228 			while ((dirIndex = file.indexOf("/../")) >= 0) {
229 				if (dirIndex != 0) {
230 					file = file.substring(0, file
231 							.lastIndexOf('/', dirIndex - 1))
232 							+ file.substring(dirIndex + 3);
233 				} else {
234                     file = file.substring(dirIndex + 3);
235                 }
236 			}
237 			if (file.endsWith("/..") && file.length() > 3) {
238                 file = file.substring(0, file.lastIndexOf('/',
239 						file.length() - 4) + 1);
240             }
241 		}
242 
243 		setURL(u, u.getProtocol(), host, port, authority, userInfo, file,
244 				query, ref);
245 	}

303 	/**
304 	 * Answers the string equivalent of an URL using HTTP parsinf format.
305 	 * 
306 	 * @return java.lang.String the string representation of this URL
307 	 * @param url
308 	 *            java.net.URL the url object to be processed
309 	 * 
310 	 * @see #parseURL
311 	 * @see URL#toExternalForm()
312 	 */
313 	protected String toExternalForm(URL url) {
314 		StringBuffer answer = new StringBuffer(url.getProtocol().length()
315 				+ url.getFile().length() + 16);
316 		answer.append(url.getProtocol());
317 		answer.append(':');
318 		String authority = url.getAuthority();
319 		if (authority != null && authority.length() > 0) {
320 			answer.append("//");
321 			answer.append(url.getAuthority());
322 		}
323 
324 		String file = url.getFile();
325 		String ref = url.getRef();
326 		// file is never null
327 		answer.append(file);
328 		if (ref != null) {
329 			answer.append('#');
330 			answer.append(ref);
331 		}
332 		return answer.toString();
333 	}

335 	/**
336 	 * Compares the two urls, and answers true if they represent the same URL.
337 	 * Two URLs are equal if they have the same file, host, port, protocol,
338 	 * query, and ref components.
339 	 * 
340 	 * @param url1
341 	 *            URL the first URL to compare
342 	 * @param url2
343 	 *            URL the second URL to compare
344 	 * @return <code>true</code> if the URLs are the same <code>false</code>
345 	 *         if the URLs are different
346 	 * 
347 	 * @see #hashCode
348 	 */
349 	protected boolean equals(URL url1, URL url2) {
350 		if (!sameFile(url1, url2)) {
351             return false;
352         }
353 		String s1 = url1.getRef(), s2 = url2.getRef();
354 		if (s1 != s2 && (s1 == null || !s1.equals(s2))) {
355             return false;
356         }
357 		s1 = url1.getQuery();
358 		s2 = url2.getQuery();
359 		return s1 == s2 || (s1 != null && s1.equals(s2));
360 	}

371 	 */
372 	protected InetAddress getHostAddress(URL url) {
373 		try {
374 			String host = url.getHost();
375 			if (host == null || host.length() == 0) {
376                 return null;
377             }
378 			return InetAddress.getByName(host);
379 		} catch (UnknownHostException e) {
380 			return null;
381 		}

393 	/**
394 	 * Compares the two urls, and answers true if they have the same host
395 	 * components.
396 	 * 
397 	 * @return <code>true</code> if the hosts of the URLs are the same
398 	 *         <code>false</code> if the hosts are different
399 	 */
400 	protected boolean hostsEqual(URL url1, URL url2) {
401         String host1 = url1.getHost(), host2 = url2.getHost();
402         if (host1 == host2 || (host1 != null && host1.equalsIgnoreCase(host2))) {
403             return true;
404         }
405         // Compare host address if the host name is not equal.
406         InetAddress address1 = getHostAddress(url1);
407         InetAddress address2 = getHostAddress(url2);
408         if (address1 != null && address1.equals(address2)) {
409             return true;
410         }
411         return false;
412     }

414 	/**
415 	 * Answers true if the urls refer to the same file. Compares the protocol,
416 	 * host, port and file components.
417 	 * 
418 	 * @return boolean true if the same resource, false otherwise
419 	 */
420 	protected boolean sameFile(URL url1, URL url2) {
421 		String s1 = url1.getProtocol();
422 		String s2 = url2.getProtocol();
423 		if (s1 != s2 && (s1 == null || !s1.equals(s2))) {
424             return false;
425         }
426 
427 		s1 = url1.getFile();
428 		s2 = url2.getFile();
429 		if (s1 != s2 && (s1 == null || !s1.equals(s2))) {
430             return false;
431         }
432 		if (!hostsEqual(url1, url2)) {
433             return false;
434         }
435 		int p1 = url1.getPort();
436 		if (p1 == -1) {
437             p1 = getDefaultPort();
438         }
439 		int p2 = url2.getPort();
440 		if (p2 == -1) {
441             p2 = getDefaultPort();
442         }
443 		return p1 == p2;
444 	}
