21  import java.io.ByteArrayOutputStream;
22  import java.io.UnsupportedEncodingException;

24  /**
25   * This class is used to encode a string using the format required by
26   * <code>application/x-www-form-urlencoded</code> MIME content type.
27   * 
28   * It contains helper methods used by the URI class, and performs encoding and
29   * decoding in a slightly different way than URLEncoder and URLDecoder.
30   */
31  class URIEncoderDecoder {
32  
33  	static final String digits = "0123456789ABCDEF";
34  
35  	static final String encoding = "UTF8";
36  
37  	/**
38  	 * Validate a string by checking if it contains any characters other than:
39  	 * 
40  	 * 1. letters ('a'..'z', 'A'..'Z') 2. numbers ('0'..'9') 3. characters in
41  	 * the legalset parameter 4. others (Unicode characters that are not in
42  	 * US-ASCII set, and are not ISO Control or are not ISO Space characters)
43  	 * <p>
44  	 * called from URI.Helper.parseURI() to validate each component
45  	 * <p>
46  	 * 
47  	 * @param s
48  	 *            java.lang.String the string to be validated
49  	 * @param legal
50  	 *            java.lang.String the characters allowed in the String s
51  	 * 
52  	 */
53  	static void validate(String s, String legal) throws URISyntaxException {
54  		for (int i = 0; i < s.length();) {
55  			char ch = s.charAt(i);
56  			if (ch == '%') {
57  				do {
58  					if (i + 2 >= s.length()) {
59  						throw new URISyntaxException(s, "K0313",
60  								i);
61  					}
62  					int d1 = Character.digit(s.charAt(i + 1), 16);
63  					int d2 = Character.digit(s.charAt(i + 2), 16);
64  					if (d1 == -1 || d2 == -1) {
65  						throw new URISyntaxException(s, s.substring(i, i + 3), i);
66  					}
67  
68  					i += 3;
69  				} while (i < s.length() && s.charAt(i) == '%');
70  
71  				continue;
72  			}
73  			if (!((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z')
74  					|| (ch >= '0' && ch <= '9') || legal.indexOf(ch) > -1 || (ch > 127
75  					&& !Character.isSpace(ch)))) {
76  				throw new URISyntaxException(s, "K00c1", i);
77  			}
78  			i++;
79  		}
80  	}

82  	static void validateSimple(String s, String legal)
83  			throws URISyntaxException {
84  		for (int i = 0; i < s.length();) {
85  			char ch = s.charAt(i);
86  			if (!((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z')
87  					|| (ch >= '0' && ch <= '9') || legal.indexOf(ch) > -1)) {
88  				throw new URISyntaxException(s, "K00c1", i);
89  			}
90  			i++;
91  		}
92  	}

94  	/**
95  	 * All characters except letters ('a'..'z', 'A'..'Z') and numbers ('0'..'9')
96  	 * and legal characters are converted into their hexidecimal value prepended
97  	 * by '%'.
98  	 * <p>
99  	 * For example: '#' -> %23
100 	 * <p>
101 	 * Other characters, which are Unicode chars that are not US-ASCII, and are
102 	 * not ISO Control or are not ISO Space chars, are preserved.
103 	 * <p>
104 	 * Called from URI.quoteComponent() (for multiple argument constructors)
105 	 * <p>
106 	 * 
107 	 * @param s
108 	 *            java.lang.String the string to be converted
109 	 * @param legal
110 	 *            java.lang.String the characters allowed to be preserved in the
111 	 *            string s
112 	 * @return java.lang.String the converted string
113 	 */
114 	static String quoteIllegal(String s, String legal)
115 			throws UnsupportedEncodingException {
116 		StringBuffer buf = new StringBuffer();
117 		for (int i = 0; i < s.length(); i++) {
118 			char ch = s.charAt(i);
119 			if ((ch >= 'a' && ch <= 'z')
120 					|| (ch >= 'A' && ch <= 'Z')
121 					|| (ch >= '0' && ch <= '9')
122 					|| legal.indexOf(ch) > -1
123 					|| (ch > 127 && !Character.isSpace(ch))) {
124 				buf.append(ch);
125 			} else {
126 				byte[] bytes = new byte[1];
127                 bytes[0] = (byte)ch;
128                 for (int j = 0; j < bytes.length; j++) {
129 					buf.append('%');
130 					buf.append(digits.charAt((bytes[j] & 0xf0) >> 4));
131 					buf.append(digits.charAt(bytes[j] & 0xf));
132 				}
133 			}
134 		}
135 		return buf.toString();
136 	}

138 	/**
139 	 * Other characters, which are Unicode chars that are not US-ASCII, and are
140 	 * not ISO Control or are not ISO Space chars are not preserved. They are
141 	 * converted into their hexidecimal value prepended by '%'.
142 	 * <p>
143 	 * For example: Euro currency symbol -> "%E2%82%AC".
144 	 * <p>
145 	 * Called from URI.toASCIIString()
146 	 * <p>
147 	 * 
148 	 * @param s
149 	 *            java.lang.String the string to be converted
150 	 * @return java.lang.String the converted string
151 	 */
152 	static String encodeOthers(String s) throws UnsupportedEncodingException {
153 		StringBuffer buf = new StringBuffer();
154 		for (int i = 0; i < s.length(); i++) {
155 			char ch = s.charAt(i);
156 			if (ch <= 127)
157 				buf.append(ch);
158 			else {
159 				byte[] bytes = new byte[1];
160                 bytes[0] = (byte)ch;
161                 for (int j = 0; j < bytes.length; j++) {
162 					buf.append('%');
163 					buf.append(digits.charAt((bytes[j] & 0xf0) >> 4));
164 					buf.append(digits.charAt(bytes[j] & 0xf));
165 				}
166 			}
167 		}
168 		return buf.toString();
169 	}

171 	/**
172 	 * Decodes the string argument which is assumed to be encoded in the
173 	 * <code>x-www-form-urlencoded</code> MIME content type using the UTF-8
174 	 * encoding scheme.
175 	 * <p>
176 	 * '%' and two following hex digit characters are converted to the
177 	 * equivalent byte value. All other characters are passed through
178 	 * unmodified.
179 	 * 
180 	 * <p>
181 	 * e.g. "A%20B%20C %24%25" -> "A B C $%"
182 	 * <p>
183 	 * Called from URI.getXYZ() methods
184 	 * <p>
185 	 * 
186 	 * @param s
187 	 *            java.lang.String The encoded string.
188 	 * @return java.lang.String The decoded version.
189 	 */
190 	static String decode(String s) throws UnsupportedEncodingException {
191 
192 		StringBuffer result = new StringBuffer();
193 		ByteArrayOutputStream out = new ByteArrayOutputStream();
194 		for (int i = 0; i < s.length();) {
195 			char c = s.charAt(i);
196 			if (c == '%') {
197 				out.reset();
198 				do {
199 					if (i + 2 >= s.length())
200 						throw new IllegalArgumentException("K01fe: " + i);
201 					int d1 = Character.digit(s.charAt(i + 1), 16);
202 					int d2 = Character.digit(s.charAt(i + 2), 16);
203 					if (d1 == -1 || d2 == -1)
204 						throw new IllegalArgumentException();
205 					out.write((byte) ((d1 << 4) + d2));
206 					i += 3;
207 				} while (i < s.length() && s.charAt(i) == '%');
208 				result.append(out.toString(encoding));
209 				continue;
210 			}
211 			result.append(c);
212 			i++;
213 		}
214 		return result.toString();
215 	}
