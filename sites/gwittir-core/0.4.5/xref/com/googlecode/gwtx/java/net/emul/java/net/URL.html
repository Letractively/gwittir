176 	/**
177 	 * Constructs a new URL by parsing the specification given by
178 	 * <code>spec</code> and using the context provided by
179 	 * <code>context</code>.
180 	 * <p>
181 	 * If the handler argument is non-null, a security check is made to verify
182 	 * that user-defined protocol handlers can be specified.
183 	 * <p>
184 	 * The protocol of the specification is obtained by parsing the
185 	 * <code> spec </code> string.
186 	 * <p>
187 	 * If the <code>spec</code> does not specify a protocol:
188 	 * <ul>
189 	 * <li>If the context is <code>null</code>, then a
190 	 * <code>MalformedURLException</code>.</li>
191 	 * <li>If the context is not <code>null</code>, then the protocol is
192 	 * obtained from the context.</li>
193 	 * </ul>
194 	 * If the <code>spec</code> does specify a protocol:
195 	 * <ul>
196 	 * <li>If the context is <code>null</code>, or specifies a different
197 	 * protocol than the spec, the context is ignored.</li>
198 	 * <li>If the context is not <code>null</code> and specifies the same
199 	 * protocol as the specification, the properties of the new <code>URL</code>
200 	 * are obtained from the context.</li>
201 	 * </ul>
202 	 * 
203 	 * @param context
204 	 *            java.net.URL URL to use as context.
205 	 * @param spec
206 	 *            java.lang.String a URL specification.
207 	 * @param handler
208 	 *            java.net.URLStreamHandler a URLStreamHandler.
209 	 * 
210 	 * @throws MalformedURLException
211 	 *             if the spec could not be parsed as an URL
212 	 */
213 	public URL(URL context, String spec, URLStreamHandler handler)
214 			throws MalformedURLException {
215 		if (handler != null) {
216 			strmHandler = handler;
217 		}
218 
219 		if (spec == null) {
220 			throw new MalformedURLException();
221 		}
222 		spec = spec.trim();
223 
224 		// The spec includes a protocol if it includes a colon character
225 		// before the first occurrence of a slash character. Note that,
226 		// "protocol" is the field which holds this URLs protocol.
227 		int index;
228 		try {
229 			index = spec.indexOf(':');
230 		} catch (NullPointerException e) {
231 			throw new MalformedURLException(e.toString());
232 		}
233 		int startIPv6Addr = spec.indexOf('[');
234 		if (index >= 0) {
235 			if ((startIPv6Addr == -1) || (index < startIPv6Addr)) {
236 				protocol = spec.substring(0, index);
237                 // According to RFC 2396 scheme part should match
238                 // the following expression:
239                 // alpha *( alpha | digit | "+" | "-" | "." )
240                 if (!protocol.matches("\\A\\p{Alpha}[\\p{Alnum}+-.]*\\z") || 
241                     protocol.indexOf('/') >= 0) {
242 					protocol = null;
243 					index = -1;
244 				} else {
245                     // Ignore case in protocol names.
246 					protocol = protocol.toLowerCase();
247                 }
248 			}
249 		}
250 
251 		if (protocol != null) {
252 			// If the context was specified, and it had the same protocol
253 			// as the spec, then fill in the receiver's slots from the values
254 			// in the context but still allow them to be over-ridden later
255 			// by the values in the spec.
256 			if (context != null && protocol.equals(context.getProtocol())) {
257 				String cPath = context.getPath();
258 				if (cPath != null && cPath.startsWith("/")) { //$NON-NLS-1$
259 					set(protocol, context.getHost(), context.getPort(), context
260 							.getAuthority(), context.getUserInfo(), cPath,
261 							context.getQuery(), null);
262 				}
263 				if (strmHandler == null) {
264                     strmHandler = context.strmHandler;
265                 }
266 			}
267 		} else {
268 			// If the spec did not include a protocol, then the context
269 			// *must* be specified. Fill in the receiver's slots from the
270 			// values in the context, but still allow them to be over-ridden
271 			// by the values in the ("relative") spec.
272 			if (context == null) {
273                 throw new MalformedURLException(spec); //$NON-NLS-1$
274             }
275 			set(context.getProtocol(), context.getHost(), context.getPort(),
276 					context.getAuthority(), context.getUserInfo(), context
277 							.getPath(), context.getQuery(), null);
278 			if (strmHandler == null) {
279                 strmHandler = context.strmHandler;
280             }
281 		}
282 
283 		// If the stream handler has not been determined, set it
284 		// to the default for the specified protocol.
285 		if (strmHandler == null) {
286 			//setupStreamHandler();
287 			if (strmHandler == null) {
288                 throw new MalformedURLException(protocol); //$NON-NLS-1$
289             }
290 		}
291 
292 		// Let the handler parse the URL. If the handler throws
293 		// any exception, throw MalformedURLException instead.
294 		//
295 		// Note: We want "index" to be the index of the start of the scheme
296 		// specific part of the URL. At this point, it will be either
297 		// -1 or the index of the colon after the protocol, so we
298 		// increment it to point at either character 0 or the character
299 		// after the colon.
300 		try {
301 			strmHandler.parseURL(this, spec, ++index, spec.length());
302 		} catch (Exception e) {
303 			throw new MalformedURLException(e.toString());
304 		}
305 
306 		if (port < -1) {
307             throw new MalformedURLException("" + port); //$NON-NLS-1$
308         }
309 	}

350 	 * Constructs a new URL instance using the arguments provided.
351 	 * <p>
352 	 * If the handler argument is non-null, a security check is made to verify
353 	 * that user-defined protocol handlers can be specified.
354 	 * 
355 	 * @param protocol
356 	 *            the protocol for the URL.
357 	 * @param host
358 	 *            the name of the host.
359 	 * @param port
360 	 *            the port number.
361 	 * @param file
362 	 *            the name of the resource.
363 	 * @param handler
364 	 *            the stream handler that this URL uses.
365 	 * 
366 	 * @throws MalformedURLException
367 	 *             if the parameters do not represent an URL.
368 	 */
369 	public URL(String protocol, String host, int port, String file,
370 			URLStreamHandler handler) throws MalformedURLException {
371 		if (port < -1) {
372             throw new MalformedURLException("" + port); //$NON-NLS-1$
373         }
374 
375 		if (host != null && host.indexOf(":") != -1 && host.charAt(0) != '[') { //$NON-NLS-1$
376 			host = "[" + host + "]"; //$NON-NLS-1$ //$NON-NLS-2$
377 		}
378 
379 		if (protocol != null){
380 		    this.protocol = protocol;
381         } else {
382             throw new NullPointerException( protocol); //$NON-NLS-1$
383         }
384         
385 		this.host = host;
386 		this.port = port;
387 
388 		// Set the fields from the arguments. Handle the case where the
389 		// passed in "file" includes both a file and a reference part.
390 		int index = -1;
391 		index = file.indexOf("#", file.lastIndexOf("/")); //$NON-NLS-1$ //$NON-NLS-2$
392 		if (index >= 0) {
393 			this.file = file.substring(0, index);
394 			ref = file.substring(index + 1);
395 		} else {
396 			this.file = file;
397 		}
398 		fixURL(false);
399 
400 		// Set the stream handler for the URL either to the handler
401 		// argument if it was specified, or to the default for the
402 		// receiver's protocol if the handler was null.
403 		if (handler == null) {
404 			//setupStreamHandler();
405 			if (strmHandler == null) {
406                 throw new MalformedURLException(protocol); //$NON-NLS-1$
407             }
408 		} else {
409 			strmHandler = handler;
410 		}
411 	}

413 	void fixURL(boolean fixHost) {
414 		int index;
415 		if (host != null && host.length() > 0) {
416 			authority = host;
417 			if (port != -1) {
418                 authority = authority + ":" + port; //$NON-NLS-1$
419             }
420 		}
421 		if (fixHost) {
422 			if (host != null && (index = host.lastIndexOf('@')) > -1) {
423 				userInfo = host.substring(0, index);
424 				host = host.substring(index + 1);
425 			} else {
426 				userInfo = null;
427 			}
428 		}
429 		if (file != null && (index = file.indexOf('?')) > -1) {
430 			query = file.substring(index + 1);
431 			path = file.substring(0, index);
432 		} else {
433 			query = null;
434 			path = file;
435 		}
436 	}
