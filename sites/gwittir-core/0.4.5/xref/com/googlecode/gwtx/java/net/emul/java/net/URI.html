75  	public URI(String scheme, String ssp, String frag)
76  			throws URISyntaxException {
77  		StringBuffer uri = new StringBuffer();
78  		if (scheme != null) {
79  			uri.append(scheme);
80  			uri.append(':');
81  		}
82  		if (ssp != null) {
83  			// QUOTE ILLEGAL CHARACTERS
84  			uri.append(quoteComponent(ssp, allLegal));
85  		}
86  		if (frag != null) {
87  			uri.append('#');
88  			// QUOTE ILLEGAL CHARACTERS
89  			uri.append(quoteComponent(frag, allLegal));
90  		}
91  
92  		new Helper().parseURI(uri.toString(), false);
93  	}

95  	public URI(String scheme, String userinfo, String host, int port,
96  			String path, String query, String fragment)
97  			throws URISyntaxException {
98  
99  		if (scheme == null && userinfo == null && host == null &&
100 				path == null && query == null && fragment == null) {
101             return;
102         }
103 
104 		if (scheme != null && path != null && path.length() > 0
105 				&& path.charAt(0) != '/') {
106 			throw new URISyntaxException(path, "K0302");
107 		}
108 
109 		StringBuffer uri = new StringBuffer();
110 		if (scheme != null) {
111 			uri.append(scheme);
112 			uri.append(':');
113 		}
114 
115 		if (userinfo != null || host != null || port != -1) {
116             uri.append("//");
117         }
118 
119 		if (userinfo != null) {
120 			// QUOTE ILLEGAL CHARACTERS in userinfo
121 			uri.append(quoteComponent(userinfo, someLegal));
122 			uri.append('@');
123 		}
124 
125 		if (host != null) {
126 			// check for ipv6 addresses that hasn't been enclosed
127 			// in square brackets
128 			if (host.indexOf(':') != -1 && host.indexOf(']') == -1
129 					&& host.indexOf('[') == -1) {
130                 host = "[" + host + "]";
131             }
132 			uri.append(host);
133 		}
134 
135 		if (port != -1) {
136 			uri.append(':');
137 			uri.append(port);
138 		}
139 
140 		if (path != null) {
141 			// QUOTE ILLEGAL CHARS
142 			uri.append(quoteComponent(path, "/@" + someLegal));
143 		}
144 
145 		if (query != null) {
146 			uri.append('?');
147 			// QUOTE ILLEGAL CHARS
148 			uri.append(quoteComponent(query, allLegal));
149 		}
150 
151 		if (fragment != null) {
152 			// QUOTE ILLEGAL CHARS
153 			uri.append('#');
154 			uri.append(quoteComponent(fragment, allLegal));
155 		}
156 
157 		new Helper().parseURI(uri.toString(), true);
158 	}

160 	public URI(String scheme, String host, String path, String fragment)
161 			throws URISyntaxException {
162 		this(scheme, null, host, -1, path, null, fragment);
163 	}

165 	public URI(String scheme, String authority, String path, String query,
166 			String fragment) throws URISyntaxException {
167 		if (scheme != null && path != null && path.length() > 0
168 				&& path.charAt(0) != '/') {
169 			throw new URISyntaxException(path, "K0302");
170 		}
171 
172 		StringBuffer uri = new StringBuffer();
173 		if (scheme != null) {
174 			uri.append(scheme);
175 			uri.append(':');
176 		}
177 		if (authority != null) {
178 			uri.append("//");
179 			// QUOTE ILLEGAL CHARS
180 			uri.append(quoteComponent(authority, "@[]" + someLegal));
181 		}
182 
183 		if (path != null) {
184 			// QUOTE ILLEGAL CHARS
185 			uri.append(quoteComponent(path, "/@" + someLegal));
186 		}
187 		if (query != null) {
188 			// QUOTE ILLEGAL CHARS
189 			uri.append('?');
190 			uri.append(quoteComponent(query, allLegal));
191 		}
192 		if (fragment != null) {
193 			// QUOTE ILLEGAL CHARS
194 			uri.append('#');
195 			uri.append(quoteComponent(fragment, allLegal));
196 		}
197 
198 		new Helper().parseURI(uri.toString(), false);
199 	}

203 		private void parseURI(String uri, boolean forceServer)
204 				throws URISyntaxException {
205 			String temp = uri;
206 			// assign uri string to the input value per spec
207 			string = uri;
208 			int index, index1, index2, index3;
209 			// parse into Fragment, Scheme, and SchemeSpecificPart
210 			// then parse SchemeSpecificPart if necessary
211 
212 			// Fragment
213 			index = temp.indexOf('#');
214 			if (index != -1) {
215 				// remove the fragment from the end
216 				fragment = temp.substring(index + 1);
217 				validateFragment(uri, fragment, index + 1);
218 				temp = temp.substring(0, index);
219 			}
220 
221 			// Scheme and SchemeSpecificPart
222 			index = index1 = temp.indexOf(':');
223 			index2 = temp.indexOf('/');
224 			index3 = temp.indexOf('?');
225 
226 			// if a '/' or '?' occurs before the first ':' the uri has no
227 			// specified scheme, and is therefore not absolute
228 			if (index != -1 && (index2 >= index || index2 == -1)
229 					&& (index3 >= index || index3 == -1)) {
230 				// the characters up to the first ':' comprise the scheme
231 				absolute = true;
232 				scheme = temp.substring(0, index);
233 				if (scheme.length() == 0) {
234 					throw new URISyntaxException(uri, "K0342",
235 							index);
236 				}
237 				validateScheme(uri, scheme, 0);
238 				schemespecificpart = temp.substring(index + 1);
239 				if (schemespecificpart.length() == 0) {
240 					throw new URISyntaxException(uri, "K0303",
241 							index + 1);
242 				}
243 			} else {
244 				absolute = false;
245 				schemespecificpart = temp;
246 			}
247 
248 			if (scheme == null || schemespecificpart.length() > 0
249 					&& schemespecificpart.charAt(0) == '/') {
250 				opaque = false;
251 				// the URI is hierarchical
252 
253 				// Query
254 				temp = schemespecificpart;
255 				index = temp.indexOf('?');
256 				if (index != -1) {
257 					query = temp.substring(index + 1);
258 					temp = temp.substring(0, index);
259 					validateQuery(uri, query, index2 + 1 + index);
260 				}
261 
262 				// Authority and Path
263 				if (temp.startsWith("//")) {
264 					index = temp.indexOf('/', 2);
265 					if (index != -1) {
266 						authority = temp.substring(2, index);
267 						path = temp.substring(index);
268 					} else {
269 						authority = temp.substring(2);
270 						if (authority.length() == 0 && query == null
271 								&& fragment == null) {
272 							throw new URISyntaxException(uri, "K0304", uri.length());
273 						}
274 
275 						path = "";
276 						// nothing left, so path is empty (not null, path should
277 						// never be null)
278 					}
279 
280 					if (authority.length() == 0) {
281                         authority = null;
282                     } else {
283                         validateAuthority(uri, authority, index1 + 3);
284                     }
285 				} else { // no authority specified
286 					path = temp;
287 				}
288 
289 				int pathIndex = 0;
290 				if (index2 > -1) {
291                     pathIndex += index2;
292                 }
293 				if (index > -1) {
294                     pathIndex += index;
295                 }
296 				validatePath(uri, path, pathIndex);
297 			} else { // if not hierarchical, URI is opaque
298 				opaque = true;
299 				validateSsp(uri, schemespecificpart, index2 + 2 + index);
300 			}
301 
302 			parseAuthority(forceServer);
303 		}
304 
305 		private void validateScheme(String uri, String scheme, int index)
306 				throws URISyntaxException {
307 			// first char needs to be an alpha char
308 			char ch = scheme.charAt(0);
309 			if (!((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z'))) {
310 				throw new URISyntaxException(uri, "K0305", 0);
311 			}
312 
313 			try {
314 				URIEncoderDecoder.validateSimple(scheme, "+-.");
315 			} catch (URISyntaxException e) {
316 				throw new URISyntaxException(uri, "K0305", index
317 						+ e.getIndex());
318 			}
319 		}

321 		private void validateSsp(String uri, String ssp, int index)
322 				throws URISyntaxException {
323 			try {
324 				URIEncoderDecoder.validate(ssp, allLegal);
325 			} catch (URISyntaxException e) {
326 				throw new URISyntaxException(uri, e.getReason(), index + e.getIndex());
327 			}
328 		}

367 		/**
368 		 * determine the host, port and userinfo if the authority parses
369 		 * successfully to a server based authority
370 		 * 
371 		 * behavour in error cases: if forceServer is true, throw
372 		 * URISyntaxException with the proper diagnostic messages. if
373 		 * forceServer is false assume this is a registry based uri, and just
374 		 * return leaving the host, port and userinfo fields undefined.
375 		 * 
376 		 * and there are some error cases where URISyntaxException is thrown
377 		 * regardless of the forceServer parameter e.g. malformed ipv6 address
378 		 */
379 		private void parseAuthority(boolean forceServer)
380 				throws URISyntaxException {
381 			if (authority == null) {
382                 return;
383             }
384 
385 			String temp, tempUserinfo = null, tempHost = null;
386 			int index, hostindex = 0;
387 			int tempPort = -1;
388 
389 			temp = authority;
390 			index = temp.indexOf('@');
391 			if (index != -1) {
392 				// remove user info
393 				tempUserinfo = temp.substring(0, index);
394 				validateUserinfo(authority, tempUserinfo, 0);
395 				temp = temp.substring(index + 1); // host[:port] is left
396 				hostindex = index + 1;
397 			}
398 
399 			index = temp.lastIndexOf(':');
400 			int endindex = temp.indexOf(']');
401 
402 			if (index != -1 && endindex < index) {
403 				// determine port and host
404 				tempHost = temp.substring(0, index);
405 				try {
406 					tempPort = Integer.parseInt(temp.substring(index + 1));
407 					if (tempPort < 0) {
408 						if (forceServer) {
409                             throw new URISyntaxException(authority, "K00b1", hostindex + index + 1);
410                         }
411 						return;
412 					}
413 				} catch (NumberFormatException e) {
414 					if (forceServer) {
415                         throw new URISyntaxException(authority, "K00b1", hostindex + index + 1);
416                     }
417 					return;
418 				}
419 			} else {
420                 tempHost = temp;
421             }
422 
423 			if (tempHost.equals("")) {
424 				if (forceServer) {
425                     throw new URISyntaxException(authority, "K030c", hostindex);
426                 }
427 				return;
428 			}
429 
430 			if (!isValidHost(forceServer, tempHost)) {
431                 return;
432             }
433 
434 			// this is a server based uri,
435 			// fill in the userinfo, host and port fields
436 			userinfo = tempUserinfo;
437 			host = tempHost;
438 			port = tempPort;
439 			serverAuthority = true;
440 		}

442 		private void validateUserinfo(String uri, String userinfo, int index)
443 				throws URISyntaxException {
444 			for (int i = 0; i < userinfo.length(); i++) {
445 				char ch = userinfo.charAt(i);
446 				if (ch == ']' || ch == '[') {
447 					throw new URISyntaxException(uri, "K030d", index + i);
448 				}
449 
450 			}
451 		}

453 		/**
454 		 * distinguish between IPv4, IPv6, domain name and validate it based on
455 		 * its type
456 		 */
457 		private boolean isValidHost(boolean forceServer, String host)
458 				throws URISyntaxException {
459 			if (host.charAt(0) == '[') {
460 				// ipv6 address
461 				if (host.charAt(host.length() - 1) != ']') {
462                     throw new URISyntaxException(host, "K030e", 0);
463                 }
464                 if (!isValidIP6Address(host)) {
465                     throw new URISyntaxException(host, "K030f");
466                 }
467                 return true;
468 			}
469 
470 			// '[' and ']' can only be the first char and last char
471 			// of the host name
472 			if (host.indexOf('[') != -1 || host.indexOf(']') != -1) {
473 				throw new URISyntaxException(host, "K0310", 0);
474 			}
475 
476 			int index = host.lastIndexOf('.');
477 			if (index < 0 || index == host.length() - 1
478 					|| !Character.isDigit(host.charAt(index + 1))) {
479 				// domain name
480 				if (isValidDomainName(host)) {
481                     return true;
482                 }
483                 if (forceServer) {
484                     throw new URISyntaxException(host, "K0310", 0);
485                 }
486                 return false;
487             }
488 
489 			// IPv4 address
490 			if (isValidIPv4Address(host)) {
491                 return true;
492             }
493             if (forceServer) {
494                 throw new URISyntaxException(host, "K0311", 0);
495             }
496             return false;
497 		}

499 		private boolean isValidDomainName(String host) {
500 			try {
501 				URIEncoderDecoder.validateSimple(host, "-.");
502 			} catch (URISyntaxException e) {
503 				return false;
504 			}
505 
506 			String label = null;
507 			StringTokenizer st = new StringTokenizer(host, ".");
508 			while (st.hasMoreTokens()) {
509 				label = st.nextToken();
510 				if (label.startsWith("-") || label.endsWith("-")) {
511                     return false;
512                 }
513 			}
514 
515 			if (!label.equals(host)) {
516 				char ch = label.charAt(0);
517 				if (ch >= '0' && ch <= '9') {
518                     return false;
519                 }
520 			}
521 			return true;
522 		}

524 		private boolean isValidIPv4Address(String host) {
525 			int index;
526 			int index2;
527 			try {
528 				int num;
529 				index = host.indexOf('.');
530 				num = Integer.parseInt(host.substring(0, index));
531 				if (num < 0 || num > 255) {
532                     return false;
533                 }
534 				index2 = host.indexOf('.', index + 1);
535 				num = Integer.parseInt(host.substring(index + 1, index2));
536 				if (num < 0 || num > 255) {
537                     return false;
538                 }
539 				index = host.indexOf('.', index2 + 1);
540 				num = Integer.parseInt(host.substring(index2 + 1, index));
541 				if (num < 0 || num > 255) {
542                     return false;
543                 }
544 				num = Integer.parseInt(host.substring(index + 1));
545 				if (num < 0 || num > 255) {
546                     return false;
547                 }
548 			} catch (Exception e) {
549 				return false;
550 			}
551 			return true;
552 		}

554 		private boolean isValidIP6Address(String ipAddress) {
555 			int length = ipAddress.length();
556 			boolean doubleColon = false;
557 			int numberOfColons = 0;
558 			int numberOfPeriods = 0;
559 			String word = "";
560 			char c = 0;
561 			char prevChar = 0;
562 			int offset = 0; // offset for [] ip addresses
563 
564 			if (length < 2) {
565                 return false;
566             }
567 
568 			for (int i = 0; i < length; i++) {
569 				prevChar = c;
570 				c = ipAddress.charAt(i);
571 				switch (c) {
572 
573 				// case for an open bracket [x:x:x:...x]
574 				case '[':
575 					if (i != 0) {
576                         return false; // must be first character
577                     }
578 					if (ipAddress.charAt(length - 1) != ']') {
579                         return false; // must have a close ]
580                     }
581                     if ((ipAddress.charAt(1) == ':')
582                             && (ipAddress.charAt(2) != ':')) {
583                         return false;
584                     }
585 					offset = 1;
586 					if (length < 4) {
587                         return false;
588                     }
589 					break;
590 
591 				// case for a closed bracket at end of IP [x:x:x:...x]
592 				case ']':
593 					if (i != length - 1) {
594                         return false; // must be last charcter
595                     }
596 					if (ipAddress.charAt(0) != '[') {
597                         return false; // must have a open [
598                     }
599 					break;
600 
601 				// case for the last 32-bits represented as IPv4
602 				// x:x:x:x:x:x:d.d.d.d
603 				case '.':
604 					numberOfPeriods++;
605 					if (numberOfPeriods > 3) {
606                         return false;
607                     }
608 					if (!isValidIP4Word(word)) {
609                         return false;
610                     }
611 					if (numberOfColons != 6 && !doubleColon) {
612                         return false;
613                     }
614 					// a special case ::1:2:3:4:5:d.d.d.d allows 7 colons with
615 					// an IPv4 ending, otherwise 7 :'s is bad
616 					if (numberOfColons == 7
617 							&& ipAddress.charAt(0 + offset) != ':'
618 							&& ipAddress.charAt(1 + offset) != ':') {
619                         return false;
620                     }
621 					word = "";
622 					break;
623 
624 				case ':':
625 					numberOfColons++;
626 					if (numberOfColons > 7) {
627                         return false;
628                     }
629 					if (numberOfPeriods > 0) {
630                         return false;
631                     }
632 					if (prevChar == ':') {
633 						if (doubleColon) {
634                             return false;
635                         }
636 						doubleColon = true;
637 					}
638 					word = "";
639 					break;
640 
641 				default:
642 					if (word.length() > 3) {
643                         return false;
644                     }
645 					if (!isValidHexChar(c)) {
646                         return false;
647                     }
648 					word += c;
649 				}
650 			}
651 
652 			// Check if we have an IPv4 ending
653 			if (numberOfPeriods > 0) {
654 				if (numberOfPeriods != 3 || !isValidIP4Word(word)) {
655                     return false;
656                 }
657 			} else {
658 				// If we're at then end and we haven't had 7 colons then there
659 				// is a problem
660 				// unless we encountered a doubleColon
661 				if (numberOfColons != 7 && !doubleColon) {
662 					return false;
663 				}
664 
665 				// If we have an empty word at the end, it means we ended in
666 				// either a : or a .
667 				// If we did not end in :: then this is invalid
668 				if (word == "" && ipAddress.charAt(length - 1 - offset) != ':'
669 						&& ipAddress.charAt(length - 2 - offset) != ':') {
670 					return false;
671 				}
672 			}
673 
674 			return true;
675 		}

677 		private boolean isValidIP4Word(String word) {
678 			char c;
679 			if (word.length() < 1 || word.length() > 3) {
680                 return false;
681             }
682 			for (int i = 0; i < word.length(); i++) {
683 				c = word.charAt(i);
684 				if (!(c >= '0' && c <= '9')) {
685                     return false;
686                 }
687 			}
688 			if (Integer.parseInt(word) > 255) {
689                 return false;
690             }
691 			return true;
692 		}
693 
694 		private boolean isValidHexChar(char c) {
695 
696 			return (c >= '0' && c <= '9') || (c >= 'A' && c <= 'F')
697 					|| (c >= 'a' && c <= 'f');
698 		}

723 	public int compareTo(Object o) {
724         URI uri = (URI)o;
725         int ret = 0;
726 
727 		// compare schemes
728 		if (scheme == null && uri.scheme != null) {
729             return -1;
730         } else if (scheme != null && uri.scheme == null) {
731             return 1;
732         } else if (scheme != null && uri.scheme != null) {
733 			ret = scheme.toLowerCase().compareTo(uri.scheme.toLowerCase());
734 			if (ret != 0) {
735                 return ret;
736             }
737 		}
738 
739 		// compare opacities
740 		if (!opaque && uri.opaque) {
741             return -1;
742         } else if (opaque && !uri.opaque) {
743             return 1;
744         } else if (opaque && uri.opaque) {
745 			ret = schemespecificpart.compareTo(uri.schemespecificpart);
746 			if (ret != 0) {
747                 return ret;
748             }
749 		} else {
750 
751 			// otherwise both must be hierarchical
752 
753 			// compare authorities
754 			if (authority != null && uri.authority == null) {
755                 return 1;
756             } else if (authority == null && uri.authority != null) {
757                 return -1;
758             } else if (authority != null && uri.authority != null) {
759 				if (host != null && uri.host != null) {
760 					// both are server based, so compare userinfo, host, port
761 					if (userinfo != null && uri.userinfo == null) {
762                         return 1;
763                     } else if (userinfo == null && uri.userinfo != null) {
764                         return -1;
765                     } else if (userinfo != null && uri.userinfo != null) {
766 						ret = userinfo.compareTo(uri.userinfo);
767 						if (ret != 0) {
768                             return ret;
769                         }
770 					}
771 
772 					// userinfo's are the same, compare hostname
773 					ret = host.toLowerCase().compareTo(uri.host.toLowerCase());
774 					if (ret != 0) {
775                         return ret;
776                     }
777 
778 					// compare port
779 					if (port != uri.port) {
780                         return port - uri.port;
781                     }
782 				} else { // one or both are registry based, compare the whole
783 					// authority
784 					ret = authority.compareTo(uri.authority);
785 					if (ret != 0) {
786                         return ret;
787                     }
788 				}
789 			}
790 
791 			// authorities are the same
792 			// compare paths
793 			ret = path.compareTo(uri.path);
794 			if (ret != 0) {
795                 return ret;
796             }
797 
798 			// compare queries
799 
800 			if (query != null && uri.query == null) {
801                 return 1;
802             } else if (query == null && uri.query != null) {
803                 return -1;
804             } else if (query != null && uri.query != null) {
805 				ret = query.compareTo(uri.query);
806 				if (ret != 0) {
807                     return ret;
808                 }
809 			}
810 
811 		}
812 
813 		// everything else is identical, so compare fragments
814 		if (fragment != null && uri.fragment == null) {
815             return 1;
816         } else if (fragment == null && uri.fragment != null) {
817             return -1;
818         } else if (fragment != null && uri.fragment != null) {
819 			ret = fragment.compareTo(uri.fragment);
820 			if (ret != 0) {
821                 return ret;
822             }
823 		}
824 
825 		// identical
826 		return 0;
827 	}

856 	private String convertHexToLowerCase(String s) {
857 		// takes a string that may contain hex sequences like %F1 or %2b
858 		// and converts the hex values following the '%' to lowercase
859 		StringBuffer result = new StringBuffer("");
860 		if (s.indexOf('%') == -1) {
861             return s;
862         }
863 
864 		int index = 0, previndex = 0;
865 		while ((index = s.indexOf('%', previndex)) != -1) {
866 			result.append(s.substring(previndex, index + 1));
867 			result.append(s.substring(index + 1, index + 3).toLowerCase());
868 			index += 3;
869 			previndex = index;
870 		}
871 		return result.toString();
872 	}

874 	private boolean equalsHexCaseInsensitive(String first, String second) {
875 		// takes two strings that may contain hex sequences like %F1 or %2b
876 		// and compares them, ignoring case for the hex values
877 		// hex values must always occur in pairs like above
878 
879 		if (first.indexOf('%') != second.indexOf('%')) {
880             return first.equals(second);
881         }
882 
883 		int index = 0, previndex = 0;
884 		while ((index = first.indexOf('%', previndex)) != -1
885 				&& second.indexOf('%', previndex) == index) {
886 			boolean match = first.substring(previndex, index).equals(
887 					second.substring(previndex, index));
888 			if (!match) {
889                 return false;
890             }
891 
892 			match = first.substring(index + 1, index + 3).equalsIgnoreCase(
893 					second.substring(index + 1, index + 3));
894 			if (!match) {
895                 return false;
896             }
897 
898 			index += 3;
899 			previndex = index;
900 		}
901 		return first.substring(previndex).equals(second.substring(previndex));
902 	}

1140 	private String normalize(String path) {
1141 		// normalize path, and return
1142 		// the resulting string
1143 
1144 		int index = path.indexOf(':');
1145 		int index2 = path.indexOf('/');
1146 
1147 		// //This code will cause the function to return right away if this
1148 		// //URI is already in normalized form
1149 		// //(the last line tests whether there is a ':' in the first segment)
1150 		// //is it worth having this test? (if removed, the method still works)
1151 		// if(!path.startsWith("./") && !path.endsWith("/.") &&
1152 		// !path.endsWith("/..") &&
1153 		// path.indexOf("/./") == -1 && path.indexOf("/../") == -1 &&
1154 		// !path.equals(".") && //(special case)
1155 		// (index > index2 && index2!=-1 || index == -1))
1156 		// return path;
1157 
1158 		StringBuffer newpath = new StringBuffer();
1159 
1160 		index = 0;
1161 		index2 = 0;
1162 		int pathlen = path.length();
1163 		index = -1;
1164 		// count the number of '/'s, to determine number of segments
1165 		int size = 0;
1166 		if (pathlen > 0 && path.charAt(0) != '/') {
1167             size++;
1168         }
1169 		while ((index = path.indexOf('/', index + 1)) != -1) {
1170             if (index + 1 < pathlen && path.charAt(index + 1) != '/') {
1171                 size++;
1172             }
1173         }
1174 
1175 		String[] seglist = new String[size];
1176 		boolean[] include = new boolean[size];
1177 
1178 		// break the path into segments and store in the list
1179 		int current = 0;
1180 		index = (pathlen > 0 && path.charAt(0) == '/') ? 1 : 0;
1181 		while ((index2 = path.indexOf('/', index + 1)) != -1) {
1182 			seglist[current++] = path.substring(index, index2);
1183 			index = index2 + 1;
1184 		}
1185 
1186 		// if current==size, then the last character was a slash
1187 		// and there are no more segments
1188 		if (current < size) {
1189             seglist[current] = path.substring(index);
1190         }
1191 
1192 		// determine which segments get included in the normalized path
1193 		for (int i = 0; i < size; i++) {
1194 			include[i] = true;
1195 			if (seglist[i].equals("..")) {
1196 				int remove = i - 1;
1197 				// search back to find a segment to remove, if possible
1198 				while (remove > -1 && !include[remove]) {
1199 					remove--;
1200 				}
1201 				// if we find a segment to remove, remove it and the ".."
1202 				// segment
1203 				if (remove > -1 && !seglist[remove].equals("..")) {
1204 					include[remove] = false;
1205 					include[i] = false;
1206 				}
1207 			} else if (seglist[i].equals(".")) {
1208                 include[i] = false;
1209             }
1210 		}
1211 
1212 		// put the path back together
1213 		newpath = new StringBuffer();
1214 		if (path.startsWith("/")) {
1215             newpath.append('/');
1216         }
1217 
1218 		for (int i = 0; i < seglist.length; i++) {
1219             if (include[i]) {
1220 				newpath.append(seglist[i]);
1221 				newpath.append('/');
1222 			}
1223         }
1224 
1225 		// if we used at least one segment
1226 		// and the path previously ended with a slash
1227 		// and the last segment is still used, then
1228 		// delete the extra trailing '/'
1229 		if (!path.endsWith("/") && seglist.length > 0
1230 				&& include[seglist.length - 1]) {
1231             newpath.deleteCharAt(newpath.length() - 1);
1232         }
1233 
1234 		String result = newpath.toString();
1235 
1236 		// check for a ':' in the first segment
1237 		// if one exists, prepend "./" to normalize
1238 		index = result.indexOf(':');
1239 		index2 = result.indexOf('/');
1240 		if (index != -1 && (index < index2 || index2 == -1)) {
1241 			newpath.insert(0, "./");
1242 			result = newpath.toString();
1243 		}
1244 		return result;
1245 	}

1276 	public URI relativize(URI relative) {
1277 		if (relative.opaque || opaque) {
1278             return relative;
1279         }
1280 
1281 		if (scheme == null ? relative.scheme != null : !scheme
1282 				.equals(relative.scheme)) {
1283             return relative;
1284         }
1285 
1286 		if (authority == null ? relative.authority != null : !authority
1287 				.equals(relative.authority)) {
1288             return relative;
1289         }        
1290 
1291 		// normalize both paths
1292         String thisPath = normalize(path);
1293 		String relativePath = normalize(relative.path);
1294 
1295         /*
1296          * if the paths aren't equal, then we need to determine if this
1297          * URI's path is a parent path (begins with) the relative URI's
1298          * path
1299          */
1300 		if (!thisPath.equals(relativePath)) {
1301             // if this URI's path doesn't end in a '/', add one 
1302             if (!thisPath.endsWith("/")) {
1303                 thisPath = thisPath + '/';
1304             }
1305             /*
1306              * if the relative URI's path doesn't start with this URI's path,
1307              * then just return the relative URI; the URIs have nothing in
1308              * common
1309              */
1310             if (!relativePath.startsWith(thisPath)) {
1311                 return relative;
1312             }
1313         }
1314 
1315 		URI result = new URI();
1316 		result.fragment = relative.fragment;
1317 		result.query = relative.query;
1318         // the result URI is the remainder of the relative URI's path
1319 		result.path = relativePath.substring(thisPath.length());
1320 		return result;
1321 	}

1323 	public URI resolve(URI relative) {
1324 		if (relative.absolute || opaque) {
1325             return relative;
1326         }
1327 
1328 		URI result;
1329 		if (relative.path.equals("") && relative.scheme == null
1330 				&& relative.authority == null && relative.query == null
1331 				&& relative.fragment != null) {
1332 			// if the relative URI only consists of fragment,
1333 			// the resulved URI is very similar to this URI,
1334 			// except that it has the fragement from the relative URI.
1335 			result = duplicate();
1336 			result.fragment = relative.fragment;
1337 			// no need to re-calculate the scheme specific part,
1338 			// since fragment is not part of scheme specific part.
1339 			return result;
1340 		}
1341 
1342 		if (relative.authority != null) {
1343 			// if the relative URI has authority,
1344 			// the resolved URI is almost the same as the relative URI,
1345 			// except that it has the scheme of this URI.
1346 			result = relative.duplicate();
1347 			result.scheme = scheme;
1348 			result.absolute = absolute;
1349 		} else {
1350 			// since relative URI has no authority,
1351 			// the resolved URI is very similar to this URI,
1352 			// except that it has the query and fragment of the relative URI,
1353 			// and the path is different.
1354 			result = duplicate();
1355 			result.fragment = relative.fragment;
1356 			result.query = relative.query;
1357 			if (relative.path.startsWith("/")) {
1358 				result.path = relative.path;
1359 			} else {
1360 				// resolve a relative reference
1361 				int endindex = path.lastIndexOf('/') + 1;
1362 				result.path = normalize(path.substring(0, endindex)
1363 						+ relative.path);
1364 			}
1365 			// re-calculate the scheme specific part since
1366 			// query and path of the resolved URI is different from this URI.
1367 			result.setSchemeSpecificPart();
1368 		}
1369 		return result;
1370 	}
