27  /**
28   * MutableDateTime is the standard implementation of a modifiable datetime class.
29   * It holds the datetime as milliseconds from the Java epoch of 1970-01-01T00:00:00Z.
30   * <p>
31   * This class uses a Chronology internally. The Chronology determines how the
32   * millisecond instant value is converted into the date time fields.
33   * The default Chronology is <code>ISOChronology</code> which is the agreed
34   * international standard and compatable with the modern Gregorian calendar.
35   * <p>
36   * Each individual field can be accessed in two ways:
37   * <ul>
38   * <li><code>getHourOfDay()</code>
39   * <li><code>hourOfDay().get()</code>
40   * </ul>
41   * The second technique also provides access to other useful methods on the
42   * field:
43   * <ul>
44   * <li>get numeric value
45   * <li>set numeric value
46   * <li>add to numeric value
47   * <li>add to numeric value wrapping with the field
48   * <li>get text vlaue
49   * <li>get short text value
50   * <li>set text value
51   * <li>field maximum value
52   * <li>field minimum value
53   * </ul>
54   *
55   * <p>
56   * MutableDateTime is mutable and not thread-safe, unless concurrent threads
57   * are not invoking mutator methods.
58   *
59   * @author Guy Allard
60   * @author Brian S O'Neill
61   * @author Stephen Colebourne
62   * @author Mike Schrag
63   * @since 1.0
64   * @see DateTime
65   */
66  public class MutableDateTime
67          extends BaseDateTime
68          implements ReadWritableDateTime, Cloneable, Serializable {

326     /**
327      * Sets the status of rounding to use the specified field and ROUND_FLOOR mode.
328      * A null field will disable rounding.
329      * Once set, the instant is then rounded using the new field and mode.
330      * <p>
331      * Enabling rounding will cause all subsequent calls to {@link #setMillis(long)}
332      * to be rounded. This can be used to control the precision of the instant,
333      * for example by setting a rounding field of minuteOfDay, the seconds and
334      * milliseconds will always be zero.
335      *
336      * @param field rounding field or null to disable
337      */
338     public void setRounding(DateTimeField field) {
339         setRounding(field, MutableDateTime.ROUND_FLOOR);
340     }
341 
342     /**
343      * Sets the status of rounding to use the specified field and mode.
344      * A null field or mode of ROUND_NONE will disable rounding.
345      * Once set, the instant is then rounded using the new field and mode.
346      * <p>
347      * Enabling rounding will cause all subsequent calls to {@link #setMillis(long)}
348      * to be rounded. This can be used to control the precision of the instant,
349      * for example by setting a rounding field of minuteOfDay, the seconds and
350      * milliseconds will always be zero.
351      *
352      * @param field  rounding field or null to disable
353      * @param mode  rounding mode or ROUND_NONE to disable
354      * @throws IllegalArgumentException if mode is unknown, no exception if field is null
355      */
356     public void setRounding(DateTimeField field, int mode) {
357         if (field != null && (mode < ROUND_NONE || mode > ROUND_HALF_EVEN)) {
358             throw new IllegalArgumentException("Illegal rounding mode: " + mode);
359         }
360         iRoundingField = (mode == ROUND_NONE ? null : field);
361         iRoundingMode = (field == null ? ROUND_NONE : mode);
362         setMillis(getMillis());
363     }

365     //-----------------------------------------------------------------------
366     /**
367      * Set the milliseconds of the datetime.
368      * <p>
369      * All changes to the millisecond field occurs via this method.
370      *
371      * @param instant  the milliseconds since 1970-01-01T00:00:00Z to set the
372      * datetime to
373      */
374     public void setMillis(long instant) {
375         switch (iRoundingMode) {
376             case ROUND_NONE:
377                 break;
378             case ROUND_FLOOR:
379                 instant = iRoundingField.roundFloor(instant);
380                 break;
381             case ROUND_CEILING:
382                 instant = iRoundingField.roundCeiling(instant);
383                 break;
384             case ROUND_HALF_FLOOR:
385                 instant = iRoundingField.roundHalfFloor(instant);
386                 break;
387             case ROUND_HALF_CEILING:
388                 instant = iRoundingField.roundHalfCeiling(instant);
389                 break;
390             case ROUND_HALF_EVEN:
391                 instant = iRoundingField.roundHalfEven(instant);
392                 break;
393         }
394 
395         super.setMillis(instant);
396     }

492     /**
493      * Sets the time zone of the datetime, changing the chronology and field values.
494      * <p>
495      * Changing the zone using this method retains the millisecond instant.
496      * The millisecond instant is adjusted in the new zone to compensate.
497      * 
498      * chronology. Setting the time zone does not affect the millisecond value
499      * of this instant.
500      * <p>
501      * If the chronology already has this time zone, no change occurs.
502      *
503      * @param newZone  the time zone to use, null means default zone
504      * @see #setZoneRetainFields
505      */
506     public void setZone(DateTimeZone newZone) {
507         newZone = DateTimeUtils.getZone(newZone);
508         Chronology chrono = getChronology();
509         if (chrono.getZone() != newZone) {
510             setChronology(chrono.withZone(newZone));  // set via this class not super
511 
512         }
513     }
514 
515     /**
516      * Sets the time zone of the datetime, changing the chronology and millisecond.
517      * <p>
518      * Changing the zone using this method retains the field values.
519      * The millisecond instant is adjusted in the new zone to compensate.
520      * <p>
521      * If the chronology already has this time zone, no change occurs.
522      *
523      * @param newZone  the time zone to use, null means default zone
524      * @see #setZone
525      */
526     public void setZoneRetainFields(DateTimeZone newZone) {
527         newZone = DateTimeUtils.getZone(newZone);
528         DateTimeZone originalZone = DateTimeUtils.getZone(getZone());
529         if (newZone == originalZone) {
530             return;
531         }
532 
533         long millis = originalZone.getMillisKeepLocal(newZone, getMillis());
534         setChronology(getChronology().withZone(newZone));  // set via this class not super
535 
536         setMillis(millis);
537     }
