16  package org.goda.time.convert;

18  /**
19   * ConverterManager controls the date and time converters.
20   * <p>
21   * This class enables additional conversion classes to be added via
22   * {@link #addInstantConverter(InstantConverter)}, which may replace an
23   * existing converter. Similar methods exist for duration, time period and
24   * interval converters.
25   * <p>
26   * This class is threadsafe, so adding/removing converters can be done at any
27   * time. Updating the set of convertors is relatively expensive, and so should
28   * not be performed often.
29   * <p>
30   * The default instant converters are:
31   * <ul>
32   * <li>ReadableInstant
33   * <li>String
34   * <li>Calendar
35   * <li>Date (includes sql package subclasses)
36   * <li>Long (milliseconds)
37   * <li>null (now)
38   * </ul>
39   * 
40   * The default partial converters are:
41   * <ul>
42   * <li>ReadablePartial
43   * <li>ReadableInstant
44   * <li>String
45   * <li>Calendar
46   * <li>Date (includes sql package subclasses)
47   * <li>Long (milliseconds)
48   * <li>null (now)
49   * </ul>
50   * 
51   * The default duration converters are:
52   * <ul>
53   * <li>ReadableDuration
54   * <li>ReadableInterval
55   * <li>String
56   * <li>Long (milliseconds)
57   * <li>null (zero ms)
58   * </ul>
59   *
60   * The default time period converters are:
61   * <ul>
62   * <li>ReadablePeriod
63   * <li>ReadableInterval
64   * <li>String
65   * <li>null (zero)
66   * </ul>
67   * 
68   * The default interval converters are:
69   * <ul>
70   * <li>ReadableInterval
71   * <li>String
72   * <li>null (zero-length from now to now)
73   * </ul>
74   *
75   * @author Stephen Colebourne
76   * @author Brian S O'Neill
77   * @since 1.0
78   */
79  public final class ConverterManager {

145     //-----------------------------------------------------------------------
146     /**
147      * Gets the best converter for the object specified.
148      * 
149      * @param object  the object to convert
150      * @return the converter to use
151      * @throws IllegalArgumentException if no suitable converter
152      * @throws IllegalStateException if multiple converters match the type
153      * equally well
154      */
155     public InstantConverter getInstantConverter(Object object) {
156         InstantConverter converter =
157             (InstantConverter)iInstantConverters.select(object == null ? null : object.getClass());
158         if (converter != null) {
159             return converter;
160         }
161         throw new IllegalArgumentException("No instant converter found for type: " +
162             (object == null ? "null" : object.getClass().getName()));
163     }

178     /**
179      * Adds a converter to the set of converters. If a matching converter is
180      * already in the set, the given converter replaces it. If the converter is
181      * exactly the same as one already in the set, no changes are made.
182      * <p>
183      * The order in which converters are added is not relevent. The best
184      * converter is selected by examining the object hierarchy.
185      * 
186      * @param converter  the converter to add, null ignored
187      * @return replaced converter, or null
188      */
189     public InstantConverter addInstantConverter(InstantConverter converter){
190          //   throws SecurityException {
191         
192         //checkAlterInstantConverters();
193         if (converter == null) {
194             return null;
195         }
196         InstantConverter[] removed = new InstantConverter[1];
197         iInstantConverters = iInstantConverters.add(converter, removed);
198         return removed[0];
199     }
