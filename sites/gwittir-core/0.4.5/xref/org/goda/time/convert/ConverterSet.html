18  /**
19   * A set of converters, which allows exact converters to be quickly
20   * selected. This class is threadsafe because it is (essentially) immutable.
21   *
22   * @author Brian S O'Neill
23   * @since 1.0
24   */
25  class ConverterSet {
26      private final Converter[] iConverters;
27  
28      // A simple immutable hashtable: closed hashing, linear probing, sized
29      // power of 2, at least one null slot.
30      private Entry[] iSelectEntries;
31  
32      ConverterSet(Converter[] converters) {
33          // Since this is a package private constructor, we trust ourselves not
34          // to alter the array outside this class.
35          iConverters = converters;
36          iSelectEntries = new Entry[1 << 4]; // 16
37      }
38  
39      /**
40       * Returns the closest matching converter for the given type, or null if
41       * none found.
42       *
43       * @param type type to select, which may be null
44       * @throws IllegalStateException if multiple converters match the type
45       * equally well
46       */
47      Converter select(Class type) throws IllegalStateException {
48          // Check the hashtable first.
49          Entry[] entries = iSelectEntries;
50          int length = entries.length;
51          int index = type == null ? 0 : type.hashCode() & (length - 1);
52  
53          Entry e;
54          // This loop depends on there being at least one null slot.
55          while ((e = entries[index]) != null) {
56              if (e.iType == type) {
57                  return e.iConverter;
58              }
59              if (++index >= length) {
60                  index = 0;
61              }
62          }
63  
64          // Not found in the hashtable, so do actual work.
65  
66          Converter converter = selectSlow(this, type);
67          e = new Entry(type, converter);
68  
69          // Save the entry for future selects. This class must be threadsafe,
70          // but there is no synchronization. Since the hashtable is being used
71          // as a cache, it is okay to destroy existing entries. This isn't
72          // likely to occur unless there is a high amount of concurrency. As
73          // time goes on, cache updates will occur less often, and the cache
74          // will fill with all the necessary entries.
75  
76          // Do all updates on a copy: slots in iSelectEntries must not be
77          // updated by multiple threads as this can allow all null slots to be
78          // consumed.
79          Entry[] newEnts = new Entry[entries.length];
80          for(int i=0; i < entries.length; i++ ){
81              newEnts[i] = entries[i];
82          }
83          entries = newEnts;
84  
85          // Add new entry.
86          entries[index] = e;
87  
88          // Verify that at least one null slot exists!
89          for (int i=0; i<length; i++) {
90              if (entries[i] == null) {
91                  // Found a null slot, swap in new hashtable.
92                  iSelectEntries = entries;
93                  return converter;
94              }
95          }
96  
97          // Double capacity and re-hash.
98  
99          int newLength = length << 1;
100         Entry[] newEntries = new Entry[newLength];
101         for (int i=0; i<length; i++) {
102             e = entries[i];
103             type = e.iType;
104             index = type == null ? 0 : type.hashCode() & (newLength - 1);
105             while (newEntries[index] != null) {
106                 if (++index >= newLength) {
107                     index = 0;
108                 }
109             }
110             newEntries[index] = e;
111         }
112 
113         // Swap in new hashtable.
114         iSelectEntries = newEntries;
115         return converter;
116     }

125     /**
126      * Copies all the converters in the set to the given array.
127      */
128     void copyInto(Converter[] converters) {
129         System.arraycopy(iConverters, 0, converters, 0, iConverters.length);

132     /**
133      * Returns a copy of this set, with the given converter added. If a
134      * matching converter is already in the set, the given converter replaces
135      * it. If the converter is exactly the same as one already in the set, the
136      * original set is returned.
137      *
138      * @param converter  converter to add, must not be null
139      * @param removed  if not null, element 0 is set to the removed converter
140      * @throws NullPointerException if converter is null
141      */
142     ConverterSet add(Converter converter, Converter[] removed) {
143         Converter[] converters = iConverters;
144         int length = converters.length;
145 
146         for (int i=0; i<length; i++) {
147             Converter existing = converters[i];
148             if (converter.equals(existing)) {
149                 // Already in the set.
150                 if (removed != null) {
151                     removed[0] = null;
152                 }
153                 return this;
154             }
155             
156             if (converter.getSupportedType() == existing.getSupportedType()) {
157                 // Replace the converter.
158                 Converter[] copy = new Converter[length];
159                     
160                 for (int j=0; j<length; j++) {
161                     if (j != i) {
162                         copy[j] = converters[j];
163                     } else {
164                         copy[j] = converter;
165                     }
166                 }
167 
168                 if (removed != null) {
169                     removed[0] = existing;
170                 }
171                 return new ConverterSet(copy);
172             }
173         }
174 
175         // Not found, so add it.
176         Converter[] copy = new Converter[length + 1];
177         System.arraycopy(converters, 0, copy, 0, length);
178         copy[length] = converter;
179         
180         if (removed != null) {
181             removed[0] = null;
182         }
183         return new ConverterSet(copy);
184     }

186     /**
187      * Returns a copy of this set, with the given converter removed. If the
188      * converter was not in the set, the original set is returned.
189      *
190      * @param converter  converter to remove, must not be null
191      * @param removed  if not null, element 0 is set to the removed converter
192      * @throws NullPointerException if converter is null
193      */
194     ConverterSet remove(Converter converter, Converter[] removed) {
195         Converter[] converters = iConverters;
196         int length = converters.length;
197 
198         for (int i=0; i<length; i++) {
199             if (converter.equals(converters[i])) {
200                 return remove(i, removed);
201             }
202         }
203 
204         // Not found.
205         if (removed != null) {
206             removed[0] = null;
207         }
208         return this;
209     }

211     /**
212      * Returns a copy of this set, with the converter at the given index
213      * removed.
214      *
215      * @param index index of converter to remove
216      * @param removed if not null, element 0 is set to the removed converter
217      * @throws IndexOutOfBoundsException if the index is invalid
218      */
219     ConverterSet remove(final int index, Converter[] removed) {
220         Converter[] converters = iConverters;
221         int length = converters.length;
222         if (index >= length) {
223             throw new IndexOutOfBoundsException();
224         }
225 
226         if (removed != null) {
227             removed[0] = converters[index];
228         }
229 
230         Converter[] copy = new Converter[length - 1];
231                 
232         int j = 0;
233         for (int i=0; i<length; i++) {
234             if (i != index) {
235                 copy[j++] = converters[i];
236             }
237         }
238         
239         return new ConverterSet(copy);
240     }

242     /**
243      * Returns the closest matching converter for the given type, but not very
244      * efficiently.
245      */
246     private static Converter selectSlow(ConverterSet set, Class type) {
247         Converter[] converters = set.iConverters;
248         int length = converters.length;
249         Converter converter;
250 
251         for (int i=length; --i>=0; ) {
252             converter = converters[i];
253             Class supportedType = converter.getSupportedType();
254 
255             if (supportedType == type) {
256                 // Exact match.
257                 return converter;
258             }
259             // BOGus change here TODO
260             if (supportedType == null || (type != null && supportedType != type)) {
261                 // Eliminate the impossible.
262                 set = set.remove(i, null);
263                 converters = set.iConverters;
264                 length = converters.length;
265             }
266         }
267 
268         // Haven't found exact match, so check what remains in the set.
269 
270         if (type == null || length == 0) {
271             return null;
272         }
273         if (length == 1) {
274             // Found the one best match.
275             return converters[0];
276         }
277 
278         // At this point, there exist multiple potential converters.
279 
280         // Eliminate supertypes.
281         for (int i=length; --i>=0; ) {
282             converter = converters[i];
283             Class supportedType = converter.getSupportedType();
284             for (int j=length; --j>=0; ) {
285                 
286                 //Bogus change here TODO
287                 if (j != i && converters[j].getSupportedType() == supportedType ) {
288                     // Eliminate supertype.
289                     set = set.remove(j, null);
290                     converters = set.iConverters;
291                     length = converters.length;
292                     i = length - 1;
293                 }
294             }
295         }        
296         
297         // Check what remains in the set.
298 
299         if (length == 1) {
300             // Found the one best match.
301             return converters[0];
302         }
303 
304         // Class c implements a, b {}
305         // Converters exist only for a and b. Which is better? Neither.
306 
307         StringBuffer msg = new StringBuffer();
308         msg.append("Unable to find best converter for type \"");
309         msg.append(type.getName());
310         msg.append("\" from remaining set: ");
311         for (int i=0; i<length; i++) {
312             converter = converters[i];
313             Class supportedType = converter.getSupportedType();
314 
315             msg.append(converter.getClass().getName());
316             msg.append('[');
317             msg.append(supportedType == null ? null : supportedType.getName());
318             msg.append("], ");
319         }
320 
321         throw new IllegalStateException(msg.toString());
322     }
