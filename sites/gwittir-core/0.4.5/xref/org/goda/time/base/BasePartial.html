31  /**
32   * BasePartial is an abstract implementation of ReadablePartial that stores
33   * data in array and <code>Chronology</code> fields.
34   * <p>
35   * This class should generally not be used directly by API users.
36   * The {@link org.joda.time.ReadablePartial} interface should be used when different 
37   * kinds of partial objects are to be referenced.
38   * <p>
39   * BasePartial subclasses may be mutable and not thread-safe.
40   *
41   * @author Stephen Colebourne
42   * @since 1.0
43   */
44  public abstract class BasePartial
45          extends AbstractPartial
46          implements ReadablePartial, Serializable {
47  
48      /** Serialization version */
49      private static final long serialVersionUID = 2353678632973660L;
50  
51      /** The chronology in use */
52      private Chronology iChronology;
53      /** The values of each field in this partial */
54      private int[] iValues;
55  
56      //-----------------------------------------------------------------------
57      /**
58       * Constructs a partial with the current time, using ISOChronology in
59       * the default zone to extract the fields.
60       * <p>
61       * The constructor uses the default time zone, resulting in the local time
62       * being initialised. Once the constructor is complete, all further calculations
63       * are performed without reference to a timezone (by switching to UTC).
64       */
65      protected BasePartial() {
66          this(DateTimeUtils.currentTimeMillis(), null);
67      }
68  
69      /**
70       * Constructs a partial with the current time, using the specified chronology
71       * and zone to extract the fields.
72       * <p>
73       * The constructor uses the time zone of the chronology specified.
74       * Once the constructor is complete, all further calculations are performed
75       * without reference to a timezone (by switching to UTC).
76       *
77       * @param chronology  the chronology, null means ISOChronology in the default zone
78       */
79      protected BasePartial(Chronology chronology) {
80          this(DateTimeUtils.currentTimeMillis(), chronology);
81      }
82  
83      /**
84       * Constructs a partial extracting the partial fields from the specified
85       * milliseconds using the ISOChronology in the default zone.
86       * <p>
87       * The constructor uses the default time zone, resulting in the local time
88       * being initialised. Once the constructor is complete, all further calculations
89       * are performed without reference to a timezone (by switching to UTC).
90       *
91       * @param instant  the milliseconds from 1970-01-01T00:00:00Z
92       */
93      protected BasePartial(long instant) {
94          this(instant, null);
95      }
96  
97      /**
98       * Constructs a partial extracting the partial fields from the specified
99       * milliseconds using the chronology provided.
100      * <p>
101      * The constructor uses the time zone of the chronology specified.
102      * Once the constructor is complete, all further calculations are performed
103      * without reference to a timezone (by switching to UTC).
104      *
105      * @param instant  the milliseconds from 1970-01-01T00:00:00Z
106      * @param chronology  the chronology, null means ISOChronology in the default zone
107      */
108     protected BasePartial(long instant, Chronology chronology) {
109         super();
110         chronology = DateTimeUtils.getChronology(chronology);
111         iChronology = chronology.withUTC();
112         iValues = chronology.get(this, instant);
113     }
114 
115     /**
116      * Constructs a partial from an Object that represents a time, using the
117      * specified chronology.
118      * <p>
119      * The recognised object types are defined in
120      * {@link org.joda.time.convert.ConverterManager ConverterManager} and
121      * include ReadableInstant, String, Calendar and Date.
122      * <p>
123      * The constructor uses the time zone of the chronology specified.
124      * Once the constructor is complete, all further calculations are performed
125      * without reference to a timezone (by switching to UTC).
126      *
127      * @param instant  the datetime object
128      * @param chronology  the chronology, null means use converter
129      * @throws IllegalArgumentException if the date is invalid
130      */
131     protected BasePartial(Object instant, Chronology chronology) {
132         super();
133         PartialConverter converter = ConverterManager.getInstance().getPartialConverter(instant);
134         chronology = converter.getChronology(instant, chronology);
135         chronology = DateTimeUtils.getChronology(chronology);
136         iChronology = chronology.withUTC();
137         iValues = converter.getPartialValues(this, instant, chronology);
138     }
139 
140     /**
141      * Constructs a partial from an Object that represents a time, using the
142      * specified chronology.
143      * <p>
144      * The recognised object types are defined in
145      * {@link org.joda.time.convert.ConverterManager ConverterManager} and
146      * include ReadableInstant, String, Calendar and Date.
147      * <p>
148      * The constructor uses the time zone of the chronology specified.
149      * Once the constructor is complete, all further calculations are performed
150      * without reference to a timezone (by switching to UTC).
151      *
152      * @param instant  the datetime object
153      * @param chronology  the chronology, null means use converter
154      * @param parser  if converting from a String, the given parser is preferred
155      * @throws IllegalArgumentException if the date is invalid
156      * @since 1.3
157      */
158     protected BasePartial(Object instant, Chronology chronology, DateTimeFormatter parser) {
159         super();
160         PartialConverter converter = ConverterManager.getInstance().getPartialConverter(instant);
161         chronology = converter.getChronology(instant, chronology);
162         chronology = DateTimeUtils.getChronology(chronology);
163         iChronology = chronology.withUTC();
164         iValues = converter.getPartialValues(this, instant, chronology, parser);
165     }
166 
167     /**
168      * Constructs a partial with specified time field values and chronology.
169      * <p>
170      * The constructor uses the time zone of the chronology specified.
171      * Once the constructor is complete, all further calculations are performed
172      * without reference to a timezone (by switching to UTC).
173      * <p>
174      * The array of values is assigned (not cloned) to the new instance.
175      *
176      * @param values  the new set of values
177      * @param chronology  the chronology, null means ISOChronology in the default zone
178      * @throws IllegalArgumentException if the values are invalid
179      */
180     protected BasePartial(int[] values, Chronology chronology) {
181         super();
182         chronology = DateTimeUtils.getChronology(chronology);
183         iChronology = chronology.withUTC();
184         chronology.validate(this, values);
185         iValues = values;
186     }
187 
188     /**
189      * Private constructor to be used by subclasses only which performs no validation.
190      * <p>
191      * Data is assigned (not cloned) to the new instance.
192      *
193      * @param base  the base partial
194      * @param values  the new set of values, not cloned, null means use base
195      */
196     protected BasePartial(BasePartial base, int[] values) {
197         super();
198         iChronology = base.iChronology;
199         iValues = values;
200     }
201 
202     /**
203      * Private constructor to be used by subclasses only which performs no validation.
204      * <p>
205      * Data is assigned (not cloned) to the new instance.
206      * This should not be used by mutable subclasses.
207      *
208      * @param base  the base partial
209      * @param chrono  the chronology to use, null means use base
210      */
211     protected BasePartial(BasePartial base, Chronology chrono) {
212         super();
213         iChronology = chrono.withUTC();
214         iValues = base.iValues;
215     }
216 
217     //-----------------------------------------------------------------------
218     /**
219      * Gets the value of the field at the specifed index.
220      * 
221      * @param index  the index
222      * @return the value
223      * @throws IndexOutOfBoundsException if the index is invalid
224      */
225     public int getValue(int index) {
226         return iValues[index];
227     }
228 
229     /**
230      * Gets an array of the value of each of the fields that this partial supports.
231      * <p>
232      * The fields are returned largest to smallest, for example Hour, Minute, Second.
233      * Each value corresponds to the same array index as <code>getFields()</code>
234      *
235      * @return the current values of each field (cloned), largest to smallest
236      */
237     public int[] getValues() {
238         int[] ret = new int[iValues.length];
239         ArrayUtils.copyArray(iValues, ret);
240         return ret;
241         
242     }
243 
244     /**
245      * Gets the chronology of the partial which is never null.
246      * <p>
247      * The {@link Chronology} is the calculation engine behind the partial and
248      * provides conversion and validation of the fields in a particular calendar system.
249      * 
250      * @return the chronology, never null
251      */
252     public Chronology getChronology() {
253         return iChronology;
254     }
255 
256     //-----------------------------------------------------------------------
257     /**
258      * Sets the value of the field at the specifed index.
259      * 
260      * @param index  the index
261      * @param value  the value to set
262      * @throws IndexOutOfBoundsException if the index is invalid
263      */
264     protected void setValue(int index, int value) {
265         DateTimeField field = getField(index);
266         iValues = field.set(this, index, iValues, value);
267     }
268 
269     /**
270      * Sets the values of all fields.
271      * 
272      * @param values  the array of values
273      */
274     protected void setValues(int[] values) {
275         getChronology().validate(this, values);
276         iValues = values;
277     }
278 
279     //-----------------------------------------------------------------------
280     /**
281      * Output the date using the specified format pattern.
282      *
283      * @param pattern  the pattern specification, null means use <code>toString</code>
284      * @see org.joda.time.format.DateTimeFormat
285      */
286     public String toString(String pattern) {
287         if (pattern == null) {
288             return toString();
289         }
290         return DateTimeFormat.forPattern(pattern).print(this);
291     }
292 
293     /**
294      * Output the date using the specified format pattern.
295      *
296      * @param pattern  the pattern specification, null means use <code>toString</code>
297      * @param locale  Locale to use, null means default
298      * @see org.joda.time.format.DateTimeFormat
299      */
300     public String toString(String pattern, Locale locale) throws IllegalArgumentException {
301         if (pattern == null) {
302             return toString();
303         }
304         return DateTimeFormat.forPattern(pattern).withLocale(locale).print(this);
305     }
