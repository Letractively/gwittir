80      // bit 1 set: hourOfDay, minuteOfHour, secondOfMinute, and millisOfSecond fields

111     public long getDateTimeMillis(int year, int monthOfYear, int dayOfMonth,
112                                   int millisOfDay)
113         throws IllegalArgumentException
114     {
115         Chronology base;
116         if ((base = iBase) != null && (iBaseFlags & 6) == 6) {
117             // Only call specialized implementation if applicable fields are the same.
118             return base.getDateTimeMillis(year, monthOfYear, dayOfMonth, millisOfDay);
119         }
120         return super.getDateTimeMillis(year, monthOfYear, dayOfMonth, millisOfDay);
121     }
