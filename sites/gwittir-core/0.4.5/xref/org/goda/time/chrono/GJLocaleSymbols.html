33  class GJLocaleSymbols {
34      private static final int FAST_CACHE_SIZE = 64;
35  
36      private static final GJLocaleSymbols[] cFastCache = new GJLocaleSymbols[FAST_CACHE_SIZE];
37  
38      private static HashMap cCache = new HashMap();
39  
40      public static GJLocaleSymbols forLocale(Locale locale) {
41          if (locale == null) {
42              locale = Locale.getDefault();
43          }
44          int index = System.identityHashCode(locale) & (FAST_CACHE_SIZE - 1);
45          GJLocaleSymbols symbols = cFastCache[index];
46          if (symbols != null && symbols.iLocale == locale) {
47              return symbols;
48          }
49          synchronized (cCache) {
50              symbols = (GJLocaleSymbols) cCache.get(locale);
51              if (symbols == null) {
52                  symbols = new GJLocaleSymbols(locale);
53                  cCache.put(locale, symbols);
54              }
55          }
56          cFastCache[index] = symbols;
57          return symbols;
58      }

60      private static String[] realignMonths(String[] months) {
61          String[] a = new String[13];
62          for (int i=1; i<13; i++) {
63              a[i] = months[i - 1];
64          }
65          return a;
66      }
67  
68      private static String[] realignDaysOfWeek(String[] daysOfWeek) {
69          String[] a = new String[8];
70          for (int i=1; i<8; i++) {
71              a[i] = daysOfWeek[(i < 7) ? i + 1 : 1];
72          }
73          return a;
74      }
75  
76      private static void addSymbols(TreeMap map, String[] symbols, Integer[] integers) {
77          for (int i=symbols.length; --i>=0; ) {
78              String symbol = symbols[i];
79              if (symbol != null) {
80                  map.put(symbol, integers[i]);
81              }
82          }
83      }
84  
85      private static void addNumerals(TreeMap map, int start, int end, Integer[] integers) {
86          for (int i=start; i<=end; i++) {
87              map.put(String.valueOf(i).intern(), integers[i]);
88          }
89      }
90  
91      private static int maxLength(String[] a) {
92          int max = 0;
93          for (int i=a.length; --i>=0; ) {
94              String s = a[i];
95              if (s != null) {
96                  int len = s.length();
97                  if (len > max) {
98                      max = len;
99                  }
100             }
101         }
102         return max;
103     }

126     /**
127      * @param locale must not be null
128      */
129     private GJLocaleSymbols(Locale locale) {
130         iLocale = locale;
131         CalendarConstants calcon = (CalendarConstants) GWT.create(CalendarConstants.class);
132         
133         //TODO i18n this
134         iEras = new String[]{"BCE", "CE"};
135         iDaysOfWeek = realignDaysOfWeek(
136                 new String[]{ calcon.sunday(), calcon.monday(), calcon.tuesday(), 
137                 calcon.wednesday(), calcon.thursday(), calcon.friday(),
138                 calcon.saturday()}
139                 );
140         iShortDaysOfWeek = realignDaysOfWeek(
141                 new String[]{ calcon.sunday_short(), calcon.monday_short(), calcon.tuesday_short(),
142                 calcon.wednesday_short(), calcon.thursday_short(), calcon.friday_short(), 
143                 calcon.saturday_short() }
144                 );
145         iMonths = realignMonths( new String[]{
146             calcon.january(), calcon.february(), calcon.march(), calcon.april(),
147             calcon.may(), calcon.june(), calcon.july(), calcon.august(), calcon.september(),
148             calcon.october(), calcon.november(), calcon.december()
149         });
150         iShortMonths = realignMonths(new String[]{
151             calcon.january_short(), calcon.february_short(), calcon.march_short(), calcon.april_short(),
152             calcon.may_short(), calcon.june_short(), calcon.july_short(), calcon.august_short(), 
153             calcon.september_short(),
154             calcon.october_short(), calcon.november_short(), calcon.december_short()
155         });
156         iHalfday = new String[]{ "AM", "PM"};
157 
158         Integer[] integers = new Integer[13];
159         for (int i=0; i<13; i++) {
160             integers[i] = new Integer(i);
161         }
162 
163         iParseEras = new TreeMap(String.CASE_INSENSITIVE_ORDER);
164         addSymbols(iParseEras, iEras, integers);
165         if ("en".equals(locale.getLanguage())) {
166             // Include support for parsing "BCE" and "CE" if the language is
167             // English. At some point Joda-Time will need an independent set of
168             // localized symbols and not depend on java.text.DateFormatSymbols.
169             iParseEras.put("BCE", integers[0]);
170             iParseEras.put("CE", integers[1]);
171         }
172 
173         iParseDaysOfWeek = new TreeMap(String.CASE_INSENSITIVE_ORDER);
174         addSymbols(iParseDaysOfWeek, iDaysOfWeek, integers);
175         addSymbols(iParseDaysOfWeek, iShortDaysOfWeek, integers);
176         addNumerals(iParseDaysOfWeek, 1, 7, integers);
177 
178         iParseMonths = new TreeMap(String.CASE_INSENSITIVE_ORDER);
179         addSymbols(iParseMonths, iMonths, integers);
180         addSymbols(iParseMonths, iShortMonths, integers);
181         addNumerals(iParseMonths, 1, 12, integers);
182 
183         iMaxEraLength = maxLength(iEras);
184         iMaxDayOfWeekLength = maxLength(iDaysOfWeek);
185         iMaxShortDayOfWeekLength = maxLength(iShortDaysOfWeek);
186         iMaxMonthLength = maxLength(iMonths);
187         iMaxShortMonthLength = maxLength(iShortMonths);
188         iMaxHalfdayLength = maxLength(iHalfday);
189     }

196         Integer era = (Integer) iParseEras.get(text);

259     public int halfdayTextToValue(String text) {
260         String[] halfday = iHalfday;
261         for (int i = halfday.length; --i>=0; ) {
262             if (halfday[i].equalsIgnoreCase(text)) {
263                 return i;
264             }
265         }
266         throw new IllegalFieldValueException(DateTimeFieldType.halfdayOfDay(), text);
267     }
