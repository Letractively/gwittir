38  /**
39   * DateTimeZone represents a time zone.
40   * <p>
41   * A time zone is a system of rules to convert time from one geographic 
42   * location to another. For example, Paris, France is one hour ahead of
43   * London, England. Thus when it is 10:00 in London, it is 11:00 in Paris.
44   * <p>
45   * All time zone rules are expressed, for historical reasons, relative to
46   * Greenwich, London. Local time in Greenwich is referred to as Greenwich Mean
47   * Time (GMT).  This is similar, but not precisely identical, to Universal 
48   * Coordinated Time, or UTC. This library only uses the term UTC.
49   * <p>
50   * Using this system, America/Los_Angeles is expressed as UTC-08:00, or UTC-07:00
51   * in the summer. The offset -08:00 indicates that America/Los_Angeles time is
52   * obtained from UTC by adding -08:00, that is, by subtracting 8 hours.
53   * <p>
54   * The offset differs in the summer because of daylight saving time, or DST.
55   * The folowing definitions of time are generally used:
56   * <ul>
57   * <li>UTC - The reference time.
58   * <li>Standard Time - The local time without a daylight saving time offset.
59   * For example, in Paris, standard time is UTC+01:00.
60   * <li>Daylight Saving Time - The local time with a daylight saving time 
61   * offset. This offset is typically one hour, but not always. It is typically
62   * used in most countries away from the equator.  In Paris, daylight saving 
63   * time is UTC+02:00.
64   * <li>Wall Time - This is what a local clock on the wall reads. This will be
65   * either Standard Time or Daylight Saving Time depending on the time of year
66   * and whether the location uses Daylight Saving Time.
67   * </ul>
68   * <p>
69   * Unlike the Java TimeZone class, DateTimeZone is immutable. It also only
70   * supports long format time zone ids. Thus EST and ECT are not accepted.
71   * However, the factory that accepts a TimeZone will attempt to convert from
72   * the old short id to a suitable long id.
73   * <p>
74   * DateTimeZone is thread-safe and immutable, and all subclasses must be as
75   * well.
76   * 
77   * @author Brian S O'Neill
78   * @author Stephen Colebourne
79   * @since 1.0
80   */

161     /**
162      * Gets a time zone instance for the specified time zone id.
163      * <p>
164      * The time zone id may be one of those returned by getAvailableIDs.
165      * Short ids, as accepted by {@link java.util.TimeZone}, are not accepted.
166      * All IDs must be specified in the long format.
167      * The exception is UTC, which is an acceptable id.
168      * <p>
169      * Alternatively a locale independent, fixed offset, datetime zone can
170      * be specified. The form <code>[+-]hh:mm</code> can be used.
171      * 
172      * @param id  the ID of the datetime zone, null means default
173      * @return the DateTimeZone object for the ID
174      * @throws IllegalArgumentException if the ID is not recognised
175      */
176     public static DateTimeZone forID(String id) {
177         if (id == null) {
178             return getDefault();
179         }
180         if (id.equals("UTC")) {
181             return DateTimeZone.UTC;
182         }
183         DateTimeZone zone = cProvider.getZone(id);
184         if (zone != null) {
185             return zone;
186         }
187         if (id.startsWith("+") || id.startsWith("-")) {
188             int offset = parseOffset(id);
189             if (offset == 0L) {
190                 return DateTimeZone.UTC;
191             } else {
192                 id = printOffset(offset);
193                 return fixedOffsetZone(id, offset);
194             }
195         }
196         throw new IllegalArgumentException("The datetime zone id is not recognised: " + id);
197     }

213     /**
214      * Gets a time zone instance for the specified offset to UTC in hours and minutes.
215      * This method assumes 60 minutes in an hour, and standard length minutes.
216      * <p>
217      * This factory is a convenient way of constructing zones with a fixed offset.
218      * The minutes value is always positive and in the range 0 to 59.
219      * If constructed with the values (-2, 30), the resultiong zone is '-02:30'.
220      * 
221      * @param hoursOffset  the offset in hours from UTC
222      * @param minutesOffset  the offset in minutes from UTC, must be between 0 and 59 inclusive
223      * @return the DateTimeZone object for the offset
224      * @throws IllegalArgumentException if the offset or minute is too large or too small
225      */
226     public static DateTimeZone forOffsetHoursMinutes(int hoursOffset, int minutesOffset) throws IllegalArgumentException {
227         if (hoursOffset == 0 && minutesOffset == 0) {
228             return DateTimeZone.UTC;
229         }
230         if (minutesOffset < 0 || minutesOffset > 59) {
231             throw new IllegalArgumentException("Minutes out of range: " + minutesOffset);
232         }
233         int offset = 0;
234         try {
235             int hoursInMinutes = FieldUtils.safeMultiply(hoursOffset, 60);
236             if (hoursInMinutes < 0) {
237                 minutesOffset = FieldUtils.safeAdd(hoursInMinutes, -minutesOffset);
238             } else {
239                 minutesOffset = FieldUtils.safeAdd(hoursInMinutes, minutesOffset);
240             }
241             offset = FieldUtils.safeMultiply(minutesOffset, DateTimeConstants.MILLIS_PER_MINUTE);
242         } catch (ArithmeticException ex) {
243             throw new IllegalArgumentException("Offset is too large");
244         }
245         return forOffsetMillis(offset);
246     }

278 //        }

280         if (id.equals("UTC")) {
281             return DateTimeZone.UTC;

285         DateTimeZone dtz = null;

287         if (convId != null) {

289         }

291             dtz = cProvider.getZone(id);
292         }
293         if (dtz != null) {
294             return dtz;
295         }

298         if (convId == null) {

300             if (convId.startsWith("GMT+") || convId.startsWith("GMT-")) {

302                 int offset = parseOffset(convId);
303                 if (offset == 0L) {
304                     return DateTimeZone.UTC;
305                 } else {
306                     convId = printOffset(offset);
307                     return fixedOffsetZone(convId, offset);
308                 }
309             }

317      * Gets the zone using a fixed offset amount.
318      * 
319      * @param id  the zone id
320      * @param offset  the offset in millis
321      * @return the zone
322      */
323     private static synchronized DateTimeZone fixedOffsetZone(String id, int offset) {
324         if (offset == 0) {
325             return DateTimeZone.UTC;
326         }
327         if (iFixedOffsetCache == null) {
328             iFixedOffsetCache = new HashMap();
329         }
330         DateTimeZone zone = (DateTimeZone) iFixedOffsetCache.get(id);
331         if (zone == null) {
332             zone = new FixedDateTimeZone(id, null, offset, offset);
333         }
334         
335         iFixedOffsetCache.put(id, zone);
336         return zone;
337     }

572     private static int parseOffset(String str) {
573         Chronology chrono;
574         if (cDefault != null) {
575             chrono = ISOChronology.getInstanceUTC();
576         } else {
577             // Can't use a real chronology if called during class
578             // initialization. Offset parser doesn't need it anyhow.
579             chrono = new BaseChronology() {
580                 public DateTimeZone getZone() {
581                     return null;
582                 }
583                 public Chronology withUTC() {
584                     return this;
585                 }
586                 public Chronology withZone(DateTimeZone zone) {
587                     return this;
588                 }
589                 public String toString() {
590                     return getClass().getName();
591                 }
592             };
593         }
594 
595         return -(int) offsetFormatter().withChronology(chrono).parseMillis(str);
596     }

598     /**
599      * Formats a timezone offset string.
600      * <p>
601      * This method is kept separate from the formatting classes to speed and
602      * simplify startup and classloading.
603      * 
604      * @param offset  the offset in milliseconds
605      * @return the time zone string
606      */
607     private static String printOffset(int offset) {
608         StringBuffer buf = new StringBuffer();
609         if (offset >= 0) {
610             buf.append('+');
611         } else {
612             buf.append('-');
613             offset = -offset;
614         }
615 
616         int hours = offset / DateTimeConstants.MILLIS_PER_HOUR;
617         FormatUtils.appendPaddedInteger(buf, hours, 2);
618         offset -= hours * (int) DateTimeConstants.MILLIS_PER_HOUR;
619 
620         int minutes = offset / DateTimeConstants.MILLIS_PER_MINUTE;
621         buf.append(':');
622         FormatUtils.appendPaddedInteger(buf, minutes, 2);
623         offset -= minutes * DateTimeConstants.MILLIS_PER_MINUTE;
624         if (offset == 0) {
625             return buf.toString();
626         }
627 
628         int seconds = offset / DateTimeConstants.MILLIS_PER_SECOND;
629         buf.append(':');
630         FormatUtils.appendPaddedInteger(buf, seconds, 2);
631         offset -= seconds * DateTimeConstants.MILLIS_PER_SECOND;
632         if (offset == 0) {
633             return buf.toString();
634         }
635 
636         buf.append('.');
637         FormatUtils.appendPaddedInteger(buf, offset, 3);
638         return buf.toString();
639     }

823     /**
824      * Gets the millisecond offset to subtract from local time to get UTC time.
825      * This offset can be used to undo adding the offset obtained by getOffset.
826      *
827      * <pre>
828      * millisLocal == millisUTC   + getOffset(millisUTC)
829      * millisUTC   == millisLocal - getOffsetFromLocal(millisLocal)
830      * </pre>
831      *
832      * NOTE: After calculating millisLocal, some error may be introduced. At
833      * offset transitions (due to DST or other historical changes), ranges of
834      * local times may map to different UTC times.
835      * <p>
836      * This method will return an offset suitable for calculating an instant
837      * after any DST gap. For example, consider a zone with a cutover
838      * from 01:00 to 01:59:<br />
839      * Input: 00:00  Output: 00:00<br />
840      * Input: 00:30  Output: 00:30<br />
841      * Input: 01:00  Output: 02:00<br />
842      * Input: 01:30  Output: 02:30<br />
843      * Input: 02:00  Output: 02:00<br />
844      * Input: 02:30  Output: 02:30<br />
845      * <p>
846      * NOTE: The behaviour of this method changed in v1.5, with the emphasis
847      * on returning a consistent result later along the time-line (shown above).
848      *
849      * @param instantLocal  the millisecond instant, relative to this time zone, to
850      * get the offset for
851      * @return the millisecond offset to subtract from local time to get UTC time
852      */
853     public int getOffsetFromLocal(long instantLocal) {
854         // get the offset at instantLocal (first estimate)
855         int offsetLocal = getOffset(instantLocal);
856         // adjust instantLocal using the estimate and recalc the offset
857         int offsetAdjusted = getOffset(instantLocal - offsetLocal);
858         // if the offsets differ, we must be near a DST boundary
859         if (offsetLocal != offsetAdjusted) {
860             // we need to ensure that time is always after the DST gap
861             // this happens naturally for positive offsets, but not for negative
862             if ((offsetLocal - offsetAdjusted) < 0) {
863                 // if we just return offsetAdjusted then the time is pushed
864                 // back before the transition, whereas it should be
865                 // on or after the transition
866                 long nextLocal = nextTransition(instantLocal - offsetLocal);
867                 long nextAdjusted = nextTransition(instantLocal - offsetAdjusted);
868                 if (nextLocal != nextAdjusted) {
869                     return offsetLocal;
870                 }
871             }
872         }
873         return offsetAdjusted;
874     }

877      * Converts a standard UTC instant to a local instant with the same
878      * local time. This conversion is used before performing a calculation
879      * so that the calculation can be done using a simple local zone.
880      *
881      * @param instantUTC  the UTC instant to convert to local
882      * @return the local instant with the same local time
883      * @throws ArithmeticException if the result overflows a long
884      * @since 1.5
885      */
886     public long convertUTCToLocal(long instantUTC) {
887         int offset = getOffset(instantUTC);
888         long instantLocal = instantUTC + offset;
889         // If there is a sign change, but the two values have the same sign...
890         if ((instantUTC ^ instantLocal) < 0 && (instantUTC ^ offset) >= 0) {
891             throw new ArithmeticException("Adding time zone offset caused overflow");
892         }
893         return instantLocal;
894     }

896     /**
897      * Converts a local instant to a standard UTC instant with the same
898      * local time. This conversion is used after performing a calculation
899      * where the calculation was done using a simple local zone.
900      *
901      * @param instantLocal  the local instant to convert to UTC
902      * @param strict  whether the conversion should reject non-existent local times
903      * @return the UTC instant with the same local time, 
904      * @throws ArithmeticException if the result overflows a long
905      * @throws IllegalArgumentException if the zone has no eqivalent local time
906      * @since 1.5
907      */
908     public long convertLocalToUTC(long instantLocal, boolean strict) {
909         // get the offset at instantLocal (first estimate)
910         int offsetLocal = getOffset(instantLocal);
911         // adjust instantLocal using the estimate and recalc the offset
912         int offset = getOffset(instantLocal - offsetLocal);
913         // if the offsets differ, we must be near a DST boundary
914         if (offsetLocal != offset) {
915             // if strict then always check if in DST gap
916             // otherwise only check if zone in Western hemisphere (as the
917             // value of offset is already correct for Eastern hemisphere)
918             if (strict || offsetLocal < 0) {
919                 // determine if we are in the DST gap
920                 long nextLocal = nextTransition(instantLocal - offsetLocal);
921                 long nextAdjusted = nextTransition(instantLocal - offset);
922                 if (nextLocal != nextAdjusted) {
923                     // yes we are in the DST gap
924                     if (strict) {
925                         // DST gap is not acceptable
926                         throw new IllegalArgumentException("Illegal instant due to time zone offset transition: " +
927                                 DateTimeFormat.forPattern("yyyy-MM-dd'T'HH:mm:ss.SSS").print(new Instant(instantLocal)) +
928                                 " (" + getID() + ")");
929                     } else {
930                         // DST gap is acceptable, but for the Western hemisphere
931                         // the offset is wrong and will result in local times
932                         // before the cutover so use the offsetLocal instead
933                         offset = offsetLocal;
934                     }
935                 }
936             }
937         }
938         // check for overflow
939         long instantUTC = instantLocal - offset;
940         // If there is a sign change, but the two values have different signs...
941         if ((instantLocal ^ instantUTC) < 0 && (instantLocal ^ offset) < 0) {
942             throw new ArithmeticException("Subtracting time zone offset caused overflow");
943         }
944         return instantUTC;
945     }
