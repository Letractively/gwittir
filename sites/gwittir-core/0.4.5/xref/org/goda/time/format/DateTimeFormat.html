27  /**
28   * Factory that creates instances of DateTimeFormatter from patterns and styles.
29   * <p>
30   * Datetime formatting is performed by the {@link DateTimeFormatter} class.
31   * Three classes provide factory methods to create formatters, and this is one.
32   * The others are {@link ISODateTimeFormat} and {@link DateTimeFormatterBuilder}.
33   * <p>
34   * This class provides two types of factory:
35   * <ul>
36   * <li>{@link #forPattern(String) Pattern} provides a DateTimeFormatter based on
37   * a pattern string that is mostly compatible with the JDK date patterns.
38   * <li>{@link #forStyle(String) Style} provides a DateTimeFormatter based on a
39   * two character style, representing short, medium, long and full.
40   * </ul>
41   * <p>
42   * For example, to use a patterm:
43   * <pre>
44   * DateTime dt = new DateTime();
45   * DateTimeFormatter fmt = DateTimeFormat.forPattern("MMMM, yyyy");
46   * String str = fmt.print(dt);
47   * </pre>
48   *
49   * The pattern syntax is mostly compatible with java.text.SimpleDateFormat -
50   * time zone names cannot be parsed and a few more symbols are supported.
51   * All ASCII letters are reserved as pattern letters, which are defined as follows:
52   * <blockquote>
53   * <pre>
54   * Symbol  Meaning                      Presentation  Examples
55   * ------  -------                      ------------  -------
56   * G       era                          text          AD
57   * C       century of era (&gt;=0)         number        20
58   * Y       year of era (&gt;=0)            year          1996
59   *
60   * x       weekyear                     year          1996
61   * w       week of weekyear             number        27
62   * e       day of week                  number        2
63   * E       day of week                  text          Tuesday; Tue
64   *
65   * y       year                         year          1996
66   * D       day of year                  number        189
67   * M       month of year                month         July; Jul; 07
68   * d       day of month                 number        10
69   *
70   * a       halfday of day               text          PM
71   * K       hour of halfday (0~11)       number        0
72   * h       clockhour of halfday (1~12)  number        12
73   *
74   * H       hour of day (0~23)           number        0
75   * k       clockhour of day (1~24)      number        24
76   * m       minute of hour               number        30
77   * s       second of minute             number        55
78   * S       fraction of second           number        978
79   *
80   * z       time zone                    text          Pacific Standard Time; PST
81   * Z       time zone offset/id          zone          -0800; -08:00; America/Los_Angeles
82   *
83   * '       escape for text              delimiter
84   * ''      single quote                 literal       '
85   * </pre>
86   * </blockquote>
87   * The count of pattern letters determine the format.
88   * <p>
89   * <strong>Text</strong>: If the number of pattern letters is 4 or more,
90   * the full form is used; otherwise a short or abbreviated form is used if
91   * available.
92   * <p>
93   * <strong>Number</strong>: The minimum number of digits. Shorter numbers
94   * are zero-padded to this amount.
95   * <p>
96   * <strong>Year</strong>: Numeric presentation for year and weekyear fields
97   * are handled specially. For example, if the count of 'y' is 2, the year
98   * will be displayed as the zero-based year of the century, which is two
99   * digits.
100  * <p>
101  * <strong>Month</strong>: 3 or over, use text, otherwise use number.
102  * <p>
103  * <strong>Zone</strong>: 'Z' outputs offset without a colon, 'ZZ' outputs
104  * the offset with a colon, 'ZZZ' or more outputs the zone id.
105  * <p>
106  * <strong>Zone names</strong>: Time zone names ('z') cannot be parsed.
107  * <p>
108  * Any characters in the pattern that are not in the ranges of ['a'..'z']
109  * and ['A'..'Z'] will be treated as quoted text. For instance, characters
110  * like ':', '.', ' ', '#' and '?' will appear in the resulting time text
111  * even they are not embraced within single quotes.
112  * <p>
113  * DateTimeFormat is thread-safe and immutable, and the formatters it returns
114  * are as well.
115  *
116  * @author Brian S O'Neill
117  * @author Maxim Zhao
118  * @since 1.0
119  * @see ISODateTimeFormat
120  * @see DateTimeFormatterBuilder
121  */
122 public class DateTimeFormat {

386     //-----------------------------------------------------------------------
387     /**
388      * Parses the given pattern and appends the rules to the given
389      * DateTimeFormatterBuilder.
390      *
391      * @param pattern  pattern specification
392      * @throws IllegalArgumentException if the pattern is invalid
393      * @see #forPattern
394      */
395     private static void parsePatternTo(DateTimeFormatterBuilder builder, String pattern) {
396         int length = pattern.length();
397         int[] indexRef = new int[1];
398 
399         for (int i=0; i<length; i++) {
400             indexRef[0] = i;
401             String token = parseToken(pattern, indexRef);
402             i = indexRef[0];
403 
404             int tokenLen = token.length();
405             if (tokenLen == 0) {
406                 break;
407             }
408             char c = token.charAt(0);
409 
410             switch (c) {
411             case 'G': // era designator (text)
412                 builder.appendEraText();
413                 break;
414             case 'C': // century of era (number)
415                 builder.appendCenturyOfEra(tokenLen, tokenLen);
416                 break;
417             case 'x': // weekyear (number)
418             case 'y': // year (number)
419             case 'Y': // year of era (number)
420                 if (tokenLen == 2) {
421                     boolean lenientParse = true;
422 
423                     // Peek ahead to next token.
424                     if (i + 1 < length) {
425                         indexRef[0]++;
426                         if (isNumericToken(parseToken(pattern, indexRef))) {
427                             // If next token is a number, cannot support
428                             // lenient parse, because it will consume digits
429                             // that it should not.
430                             lenientParse = false;
431                         }
432                         indexRef[0]--;
433                     }
434 
435                     // Use pivots which are compatible with SimpleDateFormat.
436                     switch (c) {
437                     case 'x':
438                         builder.appendTwoDigitWeekyear
439                             (new DateTime().getWeekyear() - 30, lenientParse);
440                         break;
441                     case 'y':
442                     case 'Y':
443                     default:
444                         builder.appendTwoDigitYear(new DateTime().getYear() - 30, lenientParse);
445                         break;
446                     }
447                 } else {
448                     // Try to support long year values.
449                     int maxDigits = 9;
450 
451                     // Peek ahead to next token.
452                     if (i + 1 < length) {
453                         indexRef[0]++;
454                         if (isNumericToken(parseToken(pattern, indexRef))) {
455                             // If next token is a number, cannot support long years.
456                             maxDigits = tokenLen;
457                         }
458                         indexRef[0]--;
459                     }
460 
461                     switch (c) {
462                     case 'x':
463                         builder.appendWeekyear(tokenLen, maxDigits);
464                         break;
465                     case 'y':
466                         builder.appendYear(tokenLen, maxDigits);
467                         break;
468                     case 'Y':
469                         builder.appendYearOfEra(tokenLen, maxDigits);
470                         break;
471                     }
472                 }
473                 break;
474             case 'M': // month of year (text and number)
475                 if (tokenLen >= 3) {
476                     if (tokenLen >= 4) {
477                         builder.appendMonthOfYearText();
478                     } else {
479                         builder.appendMonthOfYearShortText();
480                     }
481                 } else {
482                     builder.appendMonthOfYear(tokenLen);
483                 }
484                 break;
485             case 'd': // day of month (number)
486                 builder.appendDayOfMonth(tokenLen);
487                 break;
488             case 'a': // am/pm marker (text)
489                 builder.appendHalfdayOfDayText();
490                 break;
491             case 'h': // clockhour of halfday (number, 1..12)
492                 builder.appendClockhourOfHalfday(tokenLen);
493                 break;
494             case 'H': // hour of day (number, 0..23)
495                 builder.appendHourOfDay(tokenLen);
496                 break;
497             case 'k': // clockhour of day (1..24)
498                 builder.appendClockhourOfDay(tokenLen);
499                 break;
500             case 'K': // hour of halfday (0..11)
501                 builder.appendHourOfHalfday(tokenLen);
502                 break;
503             case 'm': // minute of hour (number)
504                 builder.appendMinuteOfHour(tokenLen);
505                 break;
506             case 's': // second of minute (number)
507                 builder.appendSecondOfMinute(tokenLen);
508                 break;
509             case 'S': // fraction of second (number)
510                 builder.appendFractionOfSecond(tokenLen, tokenLen);
511                 break;
512             case 'e': // day of week (number)
513                 builder.appendDayOfWeek(tokenLen);
514                 break;
515             case 'E': // dayOfWeek (text)
516                 if (tokenLen >= 4) {
517                     builder.appendDayOfWeekText();
518                 } else {
519                     builder.appendDayOfWeekShortText();
520                 }
521                 break;
522             case 'D': // day of year (number)
523                 builder.appendDayOfYear(tokenLen);
524                 break;
525             case 'w': // week of weekyear (number)
526                 builder.appendWeekOfWeekyear(tokenLen);
527                 break;
528             case 'z': // time zone (text)
529                 if (tokenLen >= 4) {
530                     builder.appendTimeZoneName();
531                 } else {
532                     builder.appendTimeZoneShortName();
533                 }
534                 break;
535             case 'Z': // time zone offset
536                 if (tokenLen == 1) {
537                     builder.appendTimeZoneOffset(null, false, 2, 2);
538                 } else if (tokenLen == 2) {
539                     builder.appendTimeZoneOffset(null, true, 2, 2);
540                 } else {
541                     builder.appendTimeZoneId();
542                 }
543                 break;
544             case '\'': // literal text
545                 String sub = token.substring(1);
546                 if (sub.length() == 1) {
547                     builder.appendLiteral(sub.charAt(0));
548                 } else {
549                     // Create copy of sub since otherwise the temporary quoted
550                     // string would still be referenced internally.
551                     builder.appendLiteral(new String(sub));
552                 }
553                 break;
554             default:
555                 throw new IllegalArgumentException
556                     ("Illegal pattern component: " + token);
557             }
558         }
559     }

561     /**
562      * Parses an individual token.
563      * 
564      * @param pattern  the pattern string
565      * @param indexRef  a single element array, where the input is the start
566      *  location and the output is the location after parsing the token
567      * @return the parsed token
568      */
569     private static String parseToken(String pattern, int[] indexRef) {
570         StringBuffer buf = new StringBuffer();
571 
572         int i = indexRef[0];
573         int length = pattern.length();
574 
575         char c = pattern.charAt(i);
576         if (c >= 'A' && c <= 'Z' || c >= 'a' && c <= 'z') {
577             // Scan a run of the same character, which indicates a time
578             // pattern.
579             buf.append(c);
580 
581             while (i + 1 < length) {
582                 char peek = pattern.charAt(i + 1);
583                 if (peek == c) {
584                     buf.append(c);
585                     i++;
586                 } else {
587                     break;
588                 }
589             }
590         } else {
591             // This will identify token as text.
592             buf.append('\'');
593 
594             boolean inLiteral = false;
595 
596             for (; i < length; i++) {
597                 c = pattern.charAt(i);
598                 
599                 if (c == '\'') {
600                     if (i + 1 < length && pattern.charAt(i + 1) == '\'') {
601                         // '' is treated as escaped '
602                         i++;
603                         buf.append(c);
604                     } else {
605                         inLiteral = !inLiteral;
606                     }
607                 } else if (!inLiteral &&
608                            (c >= 'A' && c <= 'Z' || c >= 'a' && c <= 'z')) {
609                     i--;
610                     break;
611                 } else {
612                     buf.append(c);
613                 }
614             }
615         }
616 
617         indexRef[0] = i;
618         return buf.toString();
619     }

621     /**
622      * Returns true if token should be parsed as a numeric field.
623      * 
624      * @param token  the token to parse
625      * @return true if numeric field
626      */
627     private static boolean isNumericToken(String token) {
628         int tokenLen = token.length();
629         if (tokenLen > 0) {
630             char c = token.charAt(0);
631             switch (c) {
632             case 'c': // century (number)
633             case 'C': // century of era (number)
634             case 'x': // weekyear (number)
635             case 'y': // year (number)
636             case 'Y': // year of era (number)
637             case 'd': // day of month (number)
638             case 'h': // hour of day (number, 1..12)
639             case 'H': // hour of day (number, 0..23)
640             case 'm': // minute of hour (number)
641             case 's': // second of minute (number)
642             case 'S': // fraction of second (number)
643             case 'e': // day of week (number)
644             case 'D': // day of year (number)
645             case 'F': // day of week in month (number)
646             case 'w': // week of year (number)
647             case 'W': // week of month (number)
648             case 'k': // hour of day (1..24)
649             case 'K': // hour of day (0..11)
650                 return true;
651             case 'M': // month of year (text and number)
652                 if (tokenLen <= 2) {
653                     return true;
654                 }
655             }
656         }
657             
658         return false;
659     }

662     /**
663      * Select a format from a custom pattern.
664      *
665      * @param pattern  pattern specification
666      * @throws IllegalArgumentException if the pattern is invalid
667      * @see #appendPatternTo
668      */
669     private static DateTimeFormatter createFormatterForPattern(String pattern) {
670         if (pattern == null || pattern.length() == 0) {
671             throw new IllegalArgumentException("Invalid pattern specification");
672         }
673         DateTimeFormatter formatter = null;
674         synchronized (cPatternedCache) {
675             formatter = (DateTimeFormatter) cPatternedCache.get(pattern);
676             if (formatter == null) {
677                 DateTimeFormatterBuilder builder = new DateTimeFormatterBuilder();
678                 parsePatternTo(builder, pattern);
679                 formatter = builder.toFormatter();
680 
681                 cPatternedCache.put(pattern, formatter);
682             }
683         }
684         return formatter;
685     }

687     /**
688      * Select a format from a two character style pattern. The first character
689      * is the date style, and the second character is the time style. Specify a
690      * character of 'S' for short style, 'M' for medium, 'L' for long, and 'F'
691      * for full. A date or time may be ommitted by specifying a style character '-'.
692      *
693      * @param style  two characters from the set {"S", "M", "L", "F", "-"}
694      * @throws IllegalArgumentException if the style is invalid
695      */
696     private static DateTimeFormatter createFormatterForStyle(String style) {
697         if (style == null || style.length() != 2) {
698             throw new IllegalArgumentException("Invalid style specification: " + style);
699         }
700         int dateStyle = selectStyle(style.charAt(0));
701         int timeStyle = selectStyle(style.charAt(1));
702         if (dateStyle == NONE && timeStyle == NONE) {
703             throw new IllegalArgumentException("Style '--' is invalid");
704         }
705         return createFormatterForStyleIndex(dateStyle, timeStyle);
706     }

712      * @param timeStyle  the time style

715     private static DateTimeFormatter createFormatterForStyleIndex(int dateStyle, int timeStyle) {
716         int index = ((dateStyle << 2) + dateStyle) + timeStyle;
717         DateTimeFormatter f = null;
718         synchronized (cStyleCache) {
719             f = cStyleCache[index];
720             if (f == null) {
721                 int type = DATETIME;
722                 if (dateStyle == NONE) {
723                     type = TIME;
724                 } else if (timeStyle == NONE) {
725                     type = DATE;
726                 }

728                         dateStyle, timeStyle, type);
729                 f = new DateTimeFormatter(llf, llf);
730                 cStyleCache[index] = f;
731             }
732         }
733         return f;
734     }

813         private DateTimeFormatter getFormatter(Locale locale) {
