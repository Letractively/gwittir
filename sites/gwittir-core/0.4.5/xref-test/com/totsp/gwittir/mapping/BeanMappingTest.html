43      public void testConvert() throws Exception {
44          System.out.println("convert");
45          
46          Properties mappings = new Properties();
47          mappings.setProperty( "test.source.*", "test.destination.*");
48          
49          test.source.ABean source = new test.source.ABean();
50          source.setStringProperty( "Foo");
51          source.setIntProperty( -2 );
52          String[] arr = { "Foo", "Bar", "Baz" };
53          source.setArrayProperty(arr);
54          Random r = new Random();
55          test.source.BBean[][] bbeans = new test.source.BBean[3][5];
56          for( int i=0; i < bbeans.length; i++ ){
57              for( int j=0; j < bbeans[i].length; j++){
58                  bbeans[i][j] = new test.source.BBean();
59                  bbeans[i][j].setValue( ""+ r.nextLong() );
60              }
61          }
62          
63          HashMap<String,  test.source.BBean> map = new HashMap<String,  test.source.BBean>();
64          map.put( "Foo", bbeans[0][0] );
65          map.put( "Bar", bbeans[0][1] );
66          source.setBeanMap( map );
67          bbeans[0][0].setParent( source );
68          
69          source.setBeanArray( bbeans );
70          test.destination.ABean result = (test.destination.ABean) 
71              BeanMapping.convert(mappings, source);
72          
73          assertEquals( "stringProperty", source.getStringProperty(), result.stringProperty);
74          assertEquals( "intProperty", source.getIntProperty(), result.intProperty);
75          BeanMappingTest.assertEquals( "arrayProperty", source.getArrayProperty()[0], result.arrayProperty[0]);
76          BeanMappingTest.assertEquals( "arrayProperty", source.getArrayProperty()[1], result.arrayProperty[1]);
77          BeanMappingTest.assertEquals( "arrayProperty", source.getArrayProperty()[2], result.arrayProperty[2]);
78          for( int i=0; i < bbeans.length; i++ ){
79              for( int j=0; j < bbeans[i].length; j++){
80                  System.out.println(result.beanArray[i][j].getValue() );
81                  BeanMappingTest.assertEquals("arrayOfBeans", bbeans[i][j].getValue(),
82                          result.beanArray[i][j].getValue());
83              }
84          }
85          BeanMappingTest.assertEquals("beanMap", map.get("Foo").getValue(), result.beanMap.get("Foo").getValue() );
86          BeanMappingTest.assertEquals("beanMap", map.get("Bar").getValue(), result.beanMap.get("Bar").getValue() );
87          BeanMappingTest.assertTrue( "cyclic", result.beanMap.get("Foo").getParent() == result );
88      }
