24      public static String MD5(String text)
25          throws NoSuchAlgorithmException, UnsupportedEncodingException {
26          MessageDigest md;
27          md = MessageDigest.getInstance("MD5");
28  
29          byte[] md5hash = new byte[16];
30          md.update(text.getBytes("UTF-8"), 0, text.length());
31          md5hash = md.digest();
32  
33          return toHexString(md5hash);
34      }

36      public void testMD5() throws Exception {
37          MD5 md5 = new MD5("This is a test.");
38          String client = md5.asHex();

40          assertEquals(server, client);
41      }

43      private static String toHexString(byte[] data) {
44          StringBuffer buf = new StringBuffer();
45  
46          for (int i = 0; i < data.length; i++) {
47              int halfbyte = (data[i] >>> 4) & 0x0F;
48              int two_halfs = 0;
49  
50              do {
51                  if ((0 <= halfbyte) && (halfbyte <= 9)) {
52                      buf.append((char) ('0' + halfbyte));
53                  } else {
54                      buf.append((char) ('a' + (halfbyte - 10)));
55                  }
56  
57                  halfbyte = data[i] & 0x0F;
58              } while (two_halfs++ < 1);
59          }
60  
61          return buf.toString();
62      }
